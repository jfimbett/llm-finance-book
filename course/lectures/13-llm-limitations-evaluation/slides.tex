\section{Calibration and Overconfidence}
% ===============================================================

\begin{frame}{Why Calibration Matters in Finance}
\begin{bigpicture}
A language model that answers with apparent certainty is not necessarily
correct. Its expressed confidence---a stated probability, a hedging phrase, or
a fluent rationale---may diverge sharply from its empirical accuracy. In
finance, a confidently wrong forecast cascades through portfolios, loan books,
and regulatory submissions.
\end{bigpicture}

\smallskip
\textbf{The calibration problem:} does ``80\% confident'' actually mean ``right
80\% of the time''?

\smallskip
\begin{block}{Five fragilities of LLMs in finance (this lecture)}
\begin{enumerate}\itemsep1pt
  \item \textbf{Calibration} --- systematic overconfidence
  \item \textbf{Temporal leakage} --- memorising the future
  \item \textbf{Stock prediction} --- statistical vs.\ economic value
  \item \textbf{Evaluation} --- beyond classification accuracy
  \item \textbf{Hallucination} --- fabricated facts with no warning signal
\end{enumerate}
\end{block}
\end{frame}

\begin{frame}{What Calibration Means: ECE and MCE}
\textbf{Perfect calibration:} among predictions with confidence $p$, exactly a
fraction $p$ are correct,
\[
  \Pr(Y = 1 \mid \hat{p} = p) = p \quad \text{for all } p \in [0,1].
\]

\begin{underhood}
Bin predictions into $M$ equal-width intervals $B_1,\dots,B_M$ (typically
$M=10$). Plot mean confidence $\overline{p}(B_m)$ vs.\ accuracy
$\text{acc}(B_m)$ --- the \emph{reliability diagram}. Aggregate the gaps:
\[
  \text{ECE} = \sum_{m=1}^{M} \frac{|B_m|}{n}\,
  \bigl|\,\text{acc}(B_m) - \overline{p}(B_m)\,\bigr|,
  \qquad
  \text{MCE} = \max_m \bigl|\,\text{acc}(B_m) - \overline{p}(B_m)\,\bigr|.
\]
ECE is a weighted average gap; MCE reports the worst-case bin.
\end{underhood}

\medskip
\textbf{Why MCE matters:} a single catastrophic region---say 90--100\%
confidence predictions that are only 40\% accurate---is unacceptable in
high-stakes finance even if the \emph{average} looks fine.
\end{frame}

\begin{frame}{Worked Example: ECE of an LLM Credit-Risk Classifier}
1{,}000 default predictions, 5 equal-width bins of width 0.2:

\medskip
\centering
\begin{tabular}{ccrcc}
\toprule
Bin & Interval & $|B_m|$ & $\overline{p}(B_m)$ & $\text{acc}(B_m)$ \\
\midrule
1 & [0.0, 0.2) & 120 & 0.08 & 0.11 \\
2 & [0.2, 0.4) & 180 & 0.31 & 0.28 \\
3 & [0.4, 0.6) & 200 & 0.50 & 0.47 \\
4 & [0.6, 0.8) & 300 & 0.71 & 0.55 \\
5 & [0.8, 1.0] & 200 & 0.88 & 0.62 \\
\bottomrule
\end{tabular}

\medskip
\raggedright
\[
  \text{ECE} = 0.12(0.03) + 0.18(0.03) + 0.20(0.03) + 0.30(0.16) + 0.20(0.26)
  = \mathbf{0.115}.
\]

\begin{alertblock}{Reading the number}
An ECE of $11.5\%$ is substantial. Well calibrated at low confidence, but
\textbf{severely overconfident at the top}: 80--100\% confidence predictions
are right only \textbf{62\%} of the time. A credit officer triaging by stated
confidence would misallocate effort toward the least reliable cases.
\end{alertblock}
\end{frame}

\begin{frame}{Why LLMs Are Systematically Overconfident}
\textbf{Three reinforcing mechanisms:}

\medskip
\begin{enumerate}
  \item \textbf{RLHF-induced overconfidence.} Human raters prefer confident,
        fluent answers over hedged ones---even when hedging is more
        appropriate. The reward signal pushes toward confident expression
        regardless of knowledge.
  \item \textbf{Token- vs.\ claim-level confidence.} Token probabilities
        compound multiplicatively across a multi-token claim. A model
        calibrated per token can be wildly miscalibrated per \emph{claim}.
  \item \textbf{Distributional shift.} Financial data is non-stationary. A
        model trained on pre-2022 text stays confident on topics that have
        structurally changed---``it does not know that it does not know.''
\end{enumerate}

\medskip
\begin{block}{Documented in finance}
\citet{ChenGreenGulenZhou2024}: LLM return forecasts over-extrapolate recent
trends and show optimistic bias---human-like extrapolation/over-optimism that
\emph{persists after prompt engineering}.
\end{block}
\end{frame}

\begin{frame}{Measuring and Improving Calibration}
\textbf{Getting confidence scores:} log-probs from the API; or, for
text-only access, self-consistency ($k$ samples, fraction agreeing) or
verbalised confidence---though \citet{wang2023survey} show the latter is
unstable (70\% under one prompt, 95\% under another).

\medskip
\begin{underhood}
Three post-hoc recalibration methods:
\begin{itemize}
  \item \textbf{Temperature scaling:} $\operatorname{softmax}(\mathbf{z}/T)$,
        one scalar $T$ fit by NLL. $T>1$ spreads (cures overconfidence);
        preserves ranking $\Rightarrow$ AUROC unchanged.
  \item \textbf{Isotonic regression:} non-parametric monotone map; flexible
        but needs a larger calibration set.
  \item \textbf{Platt scaling:} logistic fit on raw scores; robust on small sets.
\end{itemize}
\end{underhood}
\end{frame}

\begin{frame}{Calibration: Conformal Sets and Monitoring}
\textbf{Conformal prediction} (coverage-guaranteed sets): output
$\{\text{default}, \text{non-default}\}$ when uncertain, with
$\Pr(Y \in \mathcal{C}(\mathbf{x})) \geq 1-\alpha$.

\medskip
\begin{alertblock}{Operational rule}
Calibration is distribution-specific. Re-estimate \textbf{at least quarterly}---
part of model monitoring under SR~11-7.
\end{alertblock}
\end{frame}

\section{Temporal Leakage and Look-Ahead Bias}
% ===============================================================

\begin{frame}{Memorising the Future}
\begin{bigpicture}
Evaluating an LLM is not just asking questions and scoring answers. The model
may have seen the answers---or the test instances themselves---during
pretraining. If the benchmark and the training corpus draw on the same news,
filings, and forums, measured ``performance'' may be \textbf{memorisation, not
generalisation}. This is the most underappreciated failure mode in LLM finance
research.
\end{bigpicture}

\smallskip
\textbf{Canonical trap:} ``Given this Q2-2021 earnings call, did the stock rise
next month?'' A model with a Jan-2022 cutoff has \emph{already read} the news,
the analyst reactions, and the social commentary on the outcome.

\smallskip
\begin{block}{Three leakage channels \citep{ziemke2024temporal}}
\begin{enumerate}\itemsep1pt
  \item \textbf{Direct memorisation} --- exact/near-verbatim document seen.
  \item \textbf{Outcome narratives} --- ``how the trade played out'' retrospectives.
  \item \textbf{Imbalanced corpora} --- textual survivorship bias toward extreme
        outcomes that attracted commentary.
\end{enumerate}
\end{block}
\end{frame}

\begin{frame}{The Stakes Are Empirical, Not Theoretical}
\begin{alertblock}{Memorisation is measurable}
\citet{LopezLiraTangZhu2025}: LLMs have memorised \textbf{exact numerical
values} of key economic/financial variables inside their training window---it
becomes impossible to distinguish genuine out-of-sample forecasting from recall.
\end{alertblock}

\medskip
\textbf{Two confirmations of the magnitude:}
\begin{itemize}
  \item \citet{RahimikiaDrinkall2024} pre-train \emph{year-specific} LLMs
        (2007--2023). Properly time-bounded models \textbf{outperform} much
        larger general-purpose models on trading---so off-the-shelf benchmarks
        overstate true skill.
  \item \citet{GaoJiangYan2025} introduce the \textbf{Lookahead Propensity
        (LAP)}; a positive LAP--accuracy correlation on stock-return and
        capex tasks confirms apparent skill is substantially lookahead bias.
\end{itemize}

\medskip
\textbf{Practitioner taxonomy:} \citet{NoguerAlonso2024} catalogues look-ahead
sources (training cutoffs, entity embeddings, backtest design) and mitigations.
\end{frame}

\begin{frame}{Point-in-Time Discipline and Anonymisation}
\textbf{Point-in-time (PIT):} the data \emph{actually available} at a historical
moment, not later-restated values (cf.\ CRSP/Compustat snapshots).

\medskip
\begin{underhood}
A rigorous LLM protocol demands:
\begin{enumerate}
  \item \textbf{Hard temporal cutoffs} --- evaluate $\geq 3$ months past the
        confirmed knowledge cutoff (publication-lag safety margin).
  \item \textbf{Document-level timestamps} --- verified publication dates;
        exclude uncertain provenance.
  \item \textbf{Contextual anonymisation} --- replace company names, tickers,
        and executives with synthetic identifiers.
\end{enumerate}
\end{underhood}

\medskip
\begin{exampleblock}{Why anonymise}
``Will \emph{Apple} report revenue above \$100B in Q4 2022?'' is a
\textbf{recall} question, not a forecast. Replace with ``Company X, a consumer
electronics firm with $\sim$\$380B revenue'' to force reasoning from context.
\end{exampleblock}
\end{frame}

\begin{frame}{Designing Contamination-Resistant Splits}
Random splits are wrong for temporal data. Use \textbf{temporal splitting} with
a gap $\Delta$:
\begin{align*}
  \mathcal{D}_{\text{train}} &= \{(x_t,y_t): t \leq T_1 - \Delta\}, &
  \mathcal{D}_{\text{valid}} &= \{(x_t,y_t): T_1 \leq t \leq T_2 - \Delta\}, \\
  \mathcal{D}_{\text{test}}  &= \{(x_t,y_t): t \geq T_2\}.
\end{align*}
$\Delta \geq$ prediction horizon $h$ + publication-lag buffer.

\medskip
\begin{underhood}
Three additional controls:
\begin{itemize}
  \item \textbf{N-gram dedup} --- hash 13-gram windows (MinHash LSH) against the
        training corpus; drop high-overlap instances.
  \item \textbf{Factual probing} --- if the model answers facts a non-leaked
        model could not, suspect contamination.
  \item \textbf{Date-blinding ablation} --- a big accuracy drop when temporal
        markers are removed signals pattern retrieval, not reasoning.
\end{itemize}
\end{underhood}
\end{frame}

\begin{frame}{Example: A Contamination Audit}
500 earnings calls (2022--23); LLM cutoff Dec 2023; predict EPS beat/miss.

\medskip
\begin{enumerate}
  \item \textbf{N-gram check:} 34/500 transcripts have $\geq 3$ matching
        13-grams in Common Crawl $\to$ flagged and removed.
  \item \textbf{Anonymise} the remaining 466 (names, tickers, executives).
  \item \textbf{Compare:} accuracy falls \textbf{62\% $\to$ 54\%} on
        anonymisation $\Rightarrow \approx$ 8 pp was entity-level memorisation.
\end{enumerate}

\medskip
\begin{alertblock}{Conclusion}
Residual 54\% still beats the 50\% baseline---but the apparent edge has
\textbf{shrunk substantially}. Contamination auditing is not cosmetic: it
changes conclusions.
\end{alertblock}
\end{frame}

\section{Stock Movement Prediction: A Cautionary Tale}
% ===============================================================

\begin{frame}{The Theoretical Barrier: Efficient Markets}
\begin{bigpicture}
If LLMs digest news, sentiment, and earnings calls faster than humans, perhaps
they find price-relevant signals first. But market efficiency sets a hard limit
on how much edge \emph{publicly available text} can contain.
\end{bigpicture}

\medskip
\textbf{EMH (semi-strong):} prices already reflect all public information
\citep{fama1970efficient}. An LLM trained on public text is
\emph{reconstructing} information already traded on---not extracting novel alpha.

\medskip
\textbf{Grossman--Stiglitz:} markets cannot be \emph{perfectly} efficient---some
mispricing must remain to pay informed traders. The question: can LLMs reach
that residual edge before it is arbitraged away? (Where genuine value might
remain: see appendix.)
\end{frame}

\begin{frame}{Empirical Evidence: 51--65\% Accuracy}
\textbf{Consistent pattern:} directional accuracy a few points above the 50\%
chance level on daily/weekly tasks \citep{xu2024stock}.

\medskip
\begin{itemize}
  \item \citet{LopezLiraTang2023}: ChatGPT headline sentiment yields significant
        directional accuracy out-of-sample, scaling with model complexity---but
        the window was unusually volatile and returns concentrated in some news
        categories. (The benchmark later critiques target.)
  \item \citet{VidalSSRN2024}: four leading LLMs on 30-day S\&P~500 direction ---
        accuracy only \textbf{marginally above chance}, below the economic
        threshold after costs.
\end{itemize}

\medskip
\begin{alertblock}{Above 62\%? Check for these artefacts \citep{zhao2025frontiers}}
Insufficient train--eval gap \,$\cdot$\, no anonymisation \,$\cdot$\, overlapping
windows \,$\cdot$\, gross accuracy without transaction costs. After controls:
residual \textbf{51--54\%}---statistically positive at best, economically
negligible after costs.
\end{alertblock}

\medskip
\textbf{CFA Institute:} durable alpha unlikely in liquid, well-covered markets;
possible value in small-cap, EM, or private-information synthesis.
\end{frame}

\begin{frame}{When Simpler Models Outperform LLMs}
\textbf{``More sophisticated'' $\neq$ more accurate.}

\medskip
\begin{columns}[T]
\begin{column}{0.5\textwidth}
\textbf{ARIMA(-GARCH)} \citep{box1970distribution}
\begin{itemize}\footnotesize
  \item Principled time-series structure (AR, differencing, MA + variance clustering)
  \item Few parameters, fast to refit
  \item No contaminable training corpus
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\textbf{LSTM} \citep{hochreiter1997long}
\begin{itemize}\footnotesize
  \item Learns longer dependencies
  \item On \emph{pure price-sequence} prediction, often \textbf{matches or beats}
        LLMs \citep{zhao2025frontiers}
\end{itemize}
\end{column}
\end{columns}

\smallskip
\begin{block}{Three conditions favouring simpler models}
\begin{enumerate}\itemsep1pt
  \item \textbf{Text irrelevance} --- target driven by price/volume dynamics.
  \item \textbf{Short history} --- too little labelled data to justify fine-tuning.
  \item \textbf{Regime changes} --- ARIMA re-estimates cheaply; re-fine-tuning an
        LLM is an order of magnitude more expensive.
\end{enumerate}
\end{block}

\smallskip
\textbf{Methodological lesson:} a random-walk baseline is not enough---benchmark
against the \emph{best} classical alternative on the same information set.
\end{frame}

\section{Evaluation Beyond Classification Accuracy}
% ===============================================================

\begin{frame}{From Statistical to Economic Metrics}
\begin{bigpicture}
53\% directional accuracy can be worthless after costs; 51\% can be valuable if
the confident calls land on the big moves. Accuracy is necessary, not
sufficient. Financial evaluation must speak in returns, risk, and alpha.
\end{bigpicture}

\medskip
\begin{underhood}
\textbf{Sharpe ratio} (risk-adjusted return):
\[
  \text{SR} = \frac{\overline{R}_p - R_f}{\sigma_p},
  \qquad \text{annualise} \times\sqrt{252}\ \text{(daily)},\ \sqrt{12}\ \text{(monthly)}.
\]
\textbf{Information ratio} (active management):
\[
  \text{IR} = \frac{\alpha}{\text{TE}} \approx \text{IC}\cdot\sqrt{N}.
\]
$\text{IC}$ = signal quality (rank corr.\ predicted vs.\ realised); $N$ =
independent bets/year. IR $>0.5$ strong, $>1.0$ exceptional.
\end{underhood}
\end{frame}

\begin{frame}{Factor Alpha: Is the Edge Just Known Risk?}
The economic question is not ``does the strategy make money?'' but ``does it
make money \emph{beyond} known risk premia?''

\medskip
\textbf{Factor alpha:} regress excess returns on Fama--French factors,
\[
  r_{p,t} = \alpha + \beta_{\text{MKT}}r_{\text{MKT},t}
  + \beta_{\text{SMB}}r_{\text{SMB},t} + \beta_{\text{HML}}r_{\text{HML},t} + \varepsilon_t.
\]
The intercept $\alpha$ is the return unexplained by market, size, and value
exposure---the genuine skill component.

\medskip
\begin{alertblock}{The sobering finding}
Many LLM strategies produce $\alpha$ \textbf{indistinguishable from zero}: their
apparent returns are repackaged factor exposure, not new information.
\end{alertblock}
\end{frame}

\begin{frame}{Walk-Forward and Expanding-Window Validation}
\textbf{Why not $k$-fold CV?} Shuffling leaks the future into training folds.

\medskip
\begin{itemize}
  \item \textbf{Walk-forward:} train to $t$, evaluate $[t, t{+}h]$, advance; aggregate
        out-of-sample windows.
  \item \textbf{Expanding-window (anchored):} fix the start, grow the window,
        re-estimate each step---standard in financial ML
        \citep{lopezdeprado2018advances}.
\end{itemize}

\medskip
\begin{underhood}
Both need a \textbf{purge period} $\Delta_{\text{purge}} \geq h$ between train and
eval, so no observation appears in both via overlapping label windows. At
iteration $k$:
\[
  \mathcal{D}_{\text{train}}^{(k)} = \{t \leq kh - \Delta_{\text{purge}}\},\quad
  \mathcal{D}_{\text{eval}}^{(k)}  = \{kh \leq t \leq (k{+}1)h - 1\}.
\]
\end{underhood}

\medskip
\textbf{Cost trade-off:} full LLM fine-tuning is expensive to re-estimate, so
practitioners widen $h$---at the price of evaluating a staler model late in each
window.
\end{frame}

\begin{frame}{Trading-Grade Evaluation: Costs, Capacity, Drawdown}
A strategy can win on paper and fail in practice.

\smallskip
\begin{underhood}
\textbf{Transaction costs} (turnover $\tau$):
\[
  c_{\text{round-trip}} = 2\tau\,(\text{half-spread} + \text{market impact}).
\]
US large-cap half-spreads 1--3 bps; impact 5--20 bps; daily news rebalancing
can hit \textbf{10--50 bps/day}, erasing any edge.

\smallskip
\textbf{Capacity} (Almgren--Chriss): temporary impact $\approx \eta\,(q/V)$;
profitable at \$10M may be unprofitable at \$1B.

\smallskip
\textbf{Max drawdown:}
$\text{MDD} = \max_{t_1<t_2}\!\bigl(\frac{P_{t_1}-P_{t_2}}{P_{t_1}}\bigr)$;
Calmar $=$ annual return / MDD.
\end{underhood}

\smallskip
\begin{block}{A complete report should state}
Gross \& net Sharpe \,$\cdot$\, IR vs.\ benchmark \,$\cdot$\, MDD \& Calmar
\,$\cdot$\, turnover \& capacity \,$\cdot$\, factor decomposition \,$\cdot$\,
$\alpha$ significance with robust SEs + multiple-testing correction.
\end{block}
\end{frame}

\section{Hallucination in Financial Contexts}
% ===============================================================

\begin{frame}{The Silent Failure Mode}
\begin{bigpicture}
Hallucination---fluent, confident, factually wrong---is undetectable by a reader
who lacks the correct answer. A junior analyst handed a hallucinated analysis
gets no warning signal: the prose reads exactly as plausibly as accurate
content.
\end{bigpicture}

\medskip
\textbf{Four dangerous categories in finance:}
\begin{enumerate}
  \item \textbf{Fabricated citations} --- plausible journals/SSRN IDs/regulatory
        sections that do not exist \citep{bang2023multitask}. Serious liability
        in compliance filings.
  \item \textbf{Incorrect numerical facts} --- revenue, EPS, EBITDA, spreads.
        \citet{kang2023hallucination}: GPT-4 class models err on a non-trivial
        fraction, with \emph{no} lexical signal distinguishing right from wrong.
  \item \textbf{Phantom companies/executives} --- plausible but non-existent
        entities; material risk in due diligence and KYC.
  \item \textbf{Anachronistic facts} --- stale rates/ratings/leadership stated
        confidently \citep{ji2023survey}; no awareness of obsolescence.
\end{enumerate}
\end{frame}

\begin{frame}{Grounding Strategies}
\begin{underhood}
\footnotesize
Reduce reliance on \emph{parametric memory}; anchor to verified external sources.
\begin{itemize}\itemsep1pt
  \item \textbf{RAG} \citep{lewis2020rag} --- retrieve verified docs into context.
        Effectiveness hinges on retrieval quality; topically-relevant but
        factually tangential context still yields hallucination.
  \item \textbf{Retrieval verification} --- FActScore \citep{min2023factscore}
        decomposes output into atomic facts and checks each against a source
        (EDGAR, Bloomberg, a regulatory DB).
  \item \textbf{Tool use / function calling} \citep{schick2023toolformer} ---
        fetch live data; model becomes a reasoning engine, not a knowledge store.
  \item \textbf{Self-consistency} \citep{wang2022selfconsistency,
        manakul2023selfcheckgpt} --- facts stable across samples are likelier
        correct; variable facts flagged for review.
\end{itemize}
\end{underhood}

\smallskip
\begin{alertblock}{Grounding reduces but does not eliminate}
RAG fails on erroneous docs; tools return stale data; self-consistency can be
consistently wrong if the error is strong in parametric memory. High-stakes
claims still need \textbf{human-in-the-loop} review.
\end{alertblock}
\end{frame}

\begin{frame}{Hallucination Benchmarks for Finance}
\centering
\begin{tabular}{lll}
\toprule
\textbf{Benchmark} & \textbf{What it tests} & \textbf{Grounding} \\
\midrule
FinanceBench & QA over real SEC filings & retrieval vs.\ synthesis \\
FinQA        & numerical reasoning over tables & program execution \\
Kang et al.\ & revenue/EPS/balance-sheet recall & real vs.\ fictitious firms \\
\bottomrule
\end{tabular}

\medskip
\raggedright
\begin{itemize}
  \item \textbf{FinanceBench} \citep{zhang2024financebench}: high hallucination
        rates even for frontier models on multi-step numerical questions.
  \item \textbf{FinQA} \citep{chen2021finqa}: evaluation via \emph{program
        execution} catches numerical hallucinations a string match would miss.
  \item \citet{kang2023hallucination}: models confidently answer about
        \emph{fictitious} companies with invented-but-plausible figures.
\end{itemize}

\medskip
\begin{block}{Caveat}
Benchmark scores $\neq$ real-world reliability. Evaluate on \emph{your} task
type, information source, and output format---not aggregate leaderboard numbers.
\end{block}
\end{frame}

\section{Synthesis}
% ===============================================================

\begin{frame}{Lecture 13 Summary}
\begin{enumerate}
  \item \textbf{Calibration} --- LLMs overconfident; ECE can exceed 10\%.
        Recalibrate (temperature/isotonic/Platt); use conformal sets for
        high stakes.
  \item \textbf{Temporal leakage} --- in/near-cutoff evaluation inflates
        results. Hard splits, purge, anonymisation, n-gram dedup. Audits cut
        accuracy by $\geq$8 pp.
  \item \textbf{Stock prediction} --- 51--65\%, upper end from artefacts. After
        controls, residual alpha is economically modest; ARIMA/LSTM often win.
  \item \textbf{Economic evaluation} --- Sharpe, IR, factor alpha; costs,
        capacity, drawdown. Accuracy alone is not enough.
  \item \textbf{Hallucination} --- fabricated citations/numbers/entities, no
        lexical tell. Grounding helps; human review remains necessary.
\end{enumerate}

\medskip
\begin{block}{The disciplined stance}
Limitations do not preclude LLMs in finance---they demand treating outputs as
\textbf{inputs to a decision process}, not end products.
\end{block}
\end{frame}

\begin{frame}[noframenumbering]{Appendix: Adaptive Binning and MCE Detail}
\textbf{Bin-boundary sensitivity.} Equal-width ECE is sensitive to boundaries.
\textbf{Adaptive binning}---equal mass per bin---is more reliable when
confidence scores cluster near 0 or 1, as they do for overconfident models
\citep{niculescu2005predicting}.

\medskip
\textbf{When to prefer MCE over ECE.} In high-stakes regimes a single
catastrophically miscalibrated region (e.g.\ 90--100\% confidence at 40\%
accuracy) is unacceptable even with a good average ECE. Report both.

\medskip
\begin{block}{Surfacing overconfidence}
Manifestations in practice: EPS to two decimals with no interval; definitive
credit calls near the boundary; cherry-picked market rationales; ``in/not in
compliance'' that hides jurisdictional ambiguity.
\end{block}
\end{frame}

\begin{frame}[noframenumbering]{Appendix: Where Genuine LLM Value Might Remain}
Market efficiency limits the edge in \emph{public} text, but residual value can
survive in three places:
\begin{itemize}
  \item \textbf{Synthesis at scale} --- diffuse signals across thousands of
        low-attention documents.
  \item \textbf{Novel reasoning chains} --- cross-domain links a factor model misses.
  \item \textbf{Private/proprietary text} --- not in training, not in prices
        (raises regulatory questions).
\end{itemize}

\medskip
These are precisely the cases the lecture flags as the most defensible uses of
LLMs for prediction---small-cap, emerging-market, or private-information
synthesis, where public-text efficiency bites least.
\end{frame}

\begin{frame}[noframenumbering]{Appendix: The Fundamental Law of Active Management}
\[
  \text{IR} \approx \text{IC}\cdot\sqrt{N}.
\]
\begin{itemize}
  \item \textbf{IC} (Information Coefficient): cross-sectional rank correlation
        between predicted and realised returns---\emph{signal quality}.
  \item \textbf{N}: number of independent bets per year---\emph{breadth},
        driven by securities covered $\times$ prediction frequency.
\end{itemize}

\medskip
\textbf{Why this matters for LLMs.} It separates the two value sources: a weak
signal (low IC) can still produce a respectable IR if breadth $N$ is large
(many names, high frequency)---but breadth multiplies turnover, and turnover
multiplies transaction costs. The decomposition makes the cost/breadth tension
explicit.
\end{frame}
