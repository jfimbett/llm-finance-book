\section{From Language Models to Agents}
% ============================================================

% ============================================================
% FRAME 3: The passivity problem
% ============================================================
\begin{frame}
\frametitle{Why Passivity is a Limitation}

\begin{block}{The static LLM}
  A language model maps a prompt to a probability distribution over tokens.
  It is \textbf{stateless}: each call is independent; it cannot change its environment.
\end{block}

\medskip
\textbf{A portfolio manager asks:}
\begin{quote}
\textit{``Given Apple's last earnings call and the most recent 10-K, what are the key
risks to revenue growth vs.\ two years ago?''}
\end{quote}

Problems with a plain LLM:
\begin{itemize}
  \item Documents may not fit in context
  \item Cross-referencing two sources is unreliable
  \item A follow-up (``now compare to Microsoft'') restarts from scratch
\end{itemize}

\medskip
\textbf{An agent} embeds the LLM inside a loop: observe, reason, act, incorporate result.
\end{frame}

% ============================================================
% FRAME 4: The PRA cycle -- formal
% ============================================================
\begin{frame}
\frametitle{The Perceive--Reason--Act Loop}

A PRA loop is a triple $(\pi, \mu, \phi)$:

\begin{itemize}
  \item $\phi: \mathcal{O} \times \mathcal{M} \to \mathcal{M}$ \quad
        \textbf{memory update} — observation + memory $\to$ new memory
  \item $\pi: \mathcal{M} \to \Delta(\mathcal{A})$ \quad
        \textbf{policy} — memory $\to$ distribution over actions
  \item $\mu: \mathcal{A} \times \mathcal{E} \to \mathcal{O}$ \quad
        \textbf{environment response} — action $\to$ new observation
\end{itemize}

\medskip
\textbf{In an LLM agent:}
\begin{itemize}
  \item $\pi$ is the language model
  \item Memory = the structured prompt (history, tool results, system instructions)
  \item Action = text response, tool call, or termination
  \item Observation = tool output, retrieved document, user message
\end{itemize}

\medskip
\begin{alertblock}{Key bottleneck}
Memory is bounded by the context window ($\approx$100 pages for GPT-4-class).
A company's full filing history does not fit.
\end{alertblock}
\end{frame}

% ============================================================
% FRAME 5: ReAct
% ============================================================
\begin{frame}
\frametitle{ReAct: Interleaving Thought and Action}

\textbf{ReAct} (Yao et al., 2022) interleaves reasoning traces and actions:

\[
  \tau = (T_1, A_1, O_1,\; T_2, A_2, O_2,\; \ldots,\; T_n, A_n, O_n)
\]

$T_i$ = natural-language thought \quad $A_i$ = action \quad $O_i$ = observation

\medskip
\textbf{Why explicit reasoning helps:}\\
Computing $f(t) = $ closing price of AAPL on date $t$:
\begin{itemize}
  \item CoT-only: model approximates $\hat{f}(t)$ from training patterns --- unreliable
  \item ReAct: model calls \texttt{market\_data\_api(date=t)}, receives exact value
\end{itemize}
Downstream reasoning uses the precise number, not the approximation.

\medskip
\begin{block}{Practical result}
ReAct reduces hallucination by grounding each reasoning step in tool-call observations.
\end{block}
\end{frame}

% ============================================================
% FRAME 6: ReAct example
% ============================================================
\begin{frame}
\frametitle{ReAct in Action: Filing Retrieval}

\textbf{Query:} ``What was Tesla's stated liquidity risk policy in its most recent 10-K?''

\medskip
\footnotesize
\begin{tabular}{p{0.12\textwidth}p{0.78\textwidth}}
\textbf{Thought 1} & I need Tesla's most recent 10-K. Use the EDGAR search tool. \\[3pt]
\textbf{Action 1} & \texttt{edgar\_search(ticker="TSLA", form="10-K", limit=1)} \\[3pt]
\textbf{Obs 1} & \texttt{\{accession: "0001318605-24-000004", filed: "2024-01-26"\}} \\[6pt]
\textbf{Thought 2} & I have the accession. Retrieve the liquidity section specifically. \\[3pt]
\textbf{Action 2} & \texttt{edgar\_section(accession="...", section="Liquidity")} \\[3pt]
\textbf{Obs 2} & \texttt{[3,200 words of text...]} \\[6pt]
\textbf{Thought 3} & I have the relevant section. Extract the stated policy. \\[3pt]
\textbf{Action 3} & \textsc{Finish}(structured summary of liquidity policy) \\
\end{tabular}
\normalsize

\medskip
Three tool-call steps replace one hallucination-prone generation step.
\end{frame}

% ============================================================
% FRAME 7: Reflexion and planning variants
% ============================================================
\begin{frame}
\frametitle{Beyond ReAct: Planning Architectures}

\textbf{Tree-of-Thought (ToT):} structures reasoning as a search tree;
evaluates multiple candidate paths before committing.

\medskip
\textbf{Reflexion} (Shinn et al., 2023): adds verbal self-critique after each trajectory.

\[
  m_0^{(k+1)} \gets \ell^{(k)} \oplus m_0^{(k)}
\]

$\ell^{(k)}$ = lesson from episode $k$; prepended to memory on next episode.

\medskip
\textbf{Finance application:} an agent identifying comparable companies might generate the lesson:\\
\textit{``I should filter by market cap and GICS sub-industry simultaneously,
not sequentially.''}\\
This carries forward without retraining.

\medskip
\begin{block}{Key insight}
Verbal self-improvement substitutes for expensive fine-tuning in low-data regimes.
\end{block}
\end{frame}

\section{Memory}
% ============================================================

% ============================================================
% FRAME 8: Memory taxonomy
% ============================================================
\begin{frame}
\frametitle{Three Types of Agent Memory}

\begin{center}
\begin{tabular}{lccc}
\toprule
\textbf{Type} & \textbf{Size} & \textbf{Latency} & \textbf{Persistence} \\
\midrule
In-context & $\leq$ ctx window & $\approx 0$ & Session only \\
Retrieval-augmented & Unbounded & 10--100 ms & Persistent \\
Long-term (DB) & Unbounded & 1--10 ms & Persistent \\
\bottomrule
\end{tabular}
\end{center}

\medskip
\textbf{In-context}: current conversation, recent tool results. Zero latency. Discarded at session end.

\textbf{Retrieval-augmented}: vector database; approximate nearest-neighbour search. Suitable for semantic queries over large corpora (10,000 SEC filings).

\textbf{Long-term}: relational or graph DB; precise key-value lookups. Suitable for entity metadata, fiscal calendar, analyst revision history.

\medskip
Financial agents layer all three: right memory tier for each information need.
\end{frame}

% ============================================================
% FRAME 9: Retrieval and the error propagation problem
% ============================================================
\begin{frame}
\frametitle{Retrieval Error Propagates into Reasoning}

Given $N$ document vectors $\{\mathbf{v}_i\} \subset \mathbb{R}^D$ and query $\mathbf{q}$:
\[
  \text{Retrieve}(q,k) = \operatorname{argtopk}_{i}\;
  \frac{\mathbf{q} \cdot \mathbf{v}_i}{\|\mathbf{q}\|\,\|\mathbf{v}_i\|}
\]

\medskip
Two failure modes:
\begin{itemize}
  \item \textbf{Missed relevant doc} $\to$ hallucinated gap in the answer
  \item \textbf{Spurious retrieval} $\to$ noise introduced into reasoning
\end{itemize}

\medskip
\begin{alertblock}{In finance, retrieval errors matter}
A faithfulness failure in a compliance summary can be a regulatory violation.
Faithfulness $< 0.85$ $\to$ trigger human review, not automated publication.
\end{alertblock}
\end{frame}

\section{Tool Use and Function Calling}
% ============================================================

% ============================================================
% FRAME 10: Tool paradigm
% ============================================================
\begin{frame}
\frametitle{The Tool Use Paradigm}

A \textbf{tool} is a triple $\mathcal{T} = (\textit{name}, \textit{schema}, f)$:
\begin{itemize}
  \item $\textit{name}$ --- unique identifier
  \item $\textit{schema}$ --- JSON Schema: input domain and output co-domain
  \item $f: \mathcal{D} \to \mathcal{R} \cup \{\textsc{Error}\}$ --- implementation
\end{itemize}

\medskip
\textbf{How it works (OpenAI/Anthropic/Google):}
\begin{enumerate}
  \item Tools declared in system prompt as JSON array
  \item Model generates structured tool-call message with function name + arguments
  \item Execution environment dispatches function, returns result as new message
  \item Model continues reasoning with tool result in context
\end{enumerate}

\medskip
\begin{block}{Highest-leverage investment in agent reliability}
Clear, precise field descriptions --- specifying units, valid ranges, edge-case behaviour.
These descriptions propagate into the JSON Schema the model receives.
\end{block}
\end{frame}

% ============================================================
% FRAME 11: Custom tool example
% ============================================================
\begin{frame}[fragile]
\frametitle{Custom Tool: DCF Calculator}

\begin{lstlisting}
from pydantic import BaseModel, Field

class DCFInput(BaseModel):
    free_cash_flows: list[float] = Field(
        description="Projected FCFs in USD millions, chronological"
    )
    wacc: float = Field(
        description="WACC as a decimal (0.08 = 8%)", ge=0.0, le=1.0
    )
    terminal_growth_rate: float = Field(
        description="Perpetual growth rate as a decimal", ge=0.0, le=0.15
    )
\end{lstlisting}

\medskip
\begin{itemize}
  \item Pydantic field descriptions propagate into the JSON Schema
  \item Validation errors returned to LLM as structured feedback ---
        agent regenerates with corrected arguments
  \item Safer pattern than raw SQL: restrict action space to
        semantically meaningful financial operations
\end{itemize}
\end{frame}

% ============================================================
% FRAME 12: Tool reliability
% ============================================================
\begin{frame}
\frametitle{Tool Reliability and Error Recovery}

Production agents must handle tool failures gracefully. Failure sources:
network timeouts, API rate limits, schema validation errors, empty results, ambiguous queries.

\medskip
\textbf{Layered recovery strategy:}
\begin{enumerate}
  \item \textbf{Innermost}: retry with exponential back-off (transient failures)
  \item \textbf{Middle}: interpret error as observation; generate alternative action
        (different endpoint, relaxed date range, decomposed query)
  \item \textbf{Outermost}: \textsc{RequestHumanHelp}(description) rather than
        returning a hallucinated answer
\end{enumerate}

\medskip
\begin{alertblock}{Never pass errors silently}
A robust tool calling protocol always returns an informative \textsc{Error(msg)}.
The agent policy treats this as a valid observation and generates a recovery action.
\end{alertblock}
\end{frame}

% ============================================================
% FRAME 13: Skills and hooks
% ============================================================
\begin{frame}
\frametitle{Skills and Hooks}

\textbf{Skill}: a named, parameterised agent sub-workflow
$\mathcal{S}: \mathcal{P} \times \mathcal{M}_{\text{in}} \to \mathcal{M}_{\text{out}}$.
Abstracts over prompt engineering, tool selection, and error handling for a specific task.

\medskip
Examples: \texttt{SummariseEarningsCall}, \texttt{RetrieveRiskFactors},
\texttt{GenerateCompsTable}. Skills compose: a coordinator invokes them in sequence
without knowing their internals.

\medskip
\rule{\textwidth}{0.4pt}

\textbf{Hook}: a callback function $(\mathcal{E}, h)$ where $\mathcal{E}$ is an event type
and $h: \mathcal{E} \times \mathcal{M} \to \mathcal{M}$ is a handler.

\medskip
Financial hooks:
\begin{itemize}
  \item \textbf{Audit hook}: logs every tool call to an immutable audit trail (MiFID II)
  \item \textbf{Compliance hook}: checks proposed response before any \textsc{Finish}
  \item \textbf{Cost hook}: accumulates token usage; alerts on budget threshold breach
\end{itemize}

Hooks decouple cross-cutting concerns from the agent's core reasoning loop.
\end{frame}

% ============================================================
% FRAME 15: Multi-agent systems
% ============================================================
\begin{frame}
\frametitle{Multi-Agent Systems and Role Delegation}

A multi-agent system is $(\mathcal{AG}, \mathcal{C}, \delta)$:
agents, communication topology, routing function.

\textbf{Key insight}: task decomposition reduces error --- a specialist agent
handling a narrow subtask outperforms a generalist managing everything.

\medskip
\textbf{Hierarchical delegation for equity research:}
\begin{enumerate}
  \item \textit{Coordinator} --- decomposes request, assembles final output
  \item \textit{Data Agent} --- retrieves financial statements and market data
  \item \textit{Quant Agent} --- computes valuation ratios, scenario analysis
  \item \textit{Analyst Agent} --- drafts qualitative commentary
  \item \textit{Compliance Agent} --- checks regulatory compliance, appends disclosures
\end{enumerate}

\medskip
\begin{block}{Production result}
First-draft research report in $\approx 4$ minutes vs.\ 2 days for a human team.
Humans review compliance flags and quant scenario assumptions.
\end{block}
\end{frame}

\section{Retrieval-Augmented Generation in Finance}
% ============================================================

% ============================================================
% FRAME 16: RAG motivation and formulation
% ============================================================
\begin{frame}
\frametitle{RAG: Theory and Finance Motivation}

\textbf{Motivation}: a large asset manager's research library has 50,000 documents.
No context window holds this; no fine-tuning reliably memorises it.

\medskip
\textbf{Retrieval-Augmented Generation} (Lewis et al., 2020):
\[
  p(a \mid q) = \sum_{z \in \mathcal{D}} p(z \mid q)\, p(a \mid q, z)
\]

In practice: approximate by conditioning on top-$k$ retrieved documents only.

\medskip
\textbf{Dominant deployment (naive RAG):}
\begin{itemize}
  \item Frozen bi-encoder retriever (sentence transformer)
  \item Closed-source LLM generator via API
  \item Sacrifices end-to-end optimisation; dramatically reduces cost
\end{itemize}

\medskip
\begin{alertblock}{Metadata filtering is critical}
Query for ``Apple CFO supply chain'' should be restricted to Apple documents.
Metadata filtering reduces irrelevant retrievals by $\approx 60\%$.
\end{alertblock}
\end{frame}

% ============================================================
% FRAME 17: Hybrid search
% ============================================================
\begin{frame}
\frametitle{Hybrid Search: Dense + Sparse}

\textbf{Dense retrieval} (embedding-based): captures semantic similarity.
Fails on exact match queries (CUSIP numbers, regulatory citations).

\medskip
\textbf{BM25} (sparse): strong on exact term matching. Defined as:
\[
  \text{BM25}(q,d) = \sum_{i} \text{IDF}(q_i) \cdot
  \frac{f(q_i,d)(k_1+1)}{f(q_i,d) + k_1(1-b+b\,|d|/\text{avgdl})}
\]

\textbf{Hybrid score} (after min-max normalisation):
\[
  \text{score}_{\text{hybrid}} =
  \alpha \cdot \widetilde{\text{cosine}} + (1-\alpha) \cdot \widetilde{\text{BM25}}
\]

\medskip
Empirical guidance:
\begin{itemize}
  \item $\alpha \approx 0.7$ (favour dense): QA over long financial documents
  \item $\alpha \approx 0.3$ (favour sparse): keyword-heavy regulatory lookup
  \item Reciprocal Rank Fusion (RRF) avoids normalisation entirely
\end{itemize}
\end{frame}

% ============================================================
% FRAME 18: Chunking strategies
% ============================================================
\begin{frame}
\frametitle{Chunking Strategies for Financial Documents}

A 10-K may be 200 pages with hierarchical structure.
Naive fixed-size chunking severs meaning at arbitrary points.

\medskip
\textbf{Three strategies:}
\begin{enumerate}
  \item \textbf{Fixed-size}: $L$ tokens, $\delta$ overlap. Simple; ignores structure.
  \item \textbf{Semantic}: sentence/paragraph boundaries; EDGAR item structure for SEC filings.
  \item \textbf{Hierarchical}: multi-level index (document $\to$ section $\to$ paragraph).
        Query first retrieves section, then paragraph within section.
\end{enumerate}

\medskip
\textbf{Recommended for 10-K filings:} hierarchical combining EDGAR items (top level)
with sentence-boundary chunking within items (max 512 tokens, 64-token overlap).

\begin{block}{Benchmark result}
$+15$--$20\%$ on downstream QA vs.\ fixed-size chunking.
Apple 2023 10-K: 94\% paragraph recall on a manually labelled evaluation set.
\end{block}
\end{frame}

\begin{frame}
\frametitle{RAG Evaluation: RAGAS Metrics}

\begin{center}
\begin{tabular}{lp{0.65\textwidth}}
\toprule
\textbf{Metric} & \textbf{Definition} \\
\midrule
Faithfulness & Fraction of answer claims supported by retrieved context \\
Answer Relevance & Semantic similarity between answer and query \\
Context Precision & Fraction of retrieved chunks relevant to query \\
Context Recall & Fraction of ground-truth claims covered by context \\
\bottomrule
\end{tabular}
\end{center}

\medskip
\textbf{Finance priority order:}
\begin{enumerate}
  \item \textbf{Faithfulness} is most critical: unfaithful answer = hallucinated claim
        = compliance failure when presented to clients.
        Faithfulness $< 0.85$ $\to$ block automated publication; trigger human review.
  \item \textbf{Context Recall} matters for completeness:
        high faithfulness + low recall = accurate but incomplete (material disclosures may be omitted).
\end{enumerate}
\end{frame}

\section{Finance Applications}
% ============================================================

% ============================================================
% FRAME 20: Earnings call analysis
% ============================================================
\begin{frame}
\frametitle{Earnings Call Analysis Pipeline}

Five stages:
\begin{enumerate}
  \item \textbf{Ingestion}: transcript from vendor API or EDGAR
  \item \textbf{Segmentation}: classify speaker turns as guidance / analyst Q / clarification
  \item \textbf{Metric extraction}: identify numerical guidance; compare to prior and consensus
  \item \textbf{Sentiment}: directional score per segment (financially calibrated scale)
  \item \textbf{Synthesis}: structured summary with transcript citations
\end{enumerate}

\medskip
\textbf{Predictive evidence:} Kim et al.\ (2024) --- GPT-4 fed anonymised financial statements
predicts earnings direction at 60\% accuracy vs.\ 53\% for human analysts.

\medskip
\begin{alertblock}{Domain knowledge matters}
General models misinterpret financial language:
``margin compression was manageable'' = negative signal.
``exceptional inventory build'' (consumer electronics) = negative signal.
\end{alertblock}
\end{frame}

% ============================================================
% FRAME 21: Report generation and filing search
% ============================================================
\begin{frame}
\frametitle{Report Generation and Filing Search}

\textbf{Automated report generation --- core principle:}
\begin{itemize}
  \item Prompting ``write a report about NVIDIA'' $\to$ hallucinated figures
  \item Data agent assembles \emph{verified} data package first
  \item Writer agent produces prose \emph{describing} that package, citing each figure by source
  \item MiFID II substantiation requirements apply to algorithmic research too
\end{itemize}

\medskip
\rule{\textwidth}{0.4pt}

\textbf{Filing Q\&A --- citation enforcement wrapper:}
\begin{itemize}
  \item Every claim must be accompanied by a chunk citation
  \item Cited chunk must contain the claimed information
  \item Unsupported claims $\to$ \texttt{[Claim not verified; human review required]}
  \item Two-model pipeline: large generator + smaller verifier model
\end{itemize}

\medskip
\textbf{FinanceBench benchmark:} GPT-4-Turbo with retrieval incorrectly answers
or refuses \textbf{81\%} of questions --- production-grade financial RAG is hard.
\end{frame}

\begin{center}

\begin{itemize}
  \item Irreversible + high-consequence $\to$ pre-execution approval
  \item Reversible + medium-consequence $\to$ checkpoint review
  \item Low-stakes + high-volume $\to$ post-hoc audit
\end{itemize}

% ============================================================
% FRAME 23: Audit trails
% ============================================================
\begin{frame}
\frametitle{Audit Trails and Explainability}

An audit trail $\mathcal{A} = (r_1, \ldots, r_T)$ where each record $r_t$ contains:
timestamp, record type, full input/output payload, and:
\[
  \text{hash}_t = H(\text{payload}_t \,\|\, \text{hash}_{t-1})
\]

Chained hashes provide \textbf{tamper evidence}: modifying any historical record
invalidates all subsequent hashes.

\medskip
\textbf{Regulatory requirements:}
MiFID II Article 25, SEC Rule 17a-4 apply to algorithmically generated decisions
influencing investment outcomes.

\medskip
\textbf{Explainability vs.\ auditability:}
\begin{itemize}
  \item Audit trail records \emph{what} the agent did
  \item Explanation addresses \emph{why}
  \item For ReAct agents: the reasoning trace (\textsc{Thought} steps) \emph{is} the explanation
  \item Strong argument for agents that externalise reasoning in natural language
\end{itemize}
\end{frame}

% ============================================================
% FRAME 24: Prompt injection
% ============================================================
\begin{frame}
\frametitle{Prompt Injection and Adversarial Robustness}

\textbf{Prompt injection}: malicious instructions in tool outputs or retrieved documents
override the system prompt.

\medskip
\textbf{Indirect injection} (Greshake et al., 2023): attacker controls a web page;
embeds hidden instructions executed by any agent retrieving that page.

\medskip
\textbf{Defence in depth (no single control sufficient):}
\begin{enumerate}
  \item \textbf{Input sanitisation}: strip instruction-like patterns from retrieved text
  \item \textbf{Privilege separation}: trusted (system prompt) vs.\ untrusted (retrieved docs)
  \item \textbf{Invariant monitoring}: separate agent blocks actions violating invariants
  \item \textbf{Dual-key authorisation}: consequential actions require human + agent confirmation
\end{enumerate}

\medskip
\begin{block}{Formal model: confused deputy problem}
A powerful program (agent) is manipulated by a less-privileged input (document)
into exercising elevated privileges. Defence: capability-based security ---
trade-execution API key stored separately, requiring additional authentication.
\end{block}
\end{frame}
