\section{Why Explainability Matters in Regulated Finance}

\begin{frame}{Why Explainability Matters in Regulated Finance}
  \begin{itemize}
    \item Financial AI systems are subject to strict legal frameworks: EU AI Act, SR 11-7, ECOA
    \item Clients, regulators, and auditors all demand interpretable, auditable decisions
    \item LLMs act as black boxes, making it difficult to trace how conclusions are reached
    \item Unexplainable models create legal liability and erode stakeholder trust
  \end{itemize}
\end{frame}

\section{Classical XAI Methods Applied to LLMs}

\begin{frame}{Classical XAI Methods Applied to LLMs}
  \begin{itemize}
    \item SHAP (SHapley Additive exPlanations) quantifies feature contributions; extensions exist for NLP models
    \item LIME generates locally faithful, model-agnostic explanations by perturbing inputs
    \item Attention weights provide a window into transformer computations, but are not always faithful
    \item These classical methods can be adapted to language models with care
  \end{itemize}
\end{frame}

\section{LLM-Native Explainability Mechanisms}

\begin{frame}{LLM-Native Explainability Mechanisms}
  \begin{itemize}
    \item Chain-of-thought prompting elicits step-by-step reasoning that serves as an explanation trace
    \item Counterfactual explanations identify minimal input changes that would alter the model's output
    \item LLMs can generate natural language justifications tailored to different audiences
    \item These mechanisms are tightly coupled to the model's generative capabilities
  \end{itemize}
\end{frame}

\section{Case Studies in Financial Explainability}

\begin{frame}{Case Studies in Financial Explainability}
  \begin{itemize}
    \item Credit scoring: regulators and applicants require clear rejection reasons
    \item ECOA/FCRA-compliant loan denial letters generated by LLMs with legally defensible justifications
    \item MiFID II mandates that investment recommendations include a suitability explanation
    \item Each use case illustrates different constraints on explanation format and content
  \end{itemize}
\end{frame}

\section{Evaluating the Quality of Explanations}

\begin{frame}{Evaluating the Quality of Explanations}
  \begin{itemize}
    \item Faithfulness measures whether the explanation accurately reflects the model's internal reasoning
    \item Completeness and plausibility metrics assess coverage and human-perceived coherence
    \item Human-subject studies with domain experts reveal practical gaps between technical and user-facing explanations
    \item Robust evaluation requires both automated metrics and qualitative feedback loops
  \end{itemize}
\end{frame}
