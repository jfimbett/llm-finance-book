\begin{frame}{Recap: The Two Formulas You Need}
\textbf{Reciprocal rank fusion} (hybrid retrieval):
\[
\mathrm{RRF}(d) = \sum_{r \in R} \frac{1}{\kappa + \mathrm{rank}_r(d)},
\qquad \kappa = 60.
\]
$R = \{\text{dense}, \text{sparse}\}$; missing $\Rightarrow$ rank $\infty$.

\medskip
\textbf{Adverse Media Index} (text $\to$ calibrated risk):
\[
\mathrm{AMI}(s) = \sigma\!\left(\alpha + \sum_{i=1}^{n} r_i\, c_i\, t_i\,
\mathbf{w}^{\top}\mathbf{e}_i\right), \qquad \sigma(x) = \frac{1}{1+e^{-x}}.
\]
Per document $i$: relevance $r_i$, credibility $c_i$, recency $t_i$, content
score $\mathbf{w}^{\top}\mathbf{e}_i$; $\alpha$ = baseline intercept.
\end{frame}

% ====================================================================
\begin{frame}{Recap: Where These Sit in the Pipeline}
\begin{underhood}
RRF fuses the \emph{retriever}'s two ranked lists; the AMI is computed by the
\emph{output formatter} from the reader's per-document evidence scores.
\end{underhood}

\medskip
\begin{tabular}{ll}
\toprule
\textbf{Stage} & \textbf{Produces} \\
\midrule
Hybrid retriever & dense rank, sparse rank per doc \\
$\to$ RRF & fused ranking $\to$ top-$k$ passages \\
Reader / ranker & $r_i, c_i, t_i, \mathbf{e}_i$ per doc \\
$\to$ AMI & scalar risk $\in [0,1]$ \\
\bottomrule
\end{tabular}

\medskip
\textbf{Why a continuous score?} A binary low/high label cannot be compared across
subjects, tracked over time, or fed into a quantitative customer-risk model.
\end{frame}

\begin{frame}[fragile]{Case Study: RRF Fusion in Python (Notebook 11)}
From \texttt{code/notebooks/11-regtech-compliance-aml/demo.ipynb} --- deterministic,
no network. \textbf{Part 1 --- hybrid retrieval fusion:}

\begin{lstlisting}
import math
KAPPA = 60
docs        = ['court filing','news article','press release','blog post','registry record']
dense_rank  = [1, 3, 2, 5, 4]
sparse_rank = [2, 1, 4, 3, 5]

def rrf(ranks):
    return sum(1.0 / (KAPPA + r) for r in ranks)

fused = {d: rrf([dr, sr]) for d, dr, sr in zip(docs, dense_rank, sparse_rank)}
for d, s in sorted(fused.items(), key=lambda kv: -kv[1]):
    print(f'{d:16s} RRF={s:.5f}')
\end{lstlisting}
\end{frame}

% ====================================================================
\begin{frame}[fragile]{Case Study: AMI Scoring in Python (Notebook 11)}
\textbf{Part 2 --- calibrated Adverse Media Index} (same notebook):

\begin{lstlisting}
import math
def sigmoid(x): return 1.0 / (1.0 + math.exp(-x))
alpha = -2.0
# (relevance, credibility, recency, content_score) per retrieved article
articles = [(0.9,0.8,1.0,2.0), (0.6,0.5,0.7,1.2), (0.3,0.9,0.4,0.5)]
ami = sigmoid(alpha + sum(r*c*t*e for (r,c,t,e) in articles))
print(f'AMI = {ami:.3f}  (probability of genuine AML risk in [0,1])')
\end{lstlisting}
\end{frame}

\begin{frame}{Solution: Problem 1 --- RRF by Hand}
\textbf{Task A.} $\mathrm{RRF}(d) = \frac{1}{60+\text{dense}} + \frac{1}{60+\text{sparse}}$:

\medskip
\begin{tabular}{lccc}
\toprule
\textbf{Passage} & \textbf{Dense} & \textbf{Sparse} & \textbf{RRF} \\
\midrule
court filing    & 1 & 2 & $\tfrac{1}{61}+\tfrac{1}{62}=0.03252$ \\
news article    & 3 & 1 & $\tfrac{1}{63}+\tfrac{1}{61}=0.03227$ \\
press release   & 2 & 4 & $\tfrac{1}{62}+\tfrac{1}{64}=0.03175$ \\
blog post       & 5 & 3 & $\tfrac{1}{65}+\tfrac{1}{63}=0.03126$ \\
registry record & 4 & 5 & $\tfrac{1}{64}+\tfrac{1}{65}=0.03101$ \\
\bottomrule
\end{tabular}

\medskip
\textbf{Task B \& C.} Order: court filing $\succ$ news article $\succ$ press
release $\succ$ blog $\succ$ registry. \emph{News article} is ranked \#1 by sparse
but only 3rd by dense; \emph{court filing} is \#1 dense and \#2 sparse --- its
consistency across \emph{both} signals lifts it above the single-signal winner.
That is exactly the robustness property RRF buys.
\end{frame}

\begin{frame}{Solution: Problem 2 --- Adverse Media Index}
\textbf{Task A.} Contributions $r_i c_i t_i (\mathbf{w}^{\top}\mathbf{e}_i)$:
\begin{align*}
\text{Art.\ 1} &= 0.9 \times 0.8 \times 1.0 \times 2.0 = 1.440 \\
\text{Art.\ 2} &= 0.6 \times 0.5 \times 0.7 \times 1.2 = 0.252 \\
\text{Art.\ 3} &= 0.3 \times 0.9 \times 0.4 \times 0.5 = 0.054
\end{align*}

\textbf{Task B.} Sum $= 1.746$; add $\alpha = -2.0 \Rightarrow -0.254$:
\[
\mathrm{AMI}(s) = \sigma(-0.254) = \frac{1}{1+e^{0.254}} \approx \mathbf{0.437}.
\]

\medskip
\textbf{Task C.} Despite the highest credibility ($c_3 = 0.9$), Article 3 is
crushed by \emph{low relevance} ($0.3$) and \emph{stale recency} ($0.4$) ---
the factors multiply, so any single weak factor drags the whole term down.
Credibility cannot rescue a doc that is barely about the subject or years old.
\end{frame}

\begin{frame}{Wrap-Up: Key Takeaways}
\textbf{Problem 1 (RRF):}
\begin{itemize}\footnotesize
  \item Fusion rewards cross-signal agreement --- a doc strong on \emph{both}
        retrievers beats a single-signal \#1.
  \item $\kappa = 60$ compresses rank differences; no learned weighting needed.
\end{itemize}

\medskip
\textbf{Problem 2 (AMI):}
\begin{itemize}\footnotesize
  \item Multiplicative factors mean one weak component (relevance, recency)
        dominates --- credibility alone cannot save a stale, off-target document.
  \item $\alpha$ sets the prior; $\mathbf{w}$ encodes offence severity and must be
        calibrated on enforcement outcomes.
\end{itemize}

\medskip
\textbf{Discussion:}
\begin{itemize}\footnotesize
  \item Degree of automation decides the Annex III tier --- \emph{assist} vs.\
        \emph{decide / file}.
  \item Good HITL gives the human a task they do better than the model.
\end{itemize}

\begin{frame}[noframenumbering]{Appendix: Stretch Problem --- Entity Resolution Threshold}
A sanctions screener outputs a calibrated match probability
$P(\hat y = 1 \mid x_1, x_2)$. Onboarding 1{,}000{,}000 customers/year; the
prevalence of true sanctions hits is $\approx 1$ in $50{,}000$.

\begin{enumerate}
  \item At a threshold giving recall $0.999$ and FPR $0.02$, how many false
        positives must analysts review per year? (Hint: $\approx
        0.02 \times 10^6$.)
  \item Lowering the threshold raises recall toward $1.0$ but inflates FPR. Argue,
        using the cost asymmetry, why sanctions screening still favours the higher
        recall despite the review burden.
  \item How would a two-stage (ANN blocking $\to$ cross-encoder) pipeline change
        the operating point versus a single fuzzy-match stage?
\end{enumerate}
\end{frame}

\begin{frame}[noframenumbering]{Appendix: Stretch Problem --- AMI Temporal Decay}
Recency weight is $t_i = e^{-\lambda (T - T_i)}$ with $T - T_i$ in days.

\begin{enumerate}
  \item For $\lambda = 0.01$/day, compute $t_i$ for an article 30, 180, and 365
        days old. Confirm $t \to 0$ slowly.
  \item Re-run Problem 2's Article 1, now dated 365 days ago instead of today.
        Recompute its contribution and the resulting AMI.
  \item A subject was \emph{acquitted} 4 years ago but old coverage still surfaces.
        Argue for an aggressive (large $\lambda$) vs.\ conservative (small
        $\lambda$) decay, and the fairness implications of each.
\end{enumerate}
\end{frame}
