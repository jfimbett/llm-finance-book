\section{The Regulatory Landscape for AI in Finance}
% ====================================================================

\begin{frame}{Why Compliance Is the Oldest Friction in Finance}
\begin{bigpicture}
Long before neural networks, banks employed armies of analysts to read news,
cross-reference sanctions lists, and write suspicious activity reports. AI
entered an environment \emph{already saturated} with rules --- rules not written
with AI in mind.
\end{bigpicture}

\medskip
\textbf{The challenge is not too little guidance --- it is too much.} A single
LLM for adverse media screening must simultaneously satisfy:
\begin{itemize}
  \item EU AI Act --- risk classification \& conformity assessment
  \item MiFID II --- algorithmic governance \& suitability documentation
  \item GDPR --- automated processing \& data-minimisation constraints
  \item FATF --- the risk-based approach to AML/CFT
  \item SR 11-7 / EBA --- model risk management expectations
\end{itemize}
Each uses different vocabulary, evidence standards, and oversight ownership.
\end{frame}

\begin{frame}{EU AI Act: Four Risk Tiers (Reg.\ (EU) 2024/1689)}
The first comprehensive horizontal AI law. Classifies systems by risk to
health, safety, or fundamental rights.

\medskip
\begin{tabular}{lll}
\toprule
\textbf{Tier} & \textbf{Examples} & \textbf{Status} \\
\midrule
Unacceptable & social scoring, biometric ID & Prohibited \\
High risk    & creditworthiness, insurance (Annex III) & Mandatory reqs.\ \\
Limited risk & chatbots, emotion recognition & Transparency \\
Minimal risk & spam filters, recommenders & Unregulated \\
\bottomrule
\end{tabular}

\medskip
\begin{alertblock}{The critical question for finance: Annex III?}
Creditworthiness assessment is \emph{explicitly} listed. Adverse media
screening that \emph{determines} credit/insurance/payment access $\to$
high-risk. An LLM that merely \emph{summarises} an article for a human analyst
$\to$ limited / minimal risk.
\end{alertblock}
\end{frame}

\begin{frame}{High-Risk Obligations $\to$ Engineering Choices}
\begin{underhood}
Seven categories of mandatory requirement, each mapping to a concrete build
decision.
\end{underhood}

\medskip
\begin{tabular}{ll}
\toprule
\textbf{Requirement} & \textbf{Engineering implication} \\
\midrule
Risk management (lifecycle) & drift monitoring, re-validation \\
Data governance & representative, proxy-free training data \\
Technical documentation & ex-post reviewable model cards \\
Automatic logging & audit-trail infrastructure \\
Transparency to deployers & instructions for use \\
Human oversight & analyst review-queue UI \\
Accuracy / robustness / security & benchmarking + red-teaming \\
\bottomrule
\end{tabular}
\end{frame}

\begin{frame}{High-Risk Obligations: Who Bears Them?}
\begin{block}{Provider vs.\ deployer}
Banks using third-party compliance AI are \emph{deployers}, not providers --- but
still bear obligations (use per instructions, monitor, ensure oversight).
\textbf{Third-party reliance does not eliminate regulatory liability.}
\end{block}

\medskip
The seven high-risk obligations on the previous slide attach to the
\emph{provider} of the AI system, but a deployer that integrates the system into
a credit or onboarding decision inherits a parallel duty of care: verify the
system is used within its stated scope, keep humans genuinely in the loop, and
log its outputs for ex-post review.
\end{frame}

\begin{frame}{MiFID~II, Basel~III, and SR~11-7}
\begin{tabular}{ll}
\toprule
\textbf{Framework} & \textbf{What it requires of models} \\
\midrule
MiFID II & kill switches, pre/post-trade limits, annual self-assessment \\
Basel III (IRB) & supervisory validation, back-testing, stress testing \\
SR 11-7 & inventory, independent validation, ongoing monitoring \\
\bottomrule
\end{tabular}

\medskip
\textbf{SR 11-7 definition of a \emph{model}:} a quantitative method that applies
statistical / economic / mathematical theory to process input data into
quantitative estimates. Three pillars: \emph{development}, \emph{validation},
\emph{governance}.

\begin{alertblock}{Why LLMs strain SR 11-7}
Outputs are probabilistic; internals are not interpretable like a logistic
coefficient; behaviour shifts with prompt reformulation. The FSB (2024) flags
``challenges for existing MRM frameworks'' and urges pre-deployment testing,
benchmark monitoring, and adversarial-prompt scenario analysis.
\end{alertblock}
\end{frame}

\begin{frame}{GDPR: Four Provisions That Bind LLM Pipelines}
\begin{tabular}{ll}
\toprule
\textbf{Article} & \textbf{Constraint on LLM compliance systems} \\
\midrule
Art.\ 5 (minimisation) & retain only data necessary --- tension with RAG chunks \\
Art.\ 9 (special data) & ethnicity, politics, convictions $\to$ need legal basis \\
Art.\ 17 (erasure) & delete embeddings; can't ``un-train'' fine-tuned weights \\
Art.\ 22 (auto.\ decisions) & right to human review of solely-automated decisions \\
\bottomrule
\end{tabular}

\medskip
\begin{exampleblock}{Art.\ 22 in practice}
A neobank LLM scores \emph{A.\ Petrov} at 0.87 (high risk) from articles about a
\emph{phonetically similar} name in an Eastern-European fraud case, and
auto-declines onboarding. Art.\ 22 grants Petrov the right to human review, to
state his view, and to contest. ``Our AI said so'' is \emph{not} a sufficient
explanation --- the pipeline must log which articles, passages, and risk factors
drove the score.
\end{exampleblock}
\end{frame}

\section{Anti-Money Laundering with LLMs}
% ====================================================================

\begin{frame}{The Scale of the Problem}
\begin{bigpicture}
Money laundering moves criminal proceeds through the financial system to obscure
their origin. FATF estimates \textbf{2--5\% of global GDP} ($\sim$\$800bn--\$2tn)
is laundered annually --- yet only a very small fraction of illicit flows is ever
seized or frozen.
\end{bigpicture}

\medskip
\textbf{Why LLMs, why now?} The explosion of sources --- news, watchlists,
beneficial-ownership registries, court filings, social media --- has outpaced
human analytical capacity, while regulators expect \emph{comprehensive}
monitoring. The gap between obligation and capability is where LLMs add value.

\medskip
\begin{block}{This section}
False-positive crisis $\to$ agentic screening $\to$ RAG architecture $\to$ a
quantitative Adverse Media Index.
\end{block}
\end{frame}

\begin{frame}{The False-Positive Crisis in Keyword Screening}
\textbf{The dominant paradigm (20 years):} match name + DOB + nationality against
a watchlist; flag any fuzzy match (Jaro--Winkler, Soundex) above a threshold.

\medskip
\begin{underhood}
Let $\mathcal{P}$ = true matches, $\mathcal{N}$ = true negatives, $\hat P$ = flagged set.
\[
\mathrm{FPR} = \frac{|\hat P \cap \mathcal{N}|}{|\mathcal{N}|}, \qquad
\mathrm{PPV} = \frac{|\hat P \cap \mathcal{P}|}{|\hat P|}.
\]
In production, AML screening runs at \textbf{FPR 95--99\%} --- fewer than 1 in 20
alerts is a genuine match.
\end{underhood}
\end{frame}

\begin{frame}{The False-Positive Crisis: Why It Compounds}
\begin{alertblock}{The cascade}
Alert noise trains staff to treat \emph{all} alerts as noise $\to$ genuine matches
cleared without scrutiny. Banks are fined both for \emph{missing} laundering and
(paradoxically) for generating \emph{excessive low-quality SARs}.
\end{alertblock}

\medskip
\textbf{Key insight:} a name match in isolation is not evidence. What matters is
whether the \emph{contextual} evidence is consistent with a launderer's profile
--- exactly what LLMs assess and keyword matching cannot.
\end{frame}

\begin{frame}{Agentic LLM Systems for Adverse Media Screening}
Rather than an analyst querying a database, an \textbf{agent} is given a subject
and runs a five-step loop:

\medskip
\begin{enumerate}
  \item \textbf{Query formulation} --- generate searches, incl.\ local-language variants
  \item \textbf{Document retrieval} --- top-$k$ articles, filtered by date + source tier
  \item \textbf{Relevance assessment} --- same individual? material? credible source?
  \item \textbf{Evidence extraction} --- offence type, jurisdiction, date, alleged vs.\ convicted
  \item \textbf{Risk scoring} --- structured tier (low/med/high/critical) + cited evidence
\end{enumerate}

\medskip
\begin{exampleblock}{``Carlos Fernandez-Gutierrez, b.\ 1971, MX''}
Agent searches EN + ES (``corruption'', ``money laundering'', firm name), reads
top-$k$ articles, disambiguates the individual, extracts structured offence
evidence, and emits an actionable risk report --- \textbf{minutes, not hours}.
\end{exampleblock}

\footnotesize Design choices: tool selection, memory across steps, and
error-handling for common-name collisions. ReAct loop $\to$ Lecture 4.
\end{frame}

\begin{frame}{RAG-Based Adverse Media Pipeline: Five Components}
\begin{underhood}
Ingestion $\to$ vector store $\to$ retriever $\to$ reader/ranker $\to$ formatter.
\end{underhood}

\medskip
\begin{tabular}{ll}
\toprule
\textbf{Component} & \textbf{Role} \\
\midrule
Ingestion & chunk 200--400 tok (overlap), embed, store + metadata \\
Vector store & dense embeddings \emph{and} BM25; incremental + deletable \\
Retriever & hybrid search; expand query with alias / transliterations \\
Reader / ranker & cross-encoder: disambiguate entity + classify relevance \\
Formatter & grammar-constrained JSON: tier, evidence, offences \\
\bottomrule
\end{tabular}

\medskip
\begin{block}{Production constraints}
Real-time onboarding: full pipeline $<$30s (pre-built indices, cached
embeddings). Batch re-screening: latency relaxed, but throughput must scale to
\emph{millions of records/day}.
\end{block}
\end{frame}

\begin{frame}{Hybrid Retrieval via Reciprocal Rank Fusion}
Dense (semantic) and sparse (BM25, exact terms) are fused without learned weights:
\[
\mathrm{RRF}(d) = \sum_{r \in R} \frac{1}{\kappa + \mathrm{rank}_r(d)},
\qquad \kappa = 60.
\]
Documents missed by a ranker get $\mathrm{rank} = \infty$.

\medskip
\begin{exampleblock}{Why fusion is robust}
A passage ranked highly by \emph{both} retrievers is promoted \emph{above} one
that any single retriever ranks first. RRF rewards cross-signal agreement ---
the property that makes hybrid retrieval beat either modality alone on legal /
news documents.
\end{exampleblock}

\medskip
\footnotesize \textbf{Worked numerically} in the practical session
(\texttt{gen\_rrf\_fusion.py} / \texttt{demo.ipynb}).
\end{frame}

\begin{frame}{The Adverse Media Index (AMI): Text $\to$ Calibrated Score}
A binary low/high label is insufficient. We want a continuous, comparable,
trackable score.

\medskip
\begin{underhood}
For documents $D=\{d_1,\dots,d_n\}$ retrieved for subject $s$, with relevance
$r_i$, credibility $c_i$, recency $t_i = e^{-\lambda(T-T_i)}$, and offence
indicators $\mathbf{e}_i \in \{0,1\}^K$:
\[
\mathrm{AMI}(s) = \sigma\!\left(\alpha + \sum_{i=1}^{n} r_i\, c_i\, t_i\,
\mathbf{w}^{\top}\mathbf{e}_i\right) \in [0,1].
\]
$\mathbf{w}$ = offence-severity weights (terrorism financing $\gg$ minor
violation); $\lambda$ = temporal decay (recency vs.\ history).
\end{underhood}

\medskip
\begin{itemize}
  \item $\mathbf{w}$ calibrated by logistic regression on ``subject later a true
        positive in an enforcement action''.
  \item Output is a probability $\to$ feeds the bank's risk-based customer rating,
        combined with transaction monitoring + KYC.
\end{itemize}
\end{frame}

\section{KYC and Sanctions Screening}
% ====================================================================

\begin{frame}{Entity Resolution: The Central Technical Challenge}
\begin{bigpicture}
KYC and sanctions screening reduce to one hard problem: do two imperfect,
multilingual, sparse records refer to the \emph{same} real-world entity?
\end{bigpicture}

\medskip
\textbf{Why it is hard:} transliteration variance (``Muammar Gaddafi'' has dozens
of renderings), name changes (maiden/married/patronymic), watchlist errors, and
common-name collisions in large populations.
\end{frame}

\begin{frame}{Entity Resolution: Pipeline and Threshold}
\begin{underhood}
Entity resolution as \emph{pair classification}: given records $x_1,x_2$, produce
a calibrated $P(\hat y = 1 \mid x_1,x_2)$.
\textbf{Two-stage pipeline:} fast ANN blocking ($O(n^2)\to O(nk)$) then a precise
cross-encoder / LLM scorer --- mirroring the retriever--ranker pattern.
\end{underhood}

\medskip
\begin{alertblock}{Threshold asymmetry}
For sanctions, set the threshold for \emph{near-100\% recall}: a missed true hit
costs far more than a false-positive review. LLMs add value by learning
transliteration implicitly and resolving ambiguity from auxiliary context.
\end{alertblock}
\end{frame}

\begin{frame}{PEP Screening and Beneficial Ownership}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{Politically Exposed Persons}
\begin{itemize}\footnotesize
  \item Heads of state, senior officials --- \emph{plus} family \& close
        associates (FaMCAs).
  \item FATF requires enhanced due diligence: source-of-wealth, senior approval.
  \item Universe is large \& changing; commercial DBs are costly, patchy in
        lower-income jurisdictions.
  \item LLMs read Wikipedia / gov sites / news $\to$ self-updating PEP registry;
        relation extraction finds FaMCAs.
\end{itemize}

\column{0.5\textwidth}
\textbf{Beneficial Ownership}
\begin{itemize}\footnotesize
  \item Shells, trusts, nominees hide the ultimate beneficial owner (UBO).
  \item FATF Rec.\ 24 + Corporate Transparency Act require registries --- but
        data is scattered.
  \item LLM extracts owners + stakes $\to$ directed ownership graph $\to$ traverse
        to UBO.
\end{itemize}
\end{columns}

\medskip
\begin{exampleblock}{Meridian Capital Holdings (UK)}
UK PSC $\to$ Luxembourg entity $\to$ BVI company (no public registry). LLM builds
the graph, flags the chain terminates in two opacity jurisdictions, cross-refs
adverse media --- hours of analyst work compressed to minutes.
\end{exampleblock}
\end{frame}

\section{Regulatory Reporting Automation}
% ====================================================================

\begin{frame}{Reporting Shares a Common Structure}
\begin{bigpicture}
A large bank files \emph{hundreds of thousands} of regulatory reports a year.
Despite apparent heterogeneity, all share one shape: \textbf{extract} quantitative
data $\to$ \textbf{classify} into a taxonomy $\to$ \textbf{generate} explanatory
prose. Each step suits a language model.
\end{bigpicture}

\medskip
\begin{tabular}{lll}
\toprule
\textbf{Report} & \textbf{Core LLM task} & \textbf{Key risk} \\
\midrule
XBRL filing & taxonomy concept tagging & 20k+ labels, sparsity \\
SAR & narrative drafting & factual error misleads LE \\
Stress test & narrative from numbers & numeric hallucination \\
\bottomrule
\end{tabular}

\medskip
\footnotesize Across all three: LLM produces \emph{drafts}; human review +
automated fact-checking are non-negotiable.
\end{frame}

\begin{frame}{XBRL Tagging as Retrieval, Not Classification}
The eXtensible Business Reporting Language is a standardised machine-readable
taxonomy (SEC since 2009; ESEF / iXBRL in the EU). Core operation: \emph{tag}
each line item with the correct concept.

\medskip
\begin{underhood}
The US GAAP taxonomy has \textbf{20,000+ elements} $\Rightarrow$ flat
classification fails (label sparsity). Frame it as retrieval + LLM verification:
\[
\hat c^* = \arg\max_{c \in \mathcal{C}}
\Big[\, \lambda\, \cos(\mathbf{v}_{\text{item}}, \mathbf{v}_c)
   + (1-\lambda)\, P_{\text{LLM}}(c \mid \text{item}, \text{context}) \,\Big].
\]
$\lambda$ interpolates embedding similarity vs.\ the LLM posterior; cross-validate
on labelled historical filings.
\end{underhood}

\medskip
\footnotesize Ambiguous items (``other comprehensive income'') + annual taxonomy
changes make manual tagging error-prone.
\end{frame}

\begin{frame}{SAR Drafting and Stress-Test Narratives}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{SAR drafting}
\begin{itemize}\footnotesize
  \item Filed with FinCEN / NCA / TRACFIN on suspicious activity.
  \item A skilled analyst spends \textbf{2--4 hours} per complex SAR.
  \item LLM drafts the narrative $\to$ analyst reviews / validates / supplements.
  \item A SAR mischaracterising activity misleads law enforcement $\to$ liability.
        Human review \emph{non-negotiable}; flag weak-evidence passages.
\end{itemize}

\column{0.5\textwidth}
\textbf{Stress-test narratives}
\begin{itemize}\footnotesize
  \item Supervisory submissions run to \emph{hundreds of pages} of prose.
  \item Quantitative results $\to$ template-constrained LLM commentary.
  \item RAG-ground in prior submissions for consistency.
  \item Stating ``ratio fell 3pp'' when output says 2.7pp is a \emph{material}
        error $\to$ post-hoc fact-check pass.
\end{itemize}
\end{columns}

\medskip
\begin{alertblock}{EU AI Act \& degree of automation}
An LLM that \emph{assists} (pre-populates a template for human review) $\to$
limited-risk. An LLM that \emph{files} a SAR autonomously $\to$ almost certainly
high-risk (full conformity assessment). Compliant designs adopt the former.
\end{alertblock}
\end{frame}

\section{Governance and Audit Trails}
% ====================================================================

\begin{frame}{Governance Is as Important as the Model}
\begin{bigpicture}
Compliance functions are governance structures, not just technical systems. A
technically excellent system with weak oversight \emph{fails} regulatory review;
a modest system with strong governance may \emph{pass}.
\end{bigpicture}

\medskip
\textbf{Explainability serves two distinct purposes:}
\begin{tabular}{ll}
\toprule
\textbf{Purpose} & \textbf{Explanation must be\dots} \\
\midrule
Regulatory (GDPR 22, FCRA) & lay-intelligible, actionable, top factors \\
Internal governance & precise, reproducible, validation-complete \\
\bottomrule
\end{tabular}

\medskip
\textbf{Three techniques (combine for high-stakes decisions):}
\begin{itemize}\footnotesize
  \item \textbf{Retrieval transparency} --- show top-3 grounding passages + scores
  \item \textbf{Chain-of-thought rationale} --- stored, auditable reasoning trace
  \item \textbf{Post-hoc attribution} --- SHAP on risk-tier classifier; attention
        weights are only a \emph{rough} guide, not standalone explanations
\end{itemize}
\end{frame}

\begin{frame}{Human-in-the-Loop: Designing Against Automation Bias}
\begin{alertblock}{The failure mode}
Poor HITL exhibits \emph{automation bias}: reviewers rubber-stamp the model,
especially with large queues, short review time, and discouraged overrides ---
oversight in form, not substance.
\end{alertblock}

\medskip
\textbf{Four design dimensions:}
\begin{itemize}
  \item \textbf{Task decomposition} --- give the human a task they do \emph{better}
        than the model (geopolitical context, disambiguation, escalation calls)
  \item \textbf{Uncertainty communication} --- surface low-confidence items
  \item \textbf{Override facilitation + feedback} --- log override reasons; feed
        back into monitoring (systematic divergence $\Rightarrow$ drift)
  \item \textbf{Queue design} --- order by risk severity $\times$ model confidence;
        bulk-clear confident low-risk cases with spot checks
\end{itemize}
\end{frame}

\begin{frame}{Model Risk Management for LLM Compliance Systems}
\begin{underhood}
SR 11-7's three pillars, extended for LLMs.
\end{underhood}

\medskip
\textbf{Development:} document conceptual soundness --- the model, why
retrieval-generation fits, scope/limits (languages, sources, entities), and
\emph{when not to use it}.

\smallskip
\textbf{Validation} (by an independent function):
\begin{itemize}\footnotesize
  \item Benchmark: precision / recall / F1 on labelled historical cases
  \item Adversarial: prompt injection, name-variation \& paraphrase evasion
  \item Calibration, bias auditing (across nationalities / name types), robustness
\end{itemize}
\end{frame}

\begin{frame}{Model Risk Management: Governance \& Monitoring}
\textbf{Governance:} model-inventory entry, change management (re-validate after
LLM / corpus updates), monitoring dashboard with alert thresholds, incident
response.

\medskip
\begin{exampleblock}{Deployed example thresholds}
Precision / recall tracked weekly on 200 senior-reviewed cases. \textbf{Precision
$<$60\%} or \textbf{recall $<$95\%} $\Rightarrow$ automatic alert + root-cause
review within 5 business days. Every decision (docs, output, reviewer choice)
retained \textbf{7 years} in a tamper-evident log.
\end{exampleblock}
\end{frame}

\section{Summary}
% ====================================================================

\begin{frame}{Lecture 11 Summary: Five Takeaways}
\begin{enumerate}
  \item \textbf{Frameworks overlap.} EU AI Act, MiFID II, Basel III, SR 11-7,
        GDPR --- map each application to the right tier; many compliance tools are
        high-risk. GDPR Art.\ 22 demands explanation + human review.
  \item \textbf{LLMs break the false-positive crisis} (FPR $>$95\%) by scoring
        \emph{contextual} consistency --- the Adverse Media Index turns text into a
        calibrated $[0,1]$ risk score.
  \item \textbf{RAG is the architecture of choice} --- hybrid BM25 + dense + RRF,
        cross-encoder reader. Powers adverse media, entity resolution, PEP, UBO.
  \item \textbf{Reporting is automatable with verification} --- XBRL, SAR, stress
        narratives as drafts + automated fact-checking against source data.
  \item \textbf{Governance $\approx$ technology.} HITL against automation bias,
        SR 11-7 adapted to LLMs, layered explainability.
\end{enumerate}

\medskip
\begin{block}{Next}
Practical 11 --- RRF by hand, an Adverse Media Index in Python, and a governance
design clinic.
\end{block}
\end{frame}

\begin{frame}[noframenumbering]{Appendix: RAG Pipeline Dataflow}
\small
\begin{enumerate}
  \item \textbf{Feeds} --- news, court filings, press releases (continuous).
  \item \textbf{Ingestion} --- chunk + embed (200--400 tok, overlap).
  \item \textbf{Vector store} --- dense + BM25, metadata (URL, date, source tier,
        entities); incremental updates + selective deletion (GDPR 17).
  \item \textbf{Retriever} --- subject query (+ alias variants) $\to$ hybrid + RRF
        $\to$ top-$k$.
  \item \textbf{Reader / ranker} --- cross-encoder reranking scores each passage
        against the query with full cross-attention (higher quality than the
        bi-encoder, higher latency).
  \item \textbf{Output formatter} --- grammar-constrained decoding $\Rightarrow$
        always-parseable JSON (no brittle parsing downstream).
\end{enumerate}

\medskip
Metadata fields are critical: they enable post-retrieval filtering that reduces
the LLM's ranking burden.
\end{frame}

\begin{frame}[noframenumbering]{Appendix: AMI Hyperparameters \& Calibration}
\[
\mathrm{AMI}(s) = \sigma\!\left(\alpha + \sum_{i=1}^{n} r_i\, c_i\, t_i\,
\mathbf{w}^{\top}\mathbf{e}_i\right), \qquad t_i = e^{-\lambda(T - T_i)}.
\]
\begin{itemize}
  \item $\mathbf{w} \in \mathbb{R}^K$ --- offence-severity weights; estimate by
        logistic regression on confirmed enforcement outcomes.
  \item $\lambda > 0$ --- decay rate (e.g.\ $0.01$/day). Aggressive decay
        emphasises recency; slow decay keeps historical evidence. Rehabilitated /
        acquitted subjects motivate decay.
  \item $\alpha$ --- baseline intercept; $\sigma$ maps to a probability of genuine
        AML risk, integrable into the customer risk rating.
\end{itemize}
\end{frame}

\begin{frame}[noframenumbering]{Appendix: Traditional Entity-Resolution Toolkit}
Before (and alongside) LLMs:
\begin{itemize}
  \item \textbf{String distance} --- edit distance, Jaro--Winkler.
  \item \textbf{Phonetic normalisation} --- Soundex, Metaphone, Double Metaphone
        (non-English names).
  \item \textbf{Blocking} --- compare only records sharing first-two characters to
        bound the comparison space.
\end{itemize}

\medskip
LLM advantages: (1) \emph{implicit} transliteration learned from multilingual
corpora --- generalises to scripts with no hand-coded rules; (2) \emph{context}
resolves ambiguity (``Carlos Salinas'' vs.\ ``Carlos Salinas de Gortari, former
Mexican president''). Recommended reading: the Fellegi--Sunter probabilistic
record-linkage model and transformer-based blocking/matching.
\end{frame}
