\section{The Regulatory Landscape for AI in Finance}

\begin{frame}{The Regulatory Landscape for AI in Finance}
  \begin{itemize}
    \item The EU AI Act classifies financial AI systems by risk level, imposing strict requirements on high-risk applications
    \item MiFID II and Basel III set model governance standards; SR 11-7 defines model risk management expectations for banks
    \item GDPR constrains LLM use by limiting personal data retention, right-to-erasure, and automated decision-making
    \item Compliance officers must navigate overlapping and sometimes conflicting regulatory regimes across jurisdictions
  \end{itemize}
\end{frame}

\section{Anti-Money Laundering with LLMs}

\begin{frame}{Anti-Money Laundering with LLMs}
  \begin{itemize}
    \item Keyword-based AML screening generates overwhelming false-positive rates, creating analyst bottlenecks
    \item Agentic LLM systems can autonomously gather, assess, and escalate adverse media findings at scale
    \item RAG-based pipelines retrieve up-to-date news and regulatory databases to ground LLM screening decisions
    \item Adverse Media Index scores quantify reputational and financial crime risk from unstructured text
  \end{itemize}
\end{frame}

\section{KYC and Sanctions Screening}

\begin{frame}{KYC and Sanctions Screening}
  \begin{itemize}
    \item Entity resolution disambiguates names across transliterations, aliases, and typographic variants
    \item PEP screening requires matching customers against politically exposed persons lists from multiple sources
    \item Beneficial ownership extraction from legal filings identifies ultimate controlling parties behind corporate structures
    \item LLMs improve recall and reduce manual review time compared to rule-based matching approaches
  \end{itemize}
\end{frame}

\section{Regulatory Reporting Automation}

\begin{frame}{Regulatory Reporting Automation}
  \begin{itemize}
    \item XBRL tagging maps financial statement line items to taxonomy concepts, a task well-suited to LLM assistance
    \item SAR drafting benefits from LLMs that can synthesize transaction narratives into structured, regulator-ready reports
    \item Stress test narrative automation generates explanatory text from quantitative scenario outputs
    \item Automated reporting reduces compliance costs while improving consistency and auditability
  \end{itemize}
\end{frame}

\section{Governance and Audit Trails}

\begin{frame}{Governance and Audit Trails}
  \begin{itemize}
    \item Explainability requirements demand that compliance decisions be traceable and interpretable to regulators
    \item Human-in-the-loop workflows ensure that high-stakes decisions retain meaningful human oversight
    \item Model risk management frameworks must be extended to cover LLM-specific failure modes and biases
    \item Comprehensive audit trails are essential for demonstrating regulatory compliance and managing liability
  \end{itemize}
\end{frame}
