```bash
#!/bin/bash
# validate-hooks.sh — checks all 13 hook scripts exist, are executable, and have required header sections
set -euo pipefail

PASS=0; FAIL=0

check_hook() {
  local name="$1"
  local file=".claude/hooks/${name}"
  if [ ! -f "$file" ]; then
    echo "MISS  $file"
    FAIL=$((FAIL+1))
    return
  fi
  local ok=1
  # Must start with shebang
  if ! head -1 "$file" | grep -q '^#!/bin/bash'; then
    echo "NO_SHEBANG  ${name}"
    FAIL=$((FAIL+1))
    ok=0
  fi
  # Must have a Trigger comment
  if ! grep -q '# Trigger:' "$file"; then
    echo "NO_TRIGGER  ${name}"
    FAIL=$((FAIL+1))
    ok=0
  fi
  # Must have an Exit comment
  if ! grep -q '# Exit:' "$file"; then
    echo "NO_EXIT  ${name}"
    FAIL=$((FAIL+1))
    ok=0
  fi
  # Must be executable
  if [ ! -x "$file" ]; then
    echo "NOT_EXEC  ${name}"
    FAIL=$((FAIL+1))
    ok=0
  fi
  if [ "$ok" -eq 1 ]; then
    echo "  OK  ${name}"
    PASS=$((PASS+1))
  fi
}

# Category 1: Integrity hooks
```

```bash
#!/bin/bash
# check-latex.sh — runs pdflatex in draft mode to detect compilation errors before commit
# Trigger: pre-commit
# Exit: 0=pass, 1=block (LaTeX errors found)

set -euo pipefail

# Only run if book/main.tex exists
if [ ! -f "book/main.tex" ]; then
  exit 0
fi

# Only run if staged files include .tex files
STAGED_TEX=$(git diff --cached --name-only --diff-filter=ACM | grep '\.tex$' || true)
if [ -z "$STAGED_TEX" ]; then
  echo "LaTeX check: no staged .tex files, skipping."
  exit 0
fi

echo "LaTeX check: running pdflatex draft compilation..."

# Run pdflatex in draft mode (no PDF output, fast, just error detection)
# Run from book/ so relative \input paths resolve correctly
cd book
pdflatex -draftmode -interaction=nonstopmode main.tex > /dev/null 2>&1 || true
cd ..

# Parse the log for hard errors (lines starting with "! ")
LOG="book/main.log"
if [ ! -f "$LOG" ]; then
  echo "LaTeX check: no log file found — pdflatex may not be installed."
  exit 0
fi

ERRORS=$(grep '^! ' "$LOG" || true)
if [ -n "$ERRORS" ]; then
  echo "LaTeX errors found:"
  echo "$ERRORS" | while IFS= read -r line; do
    echo "  $line"
  done
  echo "Commit blocked: fix LaTeX errors first"
  exit 1
fi

echo "LaTeX check: OK"
exit 0
```

```bash
#!/bin/bash
# check-refs.sh — scans staged .tex files for \ref{} and \eqref{} calls that have no matching \label{}
# Trigger: pre-commit
# Exit: 0=pass, 1=block (undefined references found)

set -euo pipefail

# Collect staged .tex files
STAGED_TEX=$(git diff --cached --name-only --diff-filter=ACM | grep '\.tex$' || true)
if [ -z "$STAGED_TEX" ]; then
  echo "Reference check: no staged .tex files, skipping."
  exit 0
fi

# Build the complete set of \label{KEY} definitions from all .tex files in book/
# (labels might be defined in a non-staged file and referenced in a staged one)
LABEL_SET=$(find book/ -name '*.tex' -exec grep -h '\\label{' {} \; 2>/dev/null \
  | grep -o '\\label{[^}]*}' \
  | sed 's/\\label{//;s/}//' \
  | sort -u || true)

FAILED=0

for TEX_FILE in $STAGED_TEX; do
  [ -f "$TEX_FILE" ] || continue

  # Extract all \ref{KEY} and \eqref{KEY} from this staged file with line numbers
  while IFS=: read -r linenum ref_expr; do
    KEY=$(echo "$ref_expr" | grep -o '{[^}]*}' | head -1 | tr -d '{}')
    if [ -z "$KEY" ]; then
      continue
    fi
    # Check if KEY is in our label set
    if ! echo "$LABEL_SET" | grep -qx "$KEY"; then
      echo "Undefined ref: \\ref{$KEY} in $TEX_FILE:$linenum"
      FAILED=$((FAILED+1))
    fi
  done < <(grep -n '\\[e]*ref{' "$TEX_FILE" 2>/dev/null || true)
done

if [ "$FAILED" -gt 0 ]; then
  echo "Commit blocked: $FAILED undefined reference(s) found"
  exit 1
fi

echo "Reference check: OK"
exit 0
```

```bash
#!/bin/bash
# check-bib.sh — validates BibTeX entries have all required fields
# Trigger: pre-commit
# Exit: 0=pass, 1=block (invalid entries found)

set -euo pipefail

BIB="book/bibliography.bib"

# Nothing to check if the file does not exist
if [ ! -f "$BIB" ]; then
  echo "Bibliography check: no bibliography.bib found, skipping."
  exit 0
fi

# Only run if bibliography.bib or a .tex file is staged
STAGED_BIB=$(git diff --cached --name-only --diff-filter=ACM | grep -E '(\.bib$|\.tex$)' || true)
if [ -z "$STAGED_BIB" ]; then
  echo "Bibliography check: no staged .bib or .tex files, skipping."
  exit 0
fi

echo "Bibliography check: validating entries in $BIB..."

PROBLEMS=0
CURRENT_TYPE=""
CURRENT_KEY=""
declare -A FIELDS

flush_entry() {
  local type="$1"
  local key="$2"
  local -n fields_ref=$3
  local missing=""

  case "$type" in
    article)
      for field in author title journal year; do
        [ -z "${fields_ref[$field]:-}" ] && missing="$missing $field"
      done
      ;;
    book)
      for field in author title publisher year; do
        [ -z "${fields_ref[$field]:-}" ] && missing="$missing $field"
      done
      ;;
    inproceedings)
      for field in author title booktitle year; do
        [ -z "${fields_ref[$field]:-}" ] && missing="$missing $field"
      done
      ;;
  esac

  if [ -n "$missing" ]; then
    echo "  INVALID @${type}{${key}}: missing fields:$missing"
    return 1
  fi
  return 0
}

while IFS= read -r line; do
  # Detect entry opening: @article{key, or @book{key, etc.
  if echo "$line" | grep -qiE '^@(article|book|inproceedings)\{'; then
    # If we were tracking a previous entry, evaluate it
    if [ -n "$CURRENT_TYPE" ]; then
      flush_entry "$CURRENT_TYPE" "$CURRENT_KEY" FIELDS || PROBLEMS=$((PROBLEMS+1))
    fi
    CURRENT_TYPE=$(echo "$line" | grep -ioE '@(article|book|inproceedings)' | tr -d '@' | tr '[:upper:]' '[:lower:]')
    CURRENT_KEY=$(echo "$line" | sed 's/.*{//' | sed 's/,.*//' | tr -d ' ')
    declare -A FIELDS=()
    continue
  fi

  # Detect closing brace of an entry
  if [ "$line" = "}" ] && [ -n "$CURRENT_TYPE" ]; then
    flush_entry "$CURRENT_TYPE" "$CURRENT_KEY" FIELDS || PROBLEMS=$((PROBLEMS+1))
    CURRENT_TYPE=""
    CURRENT_KEY=""
    declare -A FIELDS=()
    continue
  fi

  # Detect field = value lines
  if [ -n "$CURRENT_TYPE" ]; then
    FIELD_NAME=$(echo "$line" | grep -oE '^\s*[a-zA-Z]+\s*=' | tr -d ' =' | tr '[:upper:]' '[:lower:]')
    FIELD_VAL=$(echo "$line" | sed 's/^[^=]*=//' | tr -d ' {},"')
    if [ -n "$FIELD_NAME" ] && [ -n "$FIELD_VAL" ]; then
      FIELDS[$FIELD_NAME]="$FIELD_VAL"
    fi
  fi
done < "$BIB"

# Flush the last entry if file doesn't end with a closing brace line
if [ -n "$CURRENT_TYPE" ]; then
  flush_entry "$CURRENT_TYPE" "$CURRENT_KEY" FIELDS || PROBLEMS=$((PROBLEMS+1))
fi

if [ "$PROBLEMS" -gt 0 ]; then
  echo "Commit blocked: $PROBLEMS invalid BibTeX entry/entries found"
  exit 1
fi

echo "Bibliography check: OK"
exit 0
```

```bash
#!/bin/bash
# check-notation.sh — advisory check for undefined math commands in staged .tex files
# Trigger: pre-commit
# Exit: 0=always (warnings only, never blocks)

set -euo pipefail

STAGED_TEX=$(git diff --cached --name-only --diff-filter=ACM | grep '\.tex$' || true)
if [ -z "$STAGED_TEX" ]; then
  exit 0
fi

# Build set of user-defined commands from preamble.tex
DEFINED=""
if [ -f "book/preamble.tex" ]; then
  DEFINED=$(grep -E '\\(newcommand|DeclareMathOperator)\{' book/preamble.tex \
    | grep -oE '\{\\[A-Za-z]+\}' \
    | tr -d '{}' \
    | sort -u || true)
fi

# Standard LaTeX math commands — not flagged as unknown
STANDARD="frac sum int prod lim sup inf mathbb mathcal mathrm mathbf text left right cdot ldots dots
  alpha beta gamma delta epsilon zeta eta theta iota kappa lambda mu nu xi pi rho sigma tau upsilon
  phi chi psi omega Gamma Delta Theta Lambda Xi Pi Sigma Upsilon Phi Psi Omega
  partial nabla infty forall exists in notin subset supset cup cap setminus emptyset
  to rightarrow leftarrow Rightarrow Leftarrow iff leq geq neq approx sim simeq equiv
  pm mp times div circ oplus otimes langle rangle vert Vert hat tilde bar vec
  overline underline sqrt exp log ln det tr rank dim ker im arg max min
  begin end label ref eqref cite footnote section subsection subsubsection
  item enumerate itemize equation align gather split theorem lemma proof definition
  corollary remark example proposition"

WARNINGS=0

for TEX_FILE in $STAGED_TEX; do
  [ -f "$TEX_FILE" ] || continue

  # Extract math mode content between $ ... $ and \[ ... \]
  # Then find all \commandname tokens
  MATH_COMMANDS=$(grep -oP '(?<=\$)[^$]+(?=\$)|\\\[[^\]]+\\\]' "$TEX_FILE" 2>/dev/null \
    | grep -oE '\\[A-Za-z]+' \
    | sed 's/^\\//' \
    | sort -u || true)

  for CMD in $MATH_COMMANDS; do
    # Skip if it's a standard LaTeX command
    if echo "$STANDARD" | grep -qw "$CMD"; then
      continue
    fi
    # Skip if it's defined in preamble.tex
    if echo "$DEFINED" | grep -qx "\\\\${CMD}"; then
      continue
    fi
    echo "NOTATION WARNING: \\${CMD} used in $TEX_FILE but not defined in preamble.tex"
    WARNINGS=$((WARNINGS+1))
  done
done

if [ "$WARNINGS" -gt 0 ]; then
  echo "Notation check: $WARNINGS advisory warning(s) — not blocking commit"
fi

# Always exit 0 — this is advisory only
exit 0
```

```bash
#!/bin/bash
# check-numbering.sh — advisory check for unpaired chapter/lecture directories
# Trigger: pre-commit
# Exit: 0=always (warnings only, never blocks)

set -euo pipefail

# Extract two-digit prefixes from book/chapters/NN-* directories
CHAPTER_NUMS=""
if [ -d "book/chapters" ]; then
  CHAPTER_NUMS=$(find book/chapters -maxdepth 1 -type d -name '[0-9][0-9]-*' \
    | xargs -I{} basename {} \
    | grep -oE '^[0-9]{2}' \
    | sort -u || true)
fi

# Extract two-digit prefixes from course/lectures/NN-* directories
LECTURE_NUMS=""
if [ -d "course/lectures" ]; then
  LECTURE_NUMS=$(find course/lectures -maxdepth 1 -type d -name '[0-9][0-9]-*' \
    | xargs -I{} basename {} \
    | grep -oE '^[0-9]{2}' \
    | sort -u || true)
fi

WARNINGS=0

# Find chapter numbers that have no matching lecture
for NUM in $CHAPTER_NUMS; do
  if ! echo "$LECTURE_NUMS" | grep -qx "$NUM"; then
    echo "NUMBERING WARNING: chapter $NUM exists in book/chapters/ but no matching lecture in course/lectures/"
    WARNINGS=$((WARNINGS+1))
  fi

# Find lecture numbers that have no matching chapter
for NUM in $LECTURE_NUMS; do
  if ! echo "$CHAPTER_NUMS" | grep -qx "$NUM"; then
    echo "NUMBERING WARNING: lecture $NUM exists in course/lectures/ but no matching chapter in book/chapters/"
    WARNINGS=$((WARNINGS+1))
  fi
done

if [ "$WARNINGS" -gt 0 ]; then
  echo "Numbering check: $WARNINGS advisory warning(s) — not blocking commit"
else
  echo "Numbering check: OK (all chapters and lectures are paired)"
fi

# Always exit 0 — this is advisory only
exit 0
```

```bash
#!/bin/bash
# auto-commit.sh — auto-commits a written file if auto_commit=true in TOPIC.md
# Trigger: post-tool-use (Write or Edit); receives absolute file path as $1
# Exit: 0=pass/continue, 1=block/fail

# Intentionally no set -e here — git failures must be caught explicitly

FILE="$1"

# Read auto_commit setting from TOPIC.md
AUTO_COMMIT=$(grep 'auto_commit:' TOPIC.md 2>/dev/null | sed 's/auto_commit: //' | tr -d ' ' || echo "false")
if [ "$AUTO_COMMIT" != "true" ]; then
  exit 0
fi

# Validate that a file path was provided and the file exists
if [ -z "$FILE" ]; then
  echo "auto-commit: no file path provided, skipping."
  exit 0
fi
if [ ! -f "$FILE" ]; then
  echo "auto-commit: file '$FILE' does not exist, skipping."
  exit 0
fi

# Determine commit type and scope from file path
BASENAME=$(basename "$FILE")
if echo "$FILE" | grep -qE 'book/chapters/[0-9]{2}-'; then
  NN=$(echo "$FILE" | grep -oE '[0-9]{2}-[^/]+' | head -1 | grep -oE '^[0-9]{2}')
  SCOPE="ch${NN}"
  TYPE="feat"
elif echo "$FILE" | grep -qE 'course/lectures/[0-9]{2}-'; then
  NN=$(echo "$FILE" | grep -oE '[0-9]{2}-[^/]+' | head -1 | grep -oE '^[0-9]{2}')
  SCOPE="lec${NN}"
  TYPE="feat"
elif echo "$FILE" | grep -qE 'code/notebooks/[0-9]{2}-'; then
  NN=$(echo "$FILE" | grep -oE '[0-9]{2}-[^/]+' | head -1 | grep -oE '^[0-9]{2}')
  SCOPE="code-${NN}"
  TYPE="feat"
elif echo "$FILE" | grep -q '\.claude/agents/'; then
  SCOPE="agents"
  TYPE="chore"
elif echo "$FILE" | grep -q '\.claude/skills/'; then
  SCOPE="skills"
  TYPE="chore"
elif echo "$FILE" | grep -q '\.claude/hooks/'; then
  SCOPE="hooks"
  TYPE="chore"
else
  SCOPE=""
  TYPE="chore"
fi

if [ -n "$SCOPE" ]; then
  MSG="${TYPE}(${SCOPE}): update ${BASENAME}"
else
  MSG="${TYPE}: update ${BASENAME}"
fi

git add "$FILE" 2>/dev/null || true
git commit -m "$MSG" 2>/dev/null || true  # silently ignore "nothing to commit"

exit 0
```

```bash
#!/bin/bash
# update-status.sh — regenerates docs/STATUS.md from quality report JSON files
# Trigger: post-commit
# Exit: 0=pass/continue, 1=block/fail

set -euo pipefail

STATUS_FILE="docs/STATUS.md"

# Ensure docs/ exists
mkdir -p docs

# Write the STATUS.md header
cat > "$STATUS_FILE" << 'EOF'
# Project Status

Auto-generated after each commit. Do not edit by hand — re-generated by update-status.sh.

| Chapter | Clarity | Rigor | Completeness | Pedagogy | Style | Pass | Released |
|---------|---------|-------|--------------|----------|-------|------|----------|
EOF

# Find all chapter directories, sorted
CHAPTERS=$(find book/chapters -maxdepth 1 -type d -name '[0-9][0-9]-*' 2>/dev/null | sort || true)

if [ -z "$CHAPTERS" ]; then
  echo "| (no chapters yet) | — | — | — | — | — | — | — |" >> "$STATUS_FILE"
else
  for CHAPTER_DIR in $CHAPTERS; do
    DIR_NAME=$(basename "$CHAPTER_DIR")
    NN=$(echo "$DIR_NAME" | grep -oE '^[0-9]{2}')
    TOPIC_NAME=$(echo "$DIR_NAME" | sed 's/^[0-9]*-//')
    SCORE_FILE="docs/quality/${DIR_NAME}-score.json"

    if [ -f "$SCORE_FILE" ]; then
      # Parse scores with python3
      ROW=$(python3 - "$SCORE_FILE" << 'PYEOF'
import json, sys
path = sys.argv[1]
try:
    d = json.load(open(path))
    scores = d.get("scores", {})
    clarity     = scores.get("clarity",      {}).get("score", "—")
    rigor       = scores.get("rigor",        {}).get("score", "—")
    completeness= scores.get("completeness", {}).get("score", "—")
    pedagogy    = scores.get("pedagogy",     {}).get("score", "—")
    style       = scores.get("style",        {}).get("score", "—")
    passed      = "YES" if d.get("pass", False) else "NO"
    released    = "YES" if d.get("released", False) else "—"
    print(f"{clarity}|{rigor}|{completeness}|{pedagogy}|{style}|{passed}|{released}")
except Exception as e:
    print("—|—|—|—|—|—|—")
PYEOF
)
      CLARITY=$(echo "$ROW" | cut -d'|' -f1)
      RIGOR=$(echo "$ROW" | cut -d'|' -f2)
      COMPLETENESS=$(echo "$ROW" | cut -d'|' -f3)
```

```bash
#!/bin/bash
# gate-check.sh — blocks commit if any staged content file has quality scores below threshold
# Trigger: pre-commit
# Exit: 0=pass, 1=block (quality gate failed)

set -euo pipefail

# Read threshold from TOPIC.md (default 7)
THRESHOLD=$(grep 'quality_threshold:' TOPIC.md 2>/dev/null | sed 's/quality_threshold: //' | tr -d ' ' || echo "7")
THRESHOLD=${THRESHOLD:-7}

# Find staged .tex and .md content files (exclude non-content directories)
STAGED=$(git diff --cached --name-only --diff-filter=ACM 2>/dev/null \
  | grep -E '\.(tex|md)$' \
  | grep -vE '^(\.claude/|scripts/|hooks/|docs/)' \
  || true)

if [ -z "$STAGED" ]; then
  echo "Quality gate: no staged content files, skipping."
  exit 0
fi

FAILURES=0
WARNINGS=0

for FILE in $STAGED; do
  # Derive the score file path from the file's directory name
  DIR_NAME=$(dirname "$FILE" | xargs -I{} basename {})
  SCORE_FILE="docs/quality/${DIR_NAME}-score.json"

  if [ ! -f "$SCORE_FILE" ]; then
    echo "QUALITY WARNING: no score file for $FILE — run /score-content $FILE first"
    WARNINGS=$((WARNINGS+1))
    continue
  fi

  # Parse the score file with python3 and check each dimension
  RESULT=$(python3 - "$SCORE_FILE" "$THRESHOLD" << 'PYEOF'
import json, sys
path = sys.argv[1]
threshold = float(sys.argv[2])
try:
    d = json.load(open(path))
    scores = d.get("scores", {})
    failed = []
    for dim, val in scores.items():
        score = val.get("score", 10)
        if score < threshold:
            failed.append(f"{dim}={score}")
    if failed:
        print("FAIL:" + ",".join(failed))
    else:
        print("PASS")
except Exception as e:
    print(f"ERROR:{e}")
PYEOF
)

  if echo "$RESULT" | grep -q '^FAIL:'; then
    DIMS=$(echo "$RESULT" | sed 's/^FAIL://')
    echo "QUALITY GATE FAILED: $FILE — dimensions below ${THRESHOLD}: $DIMS"
    FAILURES=$((FAILURES+1))
  elif echo "$RESULT" | grep -q '^ERROR:'; then
    echo "QUALITY WARNING: could not parse score for $FILE: $RESULT"
    WARNINGS=$((WARNINGS+1))
  else
    echo "Quality gate: $FILE — OK"
  fi
done

if [ "$FAILURES" -gt 0 ]; then
  echo ""
  echo "Commit blocked: $FAILURES file(s) failed quality gate (threshold: $THRESHOLD)"
  echo "Run /refine-until-threshold or /score-content to improve scores."
  exit 1
fi

if [ "$WARNINGS" -gt 0 ]; then
  echo "Quality gate: $WARNINGS advisory warning(s) — commit allowed"
fi

echo "Quality gate: OK (all scores >= ${THRESHOLD})"
exit 0
```

```bash
  echo "Run /score-content on some content files first."
  exit 0
fi

SCORE_FILES=$(find "$QUALITY_DIR" -name '*-score.json' | sort || true)

if [ -z "$SCORE_FILES" ]; then
  echo "No score files found in $QUALITY_DIR"
  echo "Run /score-content on some content files first."
  exit 0
fi

# Print table header
printf "%-30s  %7s  %5s  %12s  %8s  %5s  %4s  %8s\n" \
  "Chapter" "Clarity" "Rigor" "Completeness" "Pedagogy" "Style" "Pass" "Released"
printf "%s\n" "$(printf '=%.0s' {1..80})"

for SCORE_FILE in $SCORE_FILES; do
  BASENAME=$(basename "$SCORE_FILE" "-score.json")

  python3 - "$SCORE_FILE" "$BASENAME" << 'PYEOF'
import json, sys

path = sys.argv[1]
name = sys.argv[2]

try:
    d = json.load(open(path))
    scores = d.get("scores", {})
    clarity      = scores.get("clarity",      {}).get("score", "—")
    rigor        = scores.get("rigor",        {}).get("score", "—")
    completeness = scores.get("completeness", {}).get("score", "—")
    pedagogy     = scores.get("pedagogy",     {}).get("score", "—")
    style        = scores.get("style",        {}).get("score", "—")
    passed       = "YES" if d.get("pass", False) else "NO"
    released     = "YES" if d.get("released", False) else "—"
    print(f"{name:<30}  {str(clarity):>7}  {str(rigor):>5}  {str(completeness):>12}  {str(pedagogy):>8}  {str(style):>5}  {passed:>4}  {released:>8}")
except Exception as e:
    print(f"{name:<30}  {'ERR':>7}  {'ERR':>5}  {'ERR':>12}  {'ERR':>8}  {'ERR':>5}  {'ERR':>4}  {'ERR':>8}")
    print(f"  (parse error: {e})", file=__import__('sys').stderr)
PYEOF

done

echo ""
echo "Total score files: $(echo "$SCORE_FILES" | wc -l | tr -d ' ')"
echo "Run /topic-status in a Claude Code session for the same output with AI context."
exit 0
```
