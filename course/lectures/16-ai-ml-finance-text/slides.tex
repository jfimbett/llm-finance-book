\section{Artificial Intelligence: Definitions and Scope}
% =====================================================================

\begin{frame}{What Do We Mean by ``Artificial Intelligence''?}

\begin{bigpicture}
``AI'' is coined by John McCarthy (1956): making machines perform tasks that
\emph{would require intelligence if performed by humans}. The boundary is
socially constructed --- once a machine does it reliably, we call it ``mere
computation'' (the \alert{AI effect}). The field chases a moving target.
\end{bigpicture}

\medskip
\textbf{Our working hierarchy for this book:}
\begin{itemize}
  \item \textbf{AI} --- the broad enterprise of building systems that perform cognitive tasks.
  \item \textbf{Machine learning} --- the dominant contemporary methodology.
  \item \textbf{Deep learning / LLMs} --- particular implementations of that methodology.
\end{itemize}
\end{frame}

\begin{frame}{Learning Objectives for This Lecture}

\begin{block}{By the end of this lecture you should be able to}
\begin{itemize}
  \item Place AI/ML/DL/LLM in a nested hierarchy.
  \item Contrast symbolic vs.\ statistical AI.
  \item Identify where AI creates value in finance.
  \item Categorise sources of financial text.
  \item Trace NLP from bag-of-words to transformers.
  \item State scaled dot-product attention.
\end{itemize}
\end{block}
\end{frame}

\begin{frame}{From Expert Systems to Learning Machines}

\begin{columns}[T]
\begin{column}{0.5\textwidth}
\textbf{1956--1980s: Expert systems}
\begin{itemize}
  \item Hand-crafted ``if-then'' rules encoding domain knowledge.
  \item A 1980s credit system: hundreds of rules for ``high-risk applicant''.
  \item \textcolor{forestgreen}{Strength:} transparent --- every decision traces to a rule.
  \item \textcolor{crimson}{Weakness:} brittle --- rules proliferate, fail on the unanticipated.
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\textbf{Late 1980s onward: induced from data}
\begin{itemize}
  \item Intelligence \emph{learned} from data, not \emph{encoded} by hand.
  \item Larger datasets + faster hardware made empirical evaluation feasible.
  \item By the mid-2000s data-driven methods overtook symbolic ones wherever labeled data existed:
        speech, vision, spam, and eventually NLP.
\end{itemize}
\end{column}
\end{columns}

\medskip
\begin{block}{The shift in one sentence}
The brittleness of rule-based systems motivated a move toward approaches in
which knowledge is \alert{induced from data rather than encoded by hand}.
\end{block}
\end{frame}

\begin{frame}{Symbolic vs.\ Statistical AI}

\begin{columns}[T]
\begin{column}{0.5\textwidth}
\begin{block}{Symbolic AI}
Represents the world through discrete, human-readable \textbf{symbols};
reasons by applying explicit \textbf{rules}. Knowledge encoded by experts
\emph{before} deployment; static as new data arrive.\\[2pt]
\textit{Home:} logic programming, semantic networks, knowledge graphs.
\end{block}
\end{column}
\begin{column}{0.5\textwidth}
\begin{block}{Statistical AI}
Represents the world through numerical \textbf{parameters} estimated from data;
predicts by evaluating a \textbf{learned function}. Knowledge implicit in
parameters; \emph{changes} on retraining.
\end{block}
\end{column}
\end{columns}

\medskip
\begin{exampleblock}{Not mutually exclusive --- the hybrid is common in finance}
A statistical model flags an anomalous transaction; a \emph{deterministic}
compliance rule decides the regulatory report. We lean statistical (markets are
data-rich) but return to symbolic methods for interpretability
(explainability chapter) and regulatory constraints.
\end{exampleblock}
\end{frame}

\section{The AI--ML--Deep Learning Hierarchy}
% =====================================================================

\begin{frame}{Machine Learning: Learning from Data}

\begin{bigpicture}
ML = algorithms that improve at a task through \emph{experience} (exposure to
data). The central desideratum is \alert{generalization}: perform well not just
on seen data but on new inputs from the same distribution.
\end{bigpicture}

\medskip
\begin{underhood}
\textbf{Supervised learning.} Given a training set $\{(\mathbf{x}_i, y_i)\}_{i=1}^n$,
learn $f_\theta:\mathcal{X}\to\mathcal{Y}$ by minimizing
\[
  \mathcal{L}(\theta) = \frac{1}{n}\sum_{i=1}^{n}\ell\big(f_\theta(\mathbf{x}_i), y_i\big),
\]
with per-sample loss $\ell$ (cross-entropy, squared error). Covers
\emph{classification} and \emph{regression} --- the two workhorses of quant finance.
\end{underhood}
\end{frame}

\begin{frame}{Three Learning Paradigms}

\begin{bigpicture}
The same generalization goal recurs under three supervisory regimes, each mapping
to a different class of financial problem.
\end{bigpicture}

\medskip
\begin{itemize}
  \item \textbf{Supervised:} credit scoring, default prediction, factor estimation
        --- learn from labeled $(\mathbf{x}_i, y_i)$ pairs.
  \item \textbf{Unsupervised:} clustering trades into market regimes --- structure
        without labels.
  \item \textbf{Reinforcement:} maximise cumulative reward (trading, rebalancing)
        --- learn from interaction.
\end{itemize}
\end{frame}

\begin{frame}{The Cost of Learning from Data: Distribution Shift}

\begin{alertblock}{Why the learning paradigm matters in finance}
A supervised credit model needs \emph{labeled historical defaults} and is
constrained to the distribution of its training period. A model fit on a benign
credit cycle can badly \textbf{underestimate tail risk} in a crisis --- an
instance of \alert{distribution shift} with practical \emph{and regulatory} consequences.
\end{alertblock}

\medskip
\textbf{Practical implications threaded through the book:}
\begin{itemize}
  \item Generalization challenges pervade financial ML.
  \item They motivate careful cross-validation (purging look-ahead bias).
  \item Reference: \emph{Advances in Financial Machine Learning}
        (Lopez de Prado, 2018).
\end{itemize}
\end{frame}

\begin{frame}{Deep Learning: Representation at Scale}

\textbf{Deep learning} = ML with neural networks deep enough that the model
\emph{learns intermediate representations automatically} --- no hand-specified features.

\medskip
\begin{underhood}
\textbf{Universal approximation:} hierarchical compositions of simple nonlinear
functions represent arbitrarily complex mappings given enough width/depth.
\emph{Depth} confers a specific advantage:
\begin{itemize}
  \item Lower layers: general, transferable features (edges, phonemes, collocations).
  \item Higher layers: task-specific combinations.
\end{itemize}
This makes \textbf{transfer learning} efficient: pre-train broadly, fine-tune on
a narrow task with far fewer labeled examples.
\end{underhood}
\end{frame}

\begin{frame}{Deep Learning: The Market Analogy}

\begin{bigpicture}
\textbf{Deep nets $\leftrightarrow$ markets.} Both aggregate information
hierarchically across interacting units / layers, and both are hard to interpret
from the outside --- motivating the explainability methods later in the book.
\end{bigpicture}

\medskip
\begin{block}{Why depth matters in finance}
The same hierarchy that makes deep nets powerful --- transferable low-level
features composed into task-specific representations --- is what makes a single
pre-trained backbone reusable across many financial text tasks, the theme of the
pre-training and transfer-learning slides that follow.
\end{block}
\end{frame}

\begin{frame}{Large Language Models as a Special Case}

\begin{underhood}
A \textbf{token} is the atomic unit of text; the \textbf{vocabulary}
$\mathcal{V}$ is the finite set of such units. An LLM is a network with
\emph{billions} of parameters, trained on \emph{hundreds of billions} of tokens,
to assign probabilities to token sequences. Canonical objective --- next-token prediction:
\[
  \mathcal{L}_{\mathrm{LM}} = -\sum_{t=1}^{T}\log P_\theta\big(x_t \mid x_1,\dots,x_{t-1}\big).
\]
\end{underhood}
\end{frame}

\begin{frame}{What Makes LLMs Special}

\textbf{What makes LLMs \emph{special}:} scale $+$ architecture $+$ \alert{emergent abilities}
--- capabilities appearing abruptly with scale (Wei et al., 2022):
\begin{itemize}
  \item Multi-step reasoning, instruction following.
  \item \textbf{In-context learning:} adapt to a task from a few prompt examples, no parameter update.
\end{itemize}

\medskip
\begin{block}{Consequence for finance}
A \emph{single} LLM is a flexible substrate for earnings-call summarization,
regulatory parsing, and strategy explanation --- e.g.\ BloombergGPT (Wu et al., 2023).
\end{block}
\end{frame}

\section{AI Applications in Finance}
% =====================================================================

\begin{frame}{Decision-Making Under Uncertainty}

\begin{bigpicture}
Finance is fundamentally \emph{decision-making under uncertainty}: form beliefs
about future states, then act subject to constraints. AI contributes at \emph{two} levels.
\end{bigpicture}

\medskip
\begin{columns}[T]
\begin{column}{0.5\textwidth}
\begin{block}{Quality of beliefs}
Better forecasts; richer models of uncertainty.
\end{block}
\end{column}
\begin{column}{0.5\textwidth}
\begin{block}{Quality of decisions}
Portfolio optimization, dynamic programming, RL.
\end{block}
\end{column}
\end{columns}

\medskip
\begin{alertblock}{Reflexivity --- finance is not medical imaging}
The decision procedure can \emph{reshape} the market and \emph{invalidate} the
forecast model. AI systems in finance are \textbf{actors within the market}, not
mere observers --- a strategic dimension with no analogue in most prediction problems.
\end{alertblock}
\end{frame}

\begin{frame}{Example: Reflexivity in Algorithmic Trading}

\begin{exampleblock}{Deep RL market maker}
A market-making agent quotes bid/ask at high frequency, earning the spread while
managing inventory risk. A \textbf{deep reinforcement learning} agent (NN policy
maximising cumulative reward) trains across millions of simulated episodes,
learning to adapt to changing liquidity.
\end{exampleblock}

\medskip
\textbf{The feedback loop:}
\[
  \text{agent's actions} \;\longrightarrow\; \text{order book}
  \;\longrightarrow\; \text{signals the agent conditions on} \;\longrightarrow\; \dots
\]

\medskip
\begin{block}{Design lesson}
The loop must be modeled \emph{explicitly} in the training environment ---
otherwise the policy performs well in simulation but \alert{destabilizes in live markets}.
\end{block}
\end{frame}

\begin{frame}{Forecasting, Risk, and Alpha Generation}

\textbf{Core quant problems --- all transformed by ML.} The common enemy is the
\alert{signal-to-noise ratio}: returns near-unpredictable short-horizon; risk
models extrapolate into rare tails; alpha erodes once discovered.

\medskip
\begin{tabular}{lll}
\toprule
\textbf{Problem} & \textbf{ML contribution} & \textbf{Care required} \\
\midrule
Return forecasting & High-dim feature spaces; no & Purge look-ahead bias; \\
                   & pre-specified interactions     & control multiple comparisons \\[3pt]
Risk estimation    & Richer distributions; regime  & Tails rarely observed; \\
                   & switching; cross-asset learning & stress periods break Gaussian VaR \\[3pt]
Alpha / signals    & Text-based signals add info    & Genuine alpha erodes fast \\
                   & absent from price \& volume    & once public \\
\bottomrule
\end{tabular}

\medskip
\begin{block}{Where text comes in}
Variance-covariance VaR assumes Gaussian returns and stable correlations ---
assumptions that fail \emph{precisely under stress}. Language models parse
narrative risk disclosures, turning qualitative uncertainty into quantitative signals.
\end{block}
\end{frame}

\begin{frame}{Regulatory Compliance and Fraud Detection}

\textbf{Beyond investment management} --- AI in operational \& compliance functions:
\begin{itemize}
  \item AML, sanctions screening, KYC, real-time fraud detection.
  \item All: spot \emph{rare, anomalous} patterns in high-volume streams ---
        a fit for classifiers trained on labeled historical incidents.
\end{itemize}

\medskip
\begin{block}{Why language models add particular value here}
Much relevant information is \emph{textual}: suspicious activity reports,
correspondence, filings, analyst opinions. An LLM can surface unusual phrasing,
flag narrative-vs-quantitative discrepancies, or help interpret new guidance.
\end{block}

\medskip
\begin{alertblock}{Regulation constrains design}
EU AI Act and GDPR's ``right to explanation'' can make \textbf{opaque models
inadmissible} for certain decisions \emph{regardless of accuracy} --- the
recurring performance-vs-interpretability tension.
\end{alertblock}
\end{frame}

\section{The Textual Nature of Financial Data}
% =====================================================================

\begin{frame}{Structured vs.\ Unstructured Data in Finance}

\begin{columns}[T]
\begin{column}{0.5\textwidth}
\begin{block}{Structured data}
Rows \& columns; well-defined types; each row a known entity at a time.\\[2pt]
Prices, accounting items, volumes. Served well by regression, factor models,
time series \emph{since the 1960s}.
\end{block}
\end{column}
\begin{column}{0.5\textwidth}
\begin{block}{Unstructured data}
No pre-defined schema --- text, audio, images, video. Needs a \emph{pipeline}
to turn raw content into statistical features.
\end{block}
\end{column}
\end{columns}

\medskip
\textbf{Limits of structured data:} backward-looking; capture \emph{outcomes}
(prices, earnings) not the reasoning/sentiment behind them; observed by everyone
simultaneously $\Rightarrow$ limited informational edge.

\medskip
\begin{bigpicture}
A large institution may see \textbf{tens of thousands of documents per day} ---
too voluminous to read systematically. This \alert{processing bottleneck} is a
persistent opportunity for ML methods that scale.
\end{bigpicture}
\end{frame}

\begin{frame}{Sources of Financial Text}

\begin{tabular}{p{0.22\textwidth} p{0.70\textwidth}}
\toprule
\textbf{Source} & \textbf{Content \& evidence} \\
\midrule
Corporate filings & SEC EDGAR: 10-K, 10-Q, 8-K. Narrative MD\&A, Risk Factors,
forward-looking statements. \textbf{Loughran--McDonald (2011):} 10-K sentiment
(finance lexicon) predicts post-filing returns. \\[4pt]
News \& wires & High-frequency public information. \textbf{Tetlock (2007):}
pessimistic WSJ ``Abreast of the Market'' language predicts downward pressure
then mean-reversion. \\[4pt]
Earnings calls & Live: scripted remarks \emph{plus} unscripted Q\&A. Tone,
question types, hedging, even speech pace carry incremental information beyond
reported numbers. \\
\bottomrule
\end{tabular}

\medskip
\begin{exampleblock}{Signal hidden in text}
EPS \$1.20 \emph{beats} the \$1.15 consensus --- looks positive. But Risk Factors
grew \textbf{30\% YoY}, three new risks concern a politically sensitive supply
chain, and the CEO said ``challenging'' \textbf{eleven times} --- a record.
The headline reader misses the signal.
\end{exampleblock}
\end{frame}

\begin{frame}{Why Text is Hard: Noise, Ambiguity, Context}

\begin{enumerate}
  \item \textbf{Ambiguous.} Meaning depends on author, time, context.
        ``Uncertainty'' in a Fed statement $\neq$ in a founder's letter.
        ``Positive outlook'' is bullish in equity research but \emph{cautious} in
        credit ratings. \textbf{$\sim$3/4} of Harvard-Inquirer ``negative'' words
        are neutral/positive in filings (Loughran--McDonald) --- hence a domain lexicon.
  \item \textbf{High-dimensional \& sparse.} A 50k-word vocabulary $\Rightarrow$ a
        document-term matrix with far more dimensions than observations. Needs
        dimensionality reduction (topic models, embeddings, LM encoders).
  \item \textbf{Strategically produced.} Disclosures written by lawyers/comms
        aware of liability and market reaction. Taking text at face value gets you
        \alert{systematically misled}.
\end{enumerate}

\medskip
\begin{bigpicture}
\textbf{Context is a first-class variable.} Bullish vs.\ bearish, compliant vs.\
misleading, consistent vs.\ contradictory --- all need a rich representation of
entity, time, regime, and genre. This long-range contextual reasoning is exactly
what \alert{transformers} are designed to do.
\end{bigpicture}
\end{frame}

\section{From Bag-of-Words to Large Language Models}
% =====================================================================

\begin{frame}{Early NLP in Finance: BoW, Dictionaries, Topic Models}

\begin{block}{Bag-of-Words (BoW) + tf-idf}
Document = vector of word counts; \emph{all} order and grammar discarded. tf-idf
up-weights rare informative words. Applied to filings since the 1990s for tone
and readability (Kearney \& Liu, 2014).
\end{block}

\begin{block}{Dictionary sentiment}
Map words to sentiment scores; net sentiment = (positive $-$ negative) frequency.
\textbf{Loughran--McDonald lexicon:} $\sim$2{,}700 words in 6 categories
(negative, positive, uncertainty, litigious, strong/weak modal) --- still a benchmark.
\end{block}

\end{frame}

\begin{frame}{Early NLP in Finance: Topic Models and Shared Limits}

\begin{block}{Topic models (LSA / LDA)}
Learn representations from \emph{co-occurrence} --- each document a mixture over
latent themes (e.g.\ operational efficiency vs.\ regulatory risk).
\end{block}

\medskip
\begin{alertblock}{Shared limitation}
\textbf{Shallow semantics:} words are atomic, meanings context-\emph{independent},
word order carries no information.
\end{alertblock}
\end{frame}

\begin{frame}{The Transformer Revolution (Vaswani et al., 2017)}

\begin{bigpicture}
``Attention Is All You Need'' (2017) triggered the largest transition in NLP
history. \alert{Self-attention} lets each position attend to \emph{every} other
position --- resolving long-range dependencies (a pronoun to its antecedent; a
Risk Factor to a financial-statement figure) that recurrent nets, with vanishing
gradients over long contexts, cannot do reliably.
\end{bigpicture}
\end{frame}

\begin{frame}{The Transformer Revolution: Scaled Dot-Product Attention}

\begin{underhood}
For each position $i$, a weighted sum of value vectors:
\[
  \mathrm{Attention}(\mathbf{Q},\mathbf{K},\mathbf{V})
  = \mathrm{softmax}\!\left(\frac{\mathbf{Q}\mathbf{K}^\top}{\sqrt{d_k}}\right)\mathbf{V},
\]
with $\mathbf{Q},\mathbf{K}$ query/key projections (dim $d_k$), $\mathbf{V}$ the
value projection. The $\sqrt{d_k}$ scaling keeps dot products from pushing the
softmax into a saturated, vanishing-gradient regime. Multiple \textbf{heads}
attend to different aspects (syntax, coreference) in parallel.
\end{underhood}
\end{frame}

\begin{frame}{Pre-training and Transfer Learning}

\textbf{Pre-trained transformers} --- BERT, GPT, RoBERTa, successors --- combine
self-attention with \emph{unsupervised pre-training} on very large corpora.

\medskip
\begin{block}{The transfer-learning economics}
Rich, context-sensitive representations $\Rightarrow$ fine-tune on downstream
tasks with \emph{relatively small} labeled datasets. Tasks that once needed
\textbf{thousands} of labeled examples now need \textbf{hundreds} --- lowering the
barrier for financial applications where labels are expensive.
\end{block}

\medskip
\begin{itemize}
  \item Lower layers learn transferable structure; higher layers specialise.
  \item Same backbone serves sentiment, NER, similarity, summarization, Q\&A.
  \item Full attention derivation \& financial NLP implications: LLM-foundations chapter.
\end{itemize}
\end{frame}

\begin{frame}{Why LLMs are Particularly Suited to Finance}

\begin{enumerate}
  \item \textbf{Long-document reasoning.} Context windows of tens--hundreds of
        thousands of tokens process a whole 10-K or call transcript in one pass ---
        no information loss from chunking.
  \item \textbf{In-context learning.} Supply 3 examples of suspicious patterns,
        evaluate a new transaction --- \emph{few-shot}, valuable where labels are scarce.
  \item \textbf{World knowledge.} Geopolitics, supply chains, governance norms let
        the model contextualise a disclosure. BloombergGPT shows finance
        pre-training helps, but general frontier models already do well.
  \item \textbf{Natural-language interface.} ``Summarize the key risks in this
        10-K and flag changes from last year'' --- no code, democratising text analysis.
\end{enumerate}

\medskip
\begin{alertblock}{LLMs are not a panacea}
They \textbf{hallucinate} (confident but wrong); are \textbf{prompt-sensitive};
their reasoning is \textbf{opaque}; and they may be \textbf{poorly calibrated} on
rare, high-stakes events. Deploy carefully --- evaluation, monitoring, human oversight.
\end{alertblock}
\end{frame}

\begin{frame}[noframenumbering]{The AI Effect and the Shifting Definition}

\begin{block}{The AI effect (ratchet)}
Tasks once seen as hallmarks of intelligence --- chess, diagnosis, translation ---
get reclassified as ``mere computation'' once machines do them reliably. So the
\emph{boundary} of AI is socially constructed as much as technically defined, and
the field perpetually chases a moving target.
\end{block}

\medskip
\textbf{Our pragmatic convention for the book:} AI = building systems that perform
cognitive tasks; ML = the dominant methodology; DL and LLMs = particular
implementations. Avoids endless definitional debate while staying precise enough to work.
\end{frame}

\begin{frame}[noframenumbering]{Formal Definitions (Reference)}

\begin{block}{Supervised learning}
Input space $\mathcal{X}$, output space $\mathcal{Y}$, parameters $\theta\in\Theta$,
per-sample loss $\ell:\mathcal{Y}\times\mathcal{Y}\to\mathbb{R}_{\geq0}$. Minimise
$\mathcal{L}(\theta)=\frac1n\sum_i \ell(f_\theta(\mathbf{x}_i),y_i)$; goal is
generalization to unseen inputs.
\end{block}

\begin{block}{Large language model}
Network with billions of parameters, trained on hundreds of billions of tokens,
assigning probabilities to token sequences; objective
$\mathcal{L}_{\mathrm{LM}}=-\sum_t \log P_\theta(x_t\mid x_{<t})$.
\end{block}

\begin{block}{Structured vs.\ unstructured}
Structured = relational rows/columns, typed fields, known entity-at-time.
Unstructured = no schema (text, audio, image, video); needs a feature pipeline.
\end{block}
\end{frame}

\begin{frame}[noframenumbering]{Multi-Head Attention and the $\sqrt{d_k}$ Scaling}

\begin{underhood}
\textbf{Why scale by $\sqrt{d_k}$?} If query/key components are $\sim(0,1)$ i.i.d.,
$\mathrm{Var}(\mathbf{q}^\top\mathbf{k})=d_k$. Large logits push softmax into a
saturated regime where gradients vanish. Dividing by $\sqrt{d_k}$ normalises
variance to $1$.
\end{underhood}

\medskip
\textbf{Multi-head:} run $h$ heads in parallel subspaces, then concatenate and project.
One head may track \emph{syntactic agreement} while another tracks
\emph{coreference}. In the standard single-head presentation $d_v=d_k$; multi-head
implementations may differ.

\medskip
\begin{block}{Standard configuration (base Transformer)}
$h=8$ heads, $d_k=d_v=64$, $d_{\text{model}}=512$, $L=6$ layers.
Modern LLMs: $L=32$--$96$, $d_{\text{model}}=4{,}096$--$8{,}192$.
\end{block}
\end{frame}

\begin{frame}[noframenumbering]{Regulatory Context (Reference)}

\textbf{Frameworks that constrain financial AI design:}
\begin{itemize}
  \item \textbf{EU AI Act} --- high-risk categories (credit scoring, insurance
        pricing); conformity assessment, human oversight, transparency.
  \item \textbf{GDPR right to explanation} --- opaque models may be inadmissible
        for certain decisions regardless of accuracy.
\end{itemize}

\medskip
\begin{block}{Application areas (treated in depth later)}
AML, sanctions screening, KYC, transaction fraud --- rare-event detection over
high-volume streams, much of it textual (suspicious activity reports,
correspondence, filings). See the RegTech \& AML chapter.
\end{block}
\end{frame}
