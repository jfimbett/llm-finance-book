\section{A Brief History of Textual Analysis in Finance}

\begin{frame}{The Founding Moment: Tetlock (2007)}
  \begin{block}{The Setup}
    Autumn 2006. Paul Tetlock assembles a dataset:
    \begin{itemize}
      \item Daily fraction of negative words in the \textit{WSJ} ``Abreast of the Market'' column (1984--1999)
      \item Next-day Dow Jones Industrial Average return
    \end{itemize}
  \end{block}
  \medskip
  \textbf{Results:}
  \begin{itemize}
    \item High negativity $\Rightarrow$ lower next-day returns \quad [puzzling under EMH]
    \item Effect \emph{reverses} over the subsequent week $\Rightarrow$ temporary mispricing
    \item High/low negativity $\Rightarrow$ high next-day trading volume
  \end{itemize}
  \medskip
  \begin{alertblock}{The core insight}
    Human language can be turned into a quantitative return-predicting signal.
    The reversal tells us it reflects sentiment, not fundamental news.
  \end{alertblock}
\end{frame}

\begin{frame}{Dictionary Methods: Harvard IV-4 and LM}
  \begin{columns}
    \column{0.5\textwidth}
    \textbf{General Inquirer (Stone et al., 1966)}
    \begin{itemize}
      \item 11,000 word stems classified into Positive, Negative, Strong, Weak, \ldots
      \item Score a document: count words per category, normalise by length
      \item Transparent and reproducible
    \end{itemize}
    \column{0.5\textwidth}
    \textbf{Loughran--McDonald (LM, 2011)}
    \begin{itemize}
      \item \textit{tax}, \textit{liability}, \textit{depreciation} flagged ``negative'' by Harvard IV-4 but are merely financial jargon
      \item LM dictionary hand-coded from 10-K filings: Negative, Positive, Uncertainty, Litigious, Strong Modal, Weak Modal
      \item Substantially outperforms Harvard IV-4 in predicting filing-date abnormal returns
    \end{itemize}
  \end{columns}
  \medskip
  \begin{alertblock}{Fundamental limitation of all dictionary methods}
    Static, order-invariant.
    ``Earnings did \emph{not} decline'' $=$ ``Earnings \emph{did} decline'' unless negation is hard-coded.
  \end{alertblock}
\end{frame}

\begin{frame}{From Dictionaries to Neural Methods: A Timeline}
  \begin{center}
  \begin{tabular}{@{}lll@{}}
    \toprule
    Year & Method & Key advance \\
    \midrule
    1966       & General Inquirer       & Systematic word counting \\
    2003       & LDA (topic models)     & Data-driven topics, no labels needed \\
    2011       & LM Dictionary          & Financial-domain sentiment \\
    2013       & Word2Vec / GloVe       & Dense embeddings, semantic geometry \\
    1997/2015  & LSTM / Attention       & Sequential context, negation \\
    2017       & Transformer            & Parallel self-attention, massive scale \\
    2018--19   & BERT / FinBERT         & Pre-train then fine-tune \\
    2020+      & GPT-4, Claude, Llama   & Instruction-following, reasoning \\
    \bottomrule
  \end{tabular}
  \end{center}
  \medskip
  Each generation addresses the core limitation of the previous one.\\
  Understanding this history explains \emph{why} LLMs behave the way they do.
\end{frame}

\section{Why Text Matters in Finance}

\begin{frame}{Sources of Financial Text}
  \begin{description}
    \item[\textbf{Earnings calls}] Scripted presentation + unscripted Q\&A;
          Q\&A is spontaneous and more information-rich
    \item[\textbf{10-K / 10-Q filings}] MD\&A, risk factors; up to 200 pages;
          primary source for Loughran--McDonald (2011)
    \item[\textbf{Newswire / news}] High frequency; narrow exploitation window; value in \emph{speed} and \emph{breadth}
    \item[\textbf{Analyst reports}] Informationally dense; linguistic complexity $\propto$ information asymmetry
    \item[\textbf{Central bank comms}] FOMC minutes and statements move rates and equity prices on release
    \item[\textbf{Social media}] High-frequency, noisy; predictive power documented but robustness debated
  \end{description}
\end{frame}

\begin{frame}{Why Text Signals Survive Market Efficiency}
  \textbf{Semi-strong EMH (Fama, 1970):} prices reflect all publicly available information.
  Yet large empirical literature finds text predicts returns. Three reconciling mechanisms:
  \medskip
  \begin{enumerate}
    \item \textbf{Processing costs \& limited attention}\\
          {\small A 200-page 10-K is costly to read. Not all investors process it fully.
          NLP at scale makes it feasible $\Rightarrow$ anomaly should attenuate with technology.}
    \medskip
    \item \textbf{Soft information \& interpretive disagreement}\\
          {\small Cautious tone = genuine uncertainty \emph{or} expectation management?
          Disagreement $\Rightarrow$ prices reflect average of beliefs, not the most informative one.}
    \medskip
    \item \textbf{Latent information}\\
          {\small Regulation FD limits explicit selective disclosure.
          Information leaks through word choice, omissions, and rhetorical structure.}
  \end{enumerate}
  \medskip
  \small Text analysis adds value not because markets are irrational, but because
  \emph{information extraction from text is genuinely costly}.
\end{frame}

\begin{frame}{Four Empirical Regularities}
  The theory must explain these stylised facts:
  \medskip
  \begin{enumerate}
    \item \textbf{Negative tone $\Rightarrow$ negative short-run returns, with reversal.}\\
          {\small Tetlock (2007): pessimistic WSJ prose $\to$ lower next-day DJIA, then mean-reversion.}
    \medskip
    \item \textbf{Financial-domain dictionaries outperform general-domain ones.}\\
          {\small Loughran--McDonald (2011): LM Negative $\gg$ Harvard IV-4 for predicting filing-date abnormal returns.}
    \medskip
    \item \textbf{Earnings call tone predicts future earnings surprises.}\\
          {\small Larcker--Zakolyukina (2012): deceptive CEO/CFO speech has distinct linguistic markers.}
    \medskip
    \item \textbf{LLM signals show incremental predictive power.}\\
          {\small Lopez-Lira \& Tang (2023): ChatGPT sentiment for headlines predicts next-day returns;
          absent for GPT-1 and BERT $\Rightarrow$ emergent property of scale.}
  \end{enumerate}
\end{frame}

\begin{frame}{The Textual Signal: Formal Definition}
  \begin{block}{Definition}
    A \textbf{textual signal} is a measurable function
    \[
      f \colon \mathcal{D} \;\longrightarrow\; \mathbb{R}^k
    \]
    mapping document $d_t$ (at time $t$) to signal value $s_t = f(d_t)$.
  \end{block}
  \medskip
  Examples of $f$:
  \begin{itemize}
    \item Fraction of LM Negative words ($k=1$)
    \item Vector of LDA topic proportions ($k = $ number of topics)
    \item Word2Vec sentence embedding ($k = 300$)
    \item LLM classifier output ($k = $ number of classes)
  \end{itemize}
  \medskip
  \textbf{This course} = investigating the best choice of $f$:
  from simple word counts $\to$ topic models $\to$ transformer embeddings $\to$ LLMs as reasoning agents.
\end{frame}

\begin{frame}{The Linear Factor Model and Economic Significance}
  The central empirical question:
  \[
    \mathrm{Cov}(s_t,\, r_{t+1}) \neq 0 \;?
  \]
  Under a linear factor model:
  \[
    r_{t+1} = \alpha + \beta^\top s_t + \varepsilon_{t+1}, \qquad
    \mathbb{E}[\varepsilon_{t+1} \mid s_t] = 0
  \]
  \medskip
  $\beta$ is \textbf{economically significant} if and only if:
  \begin{enumerate}
    \item $|\beta| > 0$ after controlling for size, value, momentum, \ldots
    \item The implied Sharpe ratio exceeds plausible transaction costs
  \end{enumerate}
  \medskip
  \begin{alertblock}{Both conditions are necessary}
    Statistical significance without economic significance does not justify a trading strategy.
  \end{alertblock}
\end{frame}

\begin{frame}{Identification Challenges}
  Showing $\mathrm{Cov}(s_t, r_{t+1}) \neq 0$ is necessary but \emph{not sufficient}:
  \medskip
  \begin{description}
    \item[\textbf{Reverse causality}]
      Document $d_t$ may be written \emph{in response to} lagged returns,
      not in anticipation of future ones.
    \item[\textbf{Confounding}]
      Macro conditions drive both tone and returns.
      Use high-frequency event studies with known release times.
    \item[\textbf{Multiple testing}]
      NLP produces thousands of candidate signals.
      Use strict temporal out-of-sample split (train pre-2010, test post-2010).
    \item[\textbf{Model misspecification}]
      The choice of $f$ is itself an assumption.
      Dictionary $f$ assumes only word counts matter;
      LLM $f$ relaxes this but introduces training-data biases.
  \end{description}
\end{frame}

\section{Classical Text Representations}

\begin{frame}{Tokenisation and the Preprocessing Pipeline}
  Before computing any representation, raw text is standardised:
  \begin{enumerate}
    \item \textbf{Case folding} --- ``Revenue'' $\equiv$ ``revenue''
    \item \textbf{Stop-word removal} --- remove ``the'', ``and'', ``of'', \ldots\ (100--300 words)
    \item \textbf{Stemming / lemmatisation}
          \begin{itemize}
            \item Stemming: \textit{earnings, earned, earns} $\to$ \textit{earn}
            \item Lemmatisation: \textit{better} $\to$ \textit{good} (more accurate, slower)
          \end{itemize}
    \item \textbf{Vocabulary truncation} --- retain top $V'$ types; discard very rare words
  \end{enumerate}
  \medskip
  Result: finite vocabulary $\mathcal{V}$ of size $V$ (typically $10\,000$--$100\,000$)\\[4pt]
  \textbf{Note:} LLMs use \emph{sub-word tokenisation} --- ``earnings'' $\to$ [``earn'', ``ings''] ---
  allowing rare words to be represented as combinations of frequent sub-units.
\end{frame}

\begin{frame}{Bag-of-Words Model}
  \begin{block}{Definition}
    \[
      \mathbf{x}(d) = \bigl(\mathrm{count}(v_1, d),\, \ldots,\, \mathrm{count}(v_V, d)\bigr)^\top
      \in \mathbb{Z}_{\geq 0}^V
    \]
    Word order is discarded. The document is treated as an unordered multiset.
  \end{block}
  \medskip
  \begin{columns}
    \column{0.5\textwidth}
    \textbf{Virtues}
    \begin{itemize}
      \item Simple and interpretable
      \item Sparse --- efficient storage (CSR/CSC)
      \item Competitive baseline for many tasks
    \end{itemize}
    \column{0.5\textwidth}
    \textbf{The fundamental deficiency}
    \begin{itemize}
      \item ``The fund outperformed the market'' $=$
            ``The market outperformed the fund'' \quad (identical BoW vectors, opposite meanings)
    \end{itemize}
  \end{columns}
  \medskip
  The term-document matrix $\mathbf{X} \in \mathbb{Z}_{\geq 0}^{N \times V}$
  stacks BoW vectors for all $N$ documents.
\end{frame}

\begin{frame}{One-Hot Encoding: The Discrete Metric Problem}
  Each token $v_k$ maps to the standard basis vector:
  \[
    \mathbf{e}_k = (\underbrace{0,\ldots,0}_{k-1},\, 1,\, \underbrace{0,\ldots,0}_{V-k})^\top
    \in \{0,1\}^V
  \]
  BoW = sum of one-hot vectors of constituent tokens.
  \medskip
  \begin{block}{The problem: orthogonality}
    \[
      \mathbf{e}_j^\top \mathbf{e}_k = \mathbb{1}[j=k]
      \quad \Rightarrow \quad
      \mathrm{sim}(\text{revenue},\, \text{income}) = 0
      = \mathrm{sim}(\text{revenue},\, \text{penguin})
    \]
  \end{block}
  \medskip
  Every pair of distinct words is equally dissimilar --- the \emph{discrete metric}.\\
  This is semantically incorrect: financial analysts know that ``revenue'' and ``income''
  are nearly synonymous in most contexts.\\
  \medskip
  \textbf{Word embeddings} replace the discrete metric with a continuous geometry
  that reflects distributional similarity.
\end{frame}

\begin{frame}{TF-IDF: Motivation and Definitions}
  Raw BoW counts dominated by stop words. TF-IDF weights by document-level rarity:
  \medskip
  \[
    \underbrace{\mathrm{tf}(v, d)}_{\text{term freq.}}
    = \frac{\mathrm{count}(v,d)}{|d|}
    \qquad
    \underbrace{\mathrm{idf}(v, \mathcal{C})}_{\text{inv.\ doc.\ freq.}}
    = \log\!\left(\frac{|\mathcal{C}|}{\mathrm{df}(v, \mathcal{C})}\right)
  \]
  \[
    \mathrm{tfidf}(v, d, \mathcal{C}) = \mathrm{tf}(v, d) \;\times\; \mathrm{idf}(v, \mathcal{C})
  \]
  \medskip
  \textbf{Intuition of IDF:}
  \begin{itemize}
    \item Word in \emph{every} document $\Rightarrow$ $\mathrm{df} = |\mathcal{C}|$ $\Rightarrow$
          $\mathrm{idf} = \log(1) = 0$ \quad (suppressed)
    \item Word in \emph{one} document $\Rightarrow$ $\mathrm{idf} = \log|\mathcal{C}|$ \quad (maximally discriminative)
  \end{itemize}
  \medskip
  Document similarity: cosine on $\ell_2$-normalised TF-IDF vectors
  \[
    \mathrm{sim}(d_i, d_j)
    = \frac{\tilde{\mathbf{x}}(d_i)^\top \tilde{\mathbf{x}}(d_j)}
           {\|\tilde{\mathbf{x}}(d_i)\|_2\,\|\tilde{\mathbf{x}}(d_j)\|_2} \;\in\; [0,1]
  \]
\end{frame}

\begin{frame}{TF-IDF: Worked Example}
  Three pre-processed financial sentences ($N=3$, $V=8$):
  \medskip
  \begin{center}
  \begin{tabular}{@{}cl@{}}
    \toprule
    $d$ & Tokens \\
    \midrule
    $d_1$ & \texttt{revenue grew strong earnings} \\
    $d_2$ & \texttt{earnings declined weak guidance} \\
    $d_3$ & \texttt{revenue guidance raised strong} \\
    \bottomrule
  \end{tabular}
  \end{center}
  \medskip
  $\mathrm{idf}(\texttt{declined}) = \mathrm{idf}(\texttt{grew}) = \mathrm{idf}(\texttt{raised})
  = \mathrm{idf}(\texttt{weak}) = \ln(3) \approx 1.099$\\
  $\mathrm{idf}(\texttt{earnings}) = \mathrm{idf}(\texttt{guidance}) = \mathrm{idf}(\texttt{revenue})
  = \mathrm{idf}(\texttt{strong}) = \ln(3/2) \approx 0.405$
  \medskip
  \begin{block}{Cosine similarities}
    $\mathrm{sim}(d_1, d_3) \approx 0.213$ (shared: \texttt{revenue, strong})\\
    $\mathrm{sim}(d_1, d_2) \approx 0.106$ (shared: \texttt{earnings} only)
  \end{block}
  Full computation in Practical Session 1.
\end{frame}

\begin{frame}{Limitations of Classical Methods}
  Despite TF-IDF's improvements over raw counts, all classical methods share:
  \medskip
  \begin{enumerate}
    \item \textbf{Order invariance}\\
          ``Earnings beat estimates'' = ``Estimates beat earnings'' (identical BoW/TF-IDF vectors)
    \medskip
    \item \textbf{No semantic similarity}\\
          Synonyms that never co-occur (\textit{revenue}, \textit{sales}) have zero cosine similarity
    \medskip
    \item \textbf{Fixed vocabulary}\\
          New words at test time (acronyms, company names) are out-of-vocabulary and discarded
    \medskip
    \item \textbf{High-dimensional sparsity}\\
          $V \sim 10^4$--$10^5$; even with IDF weighting, unsuitable as direct inputs to
          dense neural layers without dimensionality reduction
  \end{enumerate}
  \medskip
  \textbf{Solution:} replace the discrete one-hot basis with a continuous embedding space.
\end{frame}

\section{Word Embeddings}

\begin{frame}{The Distributional Hypothesis}
  \begin{block}{J.R.\ Firth (1957)}
    \emph{``You shall know a word by the company it keeps.''}
  \end{block}
  \medskip
  Words with similar meanings appear in similar contexts.\\
  Capture contextual co-occurrence $\Rightarrow$ capture semantic similarity.
  \medskip
  \begin{block}{Definition: Word Embedding}
    An injective map $\varphi : \mathcal{V} \to \mathbb{R}^d$ with $d \ll V$.\\
    Embedding matrix $\mathbf{W} \in \mathbb{R}^{V \times d}$;
    row $i$ = $\varphi(w_i)$.\\
    Typical: $d \in \{50, 100, 300\}$; compression ratio $V/d \approx 150$--$1{,}000$.
  \end{block}
  \medskip
  \textbf{Geometry:} under a well-trained embedding, cosine similarity $\approx$ semantic similarity.\\
  \textbf{Linear analogies:} $\varphi(\text{king}) - \varphi(\text{man}) + \varphi(\text{woman})
  \approx \varphi(\text{queen})$\\
  In finance: $\varphi(\text{call option}) - \varphi(\text{upside}) + \varphi(\text{downside})
  \approx \varphi(\text{put option})$
\end{frame}

\begin{frame}{Word2Vec: Skip-Gram Objective}
  Each centre word $w_t$ predicts context words in window $\pm c$:
  \[
    J_{\mathrm{SG}}
    = \frac{1}{T} \sum_{t=1}^{T}
      \sum_{\substack{k=-c \\ k \neq 0}}^{c}
      \log P(w_{t+k} \mid w_t)
  \]
  where the softmax probability is:
  \[
    P(w_{t+k} \mid w_t)
    = \frac{\exp\!\bigl(\mathbf{v}_{w_{t+k}}^{\top} \hat{\mathbf{v}}_{w_t}\bigr)}
           {\sum_{w=1}^{V} \exp\!\bigl(\mathbf{v}_{w}^{\top} \hat{\mathbf{v}}_{w_t}\bigr)}
  \]
  \begin{alertblock}{The computational problem}
    Normalising constant sums over all $V \approx 10^4$--$10^5$ words \emph{at every gradient step}.
    For 1 billion tokens: intractable.
  \end{alertblock}
  Two embedding matrices: \emph{centre-word} vectors $\hat{\mathbf{v}}_w$ and \emph{context-word} vectors $\mathbf{v}_w$.
\end{frame}

\begin{frame}{Negative Sampling: Efficient Training}
  Replace the full softmax with a binary classification task:
  \begin{itemize}
    \item 1 \emph{positive} sample: the observed context word $w_{t+k}$
    \item $K$ \emph{negative} samples: words drawn from noise distribution $P_n$
  \end{itemize}
  \[
    \mathcal{L}_{\mathrm{NEG}}(w_t, w_{t+k})
    = \log \sigma\!\bigl(\mathbf{v}_{w_{t+k}}^{\top} \hat{\mathbf{v}}_{w_t}\bigr)
    + \sum_{j=1}^{K}
      \mathbb{E}_{w_j \sim P_n}\!
      \Bigl[\log \sigma\!\bigl(-\mathbf{v}_{w_j}^{\top} \hat{\mathbf{v}}_{w_t}\bigr)\Bigr]
  \]
  \textbf{Gradient interpretation:}
  \[
    \frac{\partial \mathcal{L}_{\mathrm{NEG}}}{\partial \mathbf{v}_{w_{t+k}}}
    = \bigl[1 - \sigma(\mathbf{v}_{w_{t+k}}^{\top} \hat{\mathbf{v}}_{w_t})\bigr]\hat{\mathbf{v}}_{w_t}
  \]
  Positive context vector is pushed \emph{toward} centre vector (large coefficient when dissimilar);
  negative sample vectors are pushed \emph{away}.\\[4pt]
  \textbf{Efficiency:} only $K{+}1$ rows of $\mathbf{W}$ touched per step, not all $V$.\\
  Recommended: $K = 5$--$20$; $P_n(w) \propto \mathrm{freq}(w)^{3/4}$.
\end{frame}

\begin{frame}{GloVe: Global Co-occurrence Factorisation}
  \textbf{Co-occurrence matrix} $X_{ij}$ = times word $j$ appears in context of word $i$ across the full corpus.
  \medskip
  \textbf{Key insight:} the \emph{ratio} $P_{ik}/P_{jk}$ encodes relational meaning more faithfully than raw probabilities.
  \medskip
  GloVe minimises:
  \[
    J_{\mathrm{GloVe}}
    = \sum_{i,j=1}^{V}
      f(X_{ij})
      \Bigl(\mathbf{w}_i^{\top} \tilde{\mathbf{w}}_j + b_i + \tilde{b}_j - \log X_{ij}\Bigr)^{2}
  \]
  with weighting function $f(x) = (x/x_{\max})^{\alpha}$ for $x < x_{\max}$, else 1.
  Recommended: $\alpha = 0.75$, $x_{\max} = 100$.
  \medskip
  \begin{block}{GloVe vs.\ Word2Vec}
    \begin{tabular}{@{}ll@{}}
      GloVe & single-pass WLS over precomputed counts; faster with stored matrix \\
      Word2Vec & online SGD; scales to streaming corpora \\
    \end{tabular}\\
    Similar embedding quality in practice.
  \end{block}
\end{frame}

\begin{frame}{Finance-Specific Domain Adaptation}
  General embeddings (Wikipedia/Common Crawl) encode general-English meaning:
  \medskip
  \begin{center}
  \begin{tabular}{@{}lll@{}}
    \toprule
    Word & General nearest neighbours & Finance nearest neighbours \\
    \midrule
    \textit{risk}   & danger, threat, hazard   & volatility, exposure, VaR \\
    \textit{short}  & brief, concise           & short-sell, bear position \\
    \textit{margin} & edge, border             & profit margin, maintenance margin \\
    \bottomrule
  \end{tabular}
  \end{center}
  \medskip
  \textbf{Domain adaptation strategies:}
  \begin{enumerate}
    \item \textbf{Pre-train from scratch} on financial corpus
          (Bloomberg Financial Word2Vec; FinancialPhraseBank embeddings)
    \item \textbf{Fine-tune transformer embeddings}
          (FinBERT: BERT re-trained on Reuters news, 10-K filings, earnings calls)
    \item \textbf{Retrofitting}: post-hoc adjustment using a financial lexicon or ontology
          to pull synonyms together and push antonyms apart
  \end{enumerate}
\end{frame}

  Project 30 financial terms into 2D via PCA on GloVe vectors.
  Colour by semantic group:
  \textcolor{blue}{rates \& macro} $\cdot$
  \textcolor{green!60!black}{equities \& returns} $\cdot$
  \textcolor{red}{risk} $\cdot$
  \textcolor{orange}{corporate actions \& derivatives}
  \medskip
  \begin{block}{Consistent patterns across pre-trained models}
    \begin{itemize}
      \item \textit{option, futures, derivative} $\to$ tight cluster
      \item \textit{default, credit} $\to$ cluster together, away from equities
      \item \textit{merger, acquisition} $\to$ nearly identical vectors

      \item Rates cluster (\textit{yield, spread, inflation, interest rate}) $\to$ most cohesive
    \end{itemize}
  \end{block}
  \begin{alertblock}{Caution on axis interpretation}
    PCA axes $\mathbf{p}_1, \mathbf{p}_2$ are \emph{not} interpretable as named financial concepts without further validation.
  \end{alertblock}
