% --- Frame 1: Document Representations ---
\begin{frame}
\frametitle{From Words to Documents: Three Aggregation Strategies}

\textbf{Why documents?} Earnings calls $\approx$10\,k tokens. 10-K filings $\approx$100\,k tokens.

\medskip
\begin{tabular}{lll}
\toprule
\textbf{Method} & \textbf{Formula} & \textbf{Weakness} \\
\midrule
Mean embedding &
  $\bar{\varphi}(d) = \frac{1}{n}\sum_{i=1}^n \varphi(w_i)$ &
  No word order \\[4pt]
TF-IDF weighted &
  $\varphi_{\text{tfidf}} = \frac{\sum \text{tfidf}(w_i)\,\varphi(w_i)}{\sum \text{tfidf}(w_i)}$ &
  Still bag-of-words \\[4pt]
SBERT (contextual) &
  Siamese BERT fine-tuned on NLI &
  Compute at encoding \\
\bottomrule
\end{tabular}

\medskip
\begin{block}{Key insight}
Mean embedding ignores word order.
``Revenue rose, costs fell'' $\equiv$ ``Costs rose, revenue fell'' under mean pooling.
\end{block}

\medskip
\textbf{SBERT advantage:} pre-compute once; search is $O(1)$ at query time.
\end{frame}

% --- Frame 2: RNN + Vanishing Gradient ---
\begin{frame}
\frametitle{RNNs and the Vanishing Gradient Problem}

\textbf{Vanilla RNN recurrence:}
\[
\mathbf{h}_t = \sigma\!\left(W_h\,\mathbf{h}_{t-1} + W_x\,\mathbf{x}_t + \mathbf{b}\right)
\]

\textbf{BPTT gradient:}
\[
\frac{\partial \mathcal{L}}{\partial \mathbf{h}_t}
= \sum_{s=t}^{T} \frac{\partial \ell_s}{\partial \mathbf{h}_s}
  \prod_{k=t}^{s-1} \frac{\partial \mathbf{h}_{k+1}}{\partial \mathbf{h}_k}
\]

\begin{alertblock}{Proposition (Vanishing \& Exploding Gradients)}
Let $\rho = \|W_h\|_2$, $M_\sigma = \sup_z |\sigma'(z)|$. Then
\[
\left\|\prod_{k=t}^{T-1}\frac{\partial \mathbf{h}_{k+1}}{\partial \mathbf{h}_k}\right\|_2
\;\leq\; \left(M_\sigma\,\rho\right)^{T-t}
\]
For logistic sigmoid: $M_\sigma = \tfrac{1}{4}$.
If $M_\sigma\rho < 1$: \textcolor{blue}{gradients vanish exponentially.}
\end{alertblock}

\textbf{Finance implication:} An earnings call of 10\,k tokens makes long-range
dependencies (page 3 $\leftrightarrow$ page 20) practically unlearnable.
\end{frame}

% --- Frame 3: LSTM Gates ---
\begin{frame}
\frametitle{LSTM: Six Equations, One Key Idea}

\textbf{The six equations:}
\begin{align*}
\mathbf{f}_t &= \sigma(W_f[\mathbf{h}_{t-1},\mathbf{x}_t] + \mathbf{b}_f)
  &&\text{forget gate}\\
\mathbf{i}_t &= \sigma(W_i[\mathbf{h}_{t-1},\mathbf{x}_t] + \mathbf{b}_i)
  &&\text{input gate}\\
\mathbf{g}_t &= \tanh(W_g[\mathbf{h}_{t-1},\mathbf{x}_t] + \mathbf{b}_g)
  &&\text{candidate cell}\\
\mathbf{c}_t &= \mathbf{f}_t \odot \mathbf{c}_{t-1} + \mathbf{i}_t \odot \mathbf{g}_t
  &&\text{cell state (additive!)}\\
\mathbf{o}_t &= \sigma(W_o[\mathbf{h}_{t-1},\mathbf{x}_t] + \mathbf{b}_o)
  &&\text{output gate}\\
\mathbf{h}_t &= \mathbf{o}_t \odot \tanh(\mathbf{c}_t)
  &&\text{hidden state}
\end{align*}

\begin{block}{Why the cell state avoids vanishing gradients}
\[
\frac{\partial \mathcal{L}}{\partial \mathbf{c}_t}
= \frac{\partial \mathcal{L}}{\partial \mathbf{c}_{t+1}} \odot \mathbf{f}_{t+1}
  + \text{(terms via } \mathbf{h}_t\text{)}
\]
When $\mathbf{f}_{t+1} \approx \mathbf{1}$: gradient flows back undamped.\\
(\emph{``Constant error carousel''} — Hochreiter \& Schmidhuber, 1997)
\end{block}

GRU = 4 equations, $\approx$25\% fewer parameters. Preferred under compute constraints.
\end{frame}

% --- Frame 4: Bahdanau Attention ---
\begin{frame}
\frametitle{Bahdanau Attention: The Bridge to Transformers}

\textbf{The bottleneck:} encoder must compress the full input into one fixed vector.

\medskip
\textbf{Solution (Bahdanau et al., 2015):} let the decoder attend to all encoder states.

\medskip
\textbf{Alignment score:}
\[
e_{t,i} = \mathbf{v}^\top \tanh\!\left(W_s\,\mathbf{s}_t + W_h\,\mathbf{h}_i\right)
\]

\textbf{Attention weights and context vector:}
\[
\alpha_{t,i} = \frac{\exp(e_{t,i})}{\sum_j \exp(e_{t,j})},
\qquad
\mathbf{c}_t = \sum_{i} \alpha_{t,i}\,\mathbf{h}_i
\]

$\{\alpha_{t,i}\}$ form a probability distribution over encoder positions.
\medskip

\begin{block}{Finance intuition}
Decoder generating a sentence about \emph{net income} assigns high $\alpha_{t,i}$
to the tokens in the transcript where income figures appear. Attention is interpretable.
\end{block}

\medskip
\textbf{Next step:} apply attention \emph{within} a single sequence $\to$ self-attention $\to$ Transformer.
\end{frame}

% --- Frame 5: Scaled Dot-Product Attention ---
\begin{frame}
\frametitle{Scaled Dot-Product Attention}

Given input matrix $X \in \mathbb{R}^{n \times d_{\text{model}}}$, project to queries, keys, values:
\[
Q = XW^Q,\quad K = XW^K,\quad V = XW^V
\]

\textbf{Attention formula:}
\[
\text{Attention}(Q, K, V)
= \text{softmax}\!\left(\frac{QK^\top}{\sqrt{d_k}}\right) V
\]

\begin{itemize}
  \item $QK^\top \in \mathbb{R}^{n \times n}$: entry $(i,j) = \mathbf{q}_i^\top\mathbf{k}_j$
  \item Row-wise softmax $\to$ attention weights $A_{ij} > 0$, $\sum_j A_{ij} = 1$
  \item Output for token $i$: $\sum_j A_{ij}\,\mathbf{v}_j$ (convex combination of values)
\end{itemize}

\medskip
\textbf{Causal mask (decoder):} set $\tilde{S}_{ij} = -\infty$ for $j > i$ to prevent attending to future tokens.
\end{frame}

% --- Frame 6: Why √d_k ---
\begin{frame}
\frametitle{Why Divide by $\sqrt{d_k}$? The Variance Proof}

\textbf{Setup:} suppose $q_\ell, k_\ell \overset{\text{i.i.d.}}{\sim} (0,1)$ for $\ell = 1,\ldots,d_k$.

\medskip
\textbf{Step 1.} Compute the variance of the dot product:
\[
\text{Var}\!\left(\mathbf{q}_i^\top \mathbf{k}_j\right) = \sum_{\ell=1}^{d_k} \text{Var}(q_\ell k_\ell) = d_k
\]

\textbf{Step 2.} For large $d_k$, logits concentrate near $\pm\sqrt{d_k}$:
\[
\text{softmax gradient:}\quad \sigma(z)(1-\sigma(z)) \approx 0 \text{ for large } |z|
\]

\textbf{Step 3.} Divide by $\sqrt{d_k}$ to normalise variance to 1:
\[
\text{Var}\!\left(\frac{\mathbf{q}_i^\top \mathbf{k}_j}{\sqrt{d_k}}\right) = 1
\]

\begin{block}{Consequence}
Without scaling, the softmax saturates and gradients vanish — we reintroduce the
same problem LSTMs were designed to fix, now inside the attention layer.
\end{block}
\end{frame}

% --- Frame 7: Multi-Head Attention ---
\begin{frame}
\frametitle{Multi-Head Attention}

Run $h$ attention heads in parallel, each with its own projections:

\[
\text{head}_i = \text{Attention}(XW^Q_i,\,XW^K_i,\,XW^V_i)
\]

\[
\text{MultiHead}(X) = \text{Concat}(\text{head}_1,\ldots,\text{head}_h)\,W^O
\]

\medskip
Original Transformer: $h = 8$,\quad $d_k = d_v = 64$,\quad $d_{\text{model}} = 512$.

\medskip
\textbf{What do heads learn?}
\begin{itemize}
  \item Lower layers: syntactic dependency (subject--verb agreement, negation scope)
  \item Deeper layers: semantic associations (entity co-reference, temporal relations)
  \item Financial models: heads specialize on ``last quarter'' $\to$ figures, entity linking
\end{itemize}

\medskip
\textbf{Complexity:} $O(n^2 d_{\text{model}})$ per layer. For $n = 4{,}096$ tokens:
$\approx$134M values per layer. Motivates FlashAttention (Dao et al., 2022).
\end{frame}

% --- Frame 8: Positional Encoding + Rotation Property ---
\begin{frame}
\frametitle{Positional Encoding and the Rotation Property}

Self-attention is permutation-equivariant $\Rightarrow$ position must be injected.

\medskip
\textbf{Sinusoidal positional encoding:}
\[
\text{PE}_{(\text{pos},2i)} = \sin\!\left(\tfrac{\text{pos}}{10000^{2i/d}}\right),
\quad
\text{PE}_{(\text{pos},2i+1)} = \cos\!\left(\tfrac{\text{pos}}{10000^{2i/d}}\right)
\]

Small $i$: high frequency (fine position). Large $i$: low frequency (coarse position).

\medskip
\textbf{Rotation property:} for any fixed offset $k$,
\[
\begin{pmatrix} \text{PE}_{(\text{pos}+k,2i)} \\ \text{PE}_{(\text{pos}+k,2i+1)} \end{pmatrix}
=
\begin{pmatrix} \cos\omega_i k & -\sin\omega_i k \\ \sin\omega_i k & \cos\omega_i k \end{pmatrix}
\begin{pmatrix} \text{PE}_{(\text{pos},2i)} \\ \text{PE}_{(\text{pos},2i+1)} \end{pmatrix}
\]

The rotation matrix depends only on $k$, not on the absolute position $\Rightarrow$
\textbf{relative positions are linearly decodable}.

\medskip
Modern variant: \textbf{RoPE} (Rotary Position Embeddings) embeds the rotation
directly into query--key dot products. Used in Llama, Mistral, Qwen.
\end{frame}

% --- Frame 9: Full Encoder Block ---
\begin{frame}
\frametitle{The Full Encoder Block}

Two sub-layers with residual connections and Layer Normalisation:

\[
Z^{(\ell)} = \text{LayerNorm}\!\left(X^{(\ell)} + \text{MultiHead}(X^{(\ell)})\right)
\]

\[
X^{(\ell+1)} = \text{LayerNorm}\!\left(Z^{(\ell)} + \text{FFN}(Z^{(\ell)})\right)
\]

where the feed-forward network is:
\[
\text{FFN}(\mathbf{z}) = \max(0,\,\mathbf{z}W_1 + \mathbf{b}_1)\,W_2 + \mathbf{b}_2,
\quad d_{\text{ff}} = 4\,d_{\text{model}}
\]

\medskip
\begin{block}{Residual connections are critical}
Direct gradient pathways from output to input eliminate the multiplicative
vanishing-gradient problem across layer depth.
\end{block}

\medskip
\textbf{Scale:} base Transformer: $L = 6$ layers.
Modern LLMs: $L = 32$--$96$ layers, $d_{\text{model}} = 4{,}096$--$8{,}192$.
\end{frame}

% --- Frame 10: BERT MLM ---
\begin{frame}
\frametitle{BERT: Masked Language Modelling}

BERT = Bidirectional Encoder Representations from Transformers (Devlin et al., 2019).

\medskip
\textbf{MLM objective:} mask 15\% of tokens; predict originals.
\[
\mathcal{L}_{\text{MLM}}(\theta)
= -\sum_{i \in \mathcal{M}} \log p_\theta\!\left(w_i \mid \mathbf{x}_{\setminus \mathcal{M}}\right)
\]

\medskip
Masking protocol:
\begin{itemize}
  \item 80\%: replaced with \texttt{[MASK]}
  \item 10\%: replaced with a random token
  \item 10\%: left unchanged (forces the model to build a representation for every token)
\end{itemize}

\medskip
\textbf{Architecture:} encoder-only, fully bidirectional attention.

\medskip
\begin{block}{Best for}
Classification (sentiment, credit rating), token-level extraction (NER, metric
extraction), semantic similarity. \textbf{Not for generation.}
\end{block}
\end{frame}

% --- Frame 11: GPT CLM + Encoder-Decoder Choice Table ---
\begin{frame}
\frametitle{GPT: Causal Language Modelling and Architecture Choice}

\textbf{CLM objective:} predict the next token from all preceding tokens.
\[
\mathcal{L}_{\text{CLM}}(\theta) = -\sum_{t=1}^{T} \log p_\theta(w_t \mid w_1,\ldots,w_{t-1})
\]

Causal mask ensures $j \leq i$ only. Natural autoregressive generation.

\medskip
\begin{tabular}{lll}
\toprule
\textbf{Task} & \textbf{Architecture} & \textbf{Why} \\
\midrule
Sentiment classification    & Encoder (BERT) & Full bidirectional context \\
NER in filings              & Encoder (BERT) & Token-level labels \\
Earnings call summarisation & Enc--Dec (T5)  & Seq2seq; full input read \\
Report drafting, Q\&A       & Decoder (GPT)  & Generative by design \\
Numerical reasoning (FinQA) & Decoder + CoT  & Multi-step generation \\
\bottomrule
\end{tabular}

\medskip
\begin{alertblock}{Reasoning models (o1, o3, DeepSeek-R1)}
Trained via RL with verifiable rewards to produce chain-of-thought traces.
Same decoder architecture; different training regime. Outperform standard
generation on FinQA-type tasks; do not eliminate hallucination.
\end{alertblock}
\end{frame}

% --- Frame 14: Sampling Strategies ---
\begin{frame}
\frametitle{Sampling Strategies: Temperature, Nucleus, Beam}

\textbf{Temperature scaling:}
\[
p_i(\tau) = \frac{\exp(z_i/\tau)}{\sum_j \exp(z_j/\tau)}
\]
$\tau \to 0$: greedy (deterministic). \quad $\tau = 1$: standard. \quad $\tau \to \infty$: uniform.

\medskip
\textbf{Nucleus (top-$p$) sampling:}
\[
\mathcal{V}_p = \arg\min_S \!\left\{|S| : \sum_{i \in S} p_i(\tau) \geq p\right\}
\]
Adapts vocabulary size to model confidence. Common defaults: $p = 0.9$--$0.95$.

\medskip
\textbf{Beam search:} keep $B$ best partial sequences at each step; expand all.
\begin{itemize}
  \item Pros: higher log-probability sequences; good for ROUGE/BLEU tasks
  \item Cons: beam search degeneration for open-ended generation
\end{itemize}
\end{frame}

% --- Frame 15: Finance Sampling Guidance ---
\begin{frame}
\frametitle{Sampling Strategy: Finance-Specific Guidance}

\small
\begin{tabular}{lll}
\toprule
\textbf{Task} & \textbf{Temperature} & \textbf{Sampling} \\
\midrule
Structured data extraction (earnings, covenants) & $\tau = 0$ (greedy)   & --- \\
Sentiment labelling                               & $\tau \approx 0.1$    & greedy / top-$k$ \\
Document summarisation                            & $\tau \approx 0.3$--$0.5$ & top-$p = 0.9$ \\
Regulatory Q\&A / compliance                     & $\tau \approx 0.1$    & greedy + RAG \\
Scenario / stress-test narrative                  & $\tau \approx 0.7$--$1.0$ & top-$p = 0.95$ \\
\bottomrule
\end{tabular}

\normalsize
\medskip
\begin{block}{Rule of thumb}
If the output feeds a downstream system or will be compared to ground truth: $\tau = 0$.\\
If a human analyst values breadth and novelty: $\tau \in [0.5, 1.0]$ with nucleus.
\end{block}

\medskip
\textbf{Self-consistency:} run $K$ samples at $\tau > 0$; take majority vote.
Especially effective for multi-step numerical reasoning (FinQA-type tasks).
\end{frame}

% --- Frame 16: Structured Generation ---
\begin{frame}[fragile]
\frametitle{Structured Generation: Validity Mask and Pydantic}

\textbf{Validity mask:}
\[
M_t[i] = \mathbf{1}\!\left[\text{token } i \text{ can extend } x_{<t}
\text{ to a string in } \mathcal{L}(\mathcal{G})\right]
\]
Invalid tokens receive $-\infty$ log-probability \emph{before} softmax.

\medskip
\textbf{Anthropic tool use + Pydantic:}

\begin{verbatim}
class EarningsReport(BaseModel):
    revenue_bn: Optional[float] = None
    net_income_bn: Optional[float] = None
    eps: Optional[float] = None

client.messages.create(
  tools=[{"name": "record_earnings",
          "input_schema": EarningsReport.model_json_schema()}],
  tool_choice={"type": "tool", "name": "record_earnings"},
  ...
)
\end{verbatim}

\textbf{Three failure modes even with constrained decoding:}
\begin{enumerate}
  \item Numerically wrong-but-valid output (pair with $\tau=0$ + RAG)
  \item Optional fields $\to$ explicit \texttt{null} (detectable)
  \item Unit confusion $\to$ application-level validation required
\end{enumerate}
\end{frame}

% --- Frame 17: API — BPE + Cost Model ---
\begin{frame}
\frametitle{API Mechanics: BPE Tokenisation and Cost Model}

\textbf{BPE tokenisation:} iteratively merge the most frequent adjacent symbol pair.
\begin{itemize}
  \item Vocabulary: 32k--100k sub-words
  \item Rule of thumb: 1 English word $\approx$ 1.3 tokens
  \item ``EBITDA'' $\to$ \texttt{["EB", "IT", "DA"]}; rare tickers get more fragments
\end{itemize}

\medskip
\textbf{Cost model:}
\[
\text{cost} = N_{\text{in}} \cdot c_{\text{in}} + N_{\text{out}} \cdot c_{\text{out}}
\]

\small
\begin{tabular}{lll}
\toprule
\textbf{Model} & \textbf{Input (\$/MTok)} & \textbf{Output (\$/MTok)} \\
\midrule
GPT-4o               & \$5.00 & \$15.00 \\
Claude 3 Haiku       & \$0.25 & \$1.25  \\
Llama 3 self-hosted  & $\approx$\$0 & $\approx$\$0 \\
\bottomrule
\end{tabular}

\normalsize
\medskip
\textbf{Worked example:} 10\,k transcripts $\times$ 7\,k input tokens + 400 output tokens:
\[
\text{cost}_{\text{GPT-4o}} = 10{,}000 \times (0.035 + 0.006) = \$410
\]
Same workload on Claude Haiku: $\approx$\$20 (20$\times$ cheaper).

\medskip
\textbf{Prompt caching:} cache shared system prompt prefix $\to$ up to 90\% cost reduction.
\end{frame}

% --- Frame 18: RAG Pipeline ---
\begin{frame}
\frametitle{RAG Pipeline: Three-Stage Definition}

\textbf{Retrieval-Augmented Generation (RAG):}
\begin{enumerate}
  \item \textbf{Retrieval:} score passages against query $q$; return top-$k$
        passages $\mathcal{R}(q)$
  \item \textbf{Augmentation:} construct augmented prompt
        $\tilde{q} = [d_{(1)},\ldots,d_{(k)},\,q]$
  \item \textbf{Generation:} $a \sim P_\theta(\,\cdot \mid \tilde{q})$
\end{enumerate}

\medskip
\textbf{Why RAG for finance?}
\begin{itemize}
  \item Knowledge base updated by adding new filings — no retraining
  \item Every factual claim traceable to a retrieved passage (auditability)
  \item Dominant architecture for document Q\&A in production financial systems
\end{itemize}

\medskip
\begin{alertblock}{FinanceBench lesson (Zhang et al., 2024)}
GPT-4-Turbo with a retrieval system \emph{incorrectly answers or refuses} 81\% of
questions from SEC filings. Naive RAG is insufficient for high-stakes financial QA.
\end{alertblock}
\end{frame}

% --- Frame 19: Dense vs. Sparse vs. Hybrid ---
\begin{frame}
\frametitle{Dense, Sparse, and Hybrid Retrieval}

\textbf{Dense retrieval:}
\[
s_{\text{dense}}(q, d_i) = \frac{\phi(q)\cdot\phi(d_i)}{\|\phi(q)\|\|\phi(d_i)\|}
\]
SBERT bi-encoder; FAISS / HNSW for ANN search.

\medskip
\textbf{BM25 (sparse lexical):}
Strong on exact matches (CUSIP numbers, filing section references, tickers).

\medskip
\textbf{Hybrid via Reciprocal Rank Fusion:}
\[
\text{RRF}(d) = \sum_{r \in \{\text{dense},\,\text{sparse}\}}
\frac{1}{60 + \text{rank}_r(d)}
\]
Robust to score-scale differences; no learned weighting parameter.

\medskip
\begin{tabular}{lll}
\toprule
\textbf{Method} & \textbf{Strength} & \textbf{Weakness} \\
\midrule
Dense   & Semantic similarity & Exact-match failures \\
BM25    & Exact-match recall  & Misses paraphrases \\
Hybrid  & Both               & Slightly slower \\
\midrule
Re-ranker & High precision    & High latency \\
\bottomrule
\end{tabular}
\end{frame}

% --- Frame 20: Hallucination Taxonomy ---
\begin{frame}
\frametitle{Hallucination Taxonomy and Detection}

\textbf{Definition:} output that is fluent, confident, and not supported by the input
or the external world.

\medskip
\textbf{Three sub-types for finance:}
\begin{enumerate}
  \item \textcolor{red}{Numerical hallucination:} D/E ratio 1.8 $\to$ reported as 0.8
  \item \textcolor{orange}{Entity hallucination:} wrong company / regulatory body
  \item \textcolor{blue}{Citation hallucination:} fabricated regulation, ruling, or standard
\end{enumerate}

\medskip
\textbf{Detection methods:}
\begin{itemize}
  \item \textbf{SelfCheckGPT:} draw $N$ samples; hallucinated claims are inconsistent across samples
  \item \textbf{FActScore:} decompose into atomic claims; verify each against retrieval
  \item \textbf{Inner confidence} (Chen et al., 2024): high-confidence LLM signals
        achieve Sharpe +20\% vs.\ unconditional benchmark
\end{itemize}

\begin{block}{Root cause}
Models are trained to produce plausible text. $\tau = 0$ removes stochasticity
but does \textbf{not} eliminate hallucination --- the most probable completion may still be wrong.
\end{block}
\end{frame}

% --- Frame 21: Mitigation Stack ---
\begin{frame}
\frametitle{Hallucination Mitigation: Five-Layer Stack}

These techniques are complementary --- use them together in production.

\medskip
\begin{enumerate}
  \item \textbf{Retrieval-augmented generation (RAG):} ground context in verified
        passages; model almost never fabricates figures absent from retrieved text
  \item \textbf{Self-consistency sampling:} $K$ independent chains; majority-vote
        answer. Best for multi-step numerical reasoning
  \item \textbf{Chain-of-thought prompting:} require explicit intermediate steps;
        each step is independently verifiable
  \item \textbf{Claim decomposition (FActScore):} break output into atomic claims;
        flag unsupported / contradicted claims before human review
  \item \textbf{Confidence elicitation + abstention:} prompt the model to flag
        uncertainty; calibrate thresholds against accuracy curves
\end{enumerate}

\medskip
\begin{alertblock}{Reasoning models}
Reduced hallucination on tasks with verifiable intermediate steps (FinQA).
But extrinsic hallucination on obscure regulatory facts persists.
All five techniques remain necessary.
\end{alertblock}
\end{frame}

\begin{frame}
\frametitle{Responsible Use: Regulation, Look-Ahead Bias, GDPR}

\textbf{EU AI Act (Regulation 2024/1689, in force August 2024):}
\begin{itemize}
  \item High-risk categories: credit scoring, insurance pricing, AI-assisted securities pricing
  \item Requirements: conformity assessments, human oversight, transparency to regulators
\end{itemize}

\medskip
\textbf{Look-ahead bias (Didisheim et al., 2025):}
\begin{itemize}
  \item LLMs can reconstruct historical aggregate time series from parametric memory
  \item Produces spurious return predictability in back-tests
  \item Any information the model might have memorised is contaminated data
\end{itemize}

\medskip
\textbf{GDPR:} fine-tuning or RAG on client personal data triggers right-to-erasure
obligations. A model trained on personal data cannot satisfy erasure without retraining.

\medskip
\textbf{Geographic bias:} all five major benchmarks are North American / English.
Sentiment models trained on US earnings calls may systematically mis-classify
sentiment in Japanese, Chinese, or Indian financial contexts.
\end{frame}
