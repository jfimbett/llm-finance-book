% ============================================================
% FRAME 2: Financial data underrepresentation
% ============================================================
\begin{frame}
\frametitle{Financial Data: A Tiny Slice of the Web}

\begin{bigpicture}
  Financial text constitutes roughly \textbf{0.5\%} of a typical filtered web crawl.
  A model that has seen the distribution of the web will be fluent in casual prose
  and weak on financial contracts.
\end{bigpicture}

\medskip
\textbf{Four key financial text sources:}

\begin{enumerate}
  \item \textbf{EDGAR} --- 10-K, 10-Q, 8-K, proxy filings; full regulatory English;
        public from 1996 to present
  \item \textbf{Earnings-call transcripts} --- spoken-register financial language,
        Q\&A with management; available via commercial vendors or scraping
\end{enumerate}
\end{frame}

% ============================================================
% FRAME 2b: Financial data sources (continued)
% ============================================================
\begin{frame}
\frametitle{Financial Data Sources (continued)}

\begin{enumerate}
  \item[3.] \textbf{Bloomberg news} --- licensed; high-density, entity-linked financial
        news with multi-decade coverage
  \item[4.] \textbf{Alternative data} --- satellite-imagery metadata, credit-card
        transaction summaries, job-posting feeds, shipping manifests
\end{enumerate}

\medskip
\begin{alertblock}{Consequence}
  Low financial representation $\Rightarrow$ high tokeniser fertility on financial
  text $\Rightarrow$ longer sequences $\Rightarrow$ higher cost and lower accuracy.
\end{alertblock}
\end{frame}

% ============================================================
% FRAME 3: BPE + fertility
% ============================================================
\begin{frame}
\frametitle{Tokenisation: BPE and Fertility}

Byte-pair encoding (BPE) iteratively merges the most frequent adjacent symbol pair
until the target vocabulary size is reached.

\medskip
\textbf{Fertility formula:}
\[
  F(\tau, C) = \frac{\text{tokens produced by } \tau \text{ on corpus } C}
                    {\text{whitespace-separated words in } C}
\]

\medskip
Fertility $\approx 1.0$ means efficient coverage; fertility $> 1.5$ means heavy
fragmentation of domain vocabulary.

\medskip
\begin{center}
\begin{tabular}{lcc}
\toprule
\textbf{Text type} & \textbf{Fertility (GPT-2 BPE)} & \textbf{Effect} \\
\midrule
Standard English prose  & 1.10--1.20 & Efficient \\
Financial regulatory text & 1.40--1.60 & Fragmented \\
\bottomrule
\end{tabular}
\end{center}

\medskip
Words like \texttt{CUSIP}, \texttt{EBITDA}, and \texttt{ISDA} may be split into
3--5 sub-tokens by a general BPE tokeniser, each consuming an attention slot.
Training a domain-specific tokeniser alongside domain-specific weights reduces
fertility and shortens effective sequence lengths.
\end{frame}

% ============================================================
% FRAME 4: Quality pipeline + MinHash
% ============================================================
\begin{frame}
\frametitle{Data Quality Pipeline}

Raw web crawl data is $\approx 90\%$ noise. A standard pipeline applies four stages:

\medskip
\begin{enumerate}
  \item \textbf{Language identification} --- fastText classifier assigns each document
        a language label; non-target languages discarded
  \item \textbf{Heuristic filters} --- punctuation rate, symbol-to-word ratio,
        minimum token count, repeated n-gram detection
  \item \textbf{Perplexity filtering} --- small KenLM reference model scores each
        document; high-perplexity outliers removed
  \item \textbf{MinHash deduplication + PII scrubbing} --- near-duplicate removal
        by Jaccard similarity; removal of emails, phone numbers, SSNs
\end{enumerate}
\end{frame}

% ============================================================
% FRAME 4b: MinHash deduplication (continued)
% ============================================================
\begin{frame}
\frametitle{Data Quality: MinHash Deduplication}

\textbf{MinHash: Jaccard similarity definition}
\[
  J(A, B) = \frac{|S(A) \cap S(B)|}{|S(A) \cup S(B)|}
\]
where $S(d)$ is the set of $k$-gram shingles of document $d$.
Documents with $J > 0.8$ are near-duplicates; all but one copy is removed.

\medskip
\begin{alertblock}{Why deduplication matters most}
  Duplicated training data causes memorisation, not generalisation.
  Financial press releases are reproduced verbatim by hundreds of portals.
\end{alertblock}
\end{frame}

% ============================================================
% FRAME 5: CLM / MLM / span-corruption objectives
% ============================================================
\begin{frame}
\frametitle{Pre-training Objectives: CLM, MLM, Span Corruption}

\begin{center}
{\small
\begin{tabular}{p{2.4cm}p{3.8cm}p{3.4cm}}
\toprule
\textbf{Objective} & \textbf{Loss} & \textbf{Best suited for} \\
\midrule
CLM (causal LM) &
  $-\displaystyle\sum_t \log P(x_t \mid x_{<t})$ &
  Generation, chat, decoder-only (GPT) \\[6pt]
MLM (masked LM) &
  $-\displaystyle\sum_{t\in\mathcal{M}} \log P(x_t \mid x_{\setminus\mathcal{M}})$ &
  Classification, NER, encoder-only (BERT, FinBERT) \\[6pt]
Span corruption (T5) &
  Reconstruct masked spans from sentinels &
  Seq2seq, QA, summarisation \\
\bottomrule
\end{tabular}
}
\end{center}

\medskip
CLM uses causal (lower-triangular) attention masking.
MLM attends \emph{bidirectionally} to all unmasked tokens.
Span corruption replaces runs of consecutive tokens with a single sentinel
\texttt{<extra\_id\_k>} and trains the decoder to produce all spans in order.

\medskip
\begin{block}{Finance rule of thumb}
  Classification / extraction tasks $\rightarrow$ MLM encoder (FinBERT). \\
  Open-ended analysis / generation $\rightarrow$ CLM decoder.
\end{block}
\end{frame}

% ============================================================
% FRAME 6: Chinchilla setup
% ============================================================
\begin{frame}
\frametitle{Chinchilla Scaling Laws: Setup}

Hoffmann et al.\ (2022) fit the empirical loss of an autoregressive LM as:
\[
  L(N, D) = E + \frac{A}{N^{\alpha}} + \frac{B}{D^{\beta}}
\]

\begin{itemize}
  \item $N$ = number of trainable parameters
  \item $D$ = number of training tokens
  \item $E$ = irreducible entropy floor
  \item $\alpha, \beta > 0$ govern the decay rate of each term
\end{itemize}

\medskip
\textbf{Compute constraint:}
\[
  C \approx 6ND
\]
(factor 6: 2 for forward pass, 4 for backward pass and parameter update,
standard mixed-precision accounting)

\medskip
\begin{block}{The question}
  Given a fixed FLOP budget $C$, what is the optimal split $(N^*, D^*)$?
\end{block}
\end{frame}

% ============================================================
% FRAME 7: Chinchilla result
% ============================================================
\begin{frame}
\frametitle{Chinchilla Scaling Laws: Result}

Minimising $L(N,D)$ subject to $C = 6ND$ gives:
\[
  N^*(C) \propto C^{\beta/(\alpha+\beta)}, \qquad
  D^*(C) \propto C^{\alpha/(\alpha+\beta)}
\]

With Chinchilla estimates $\alpha \approx 0.34$, $\beta \approx 0.28$:
\[
  \boxed{\frac{D^*}{N^*} \approx 20}
\]

\medskip
\textbf{Worked example: 7B-parameter model}

\medskip
\begin{center}
\begin{tabular}{lcc}
\toprule
 & GPT-3 (175B) & Chinchilla-optimal 7B \\
\midrule
Parameters         & 175B  & 7B \\
Training tokens    & 300B  & 140B \\
$D/N$ ratio        & 1.7   & 20 \\
Status             & Over-parameterised & Optimal \\
\bottomrule
\end{tabular}
\end{center}

\medskip
\begin{alertblock}{Practical implication}
  A smaller model trained on more data often \emph{outperforms} a larger
  model trained on less. Inference cost favours the smaller model.
\end{alertblock}
\end{frame}

% ============================================================
% FRAME 10b: Adaptation tax (continued)
% ============================================================
\begin{frame}
\frametitle{The Adaptation Spectrum: The Adaptation Tax}

\begin{bigpicture}
  \textbf{Adaptation tax.} Full fine-tuning on a narrow domain can degrade performance
  on adjacent tasks --- a real risk when a deployed finance LLM must also handle general
  client queries. The art of adaptation is buying in-domain accuracy without
  surrendering the breadth that made the base model useful.
\end{bigpicture}
\end{frame}

% ============================================================
% FRAME 11: LoRA
% ============================================================
\begin{frame}
\frametitle{LoRA: Low-Rank Adaptation}

\begin{underhood}
For a pre-trained weight matrix $W_0 \in \mathbb{R}^{d\times k}$, LoRA sets:
\[
  W = W_0 + BA, \quad B \in \mathbb{R}^{d\times r},\; A \in \mathbb{R}^{r\times k},\;
  r \ll \min(d,k)
\]
$W_0$ is frozen. Only $B$ and $A$ are trained.
Trainable parameters per layer: $r(d + k)$.
\end{underhood}

\medskip
\textbf{Worked example: 7B LLaMA, $r = 8$, all 4 projections, 32 layers}

\[
  32 \times 4 \times 8 \times (4096 + 4096) = 8{,}388{,}608 \approx 8.4\text{M}
\]
\[
  \frac{8.4\text{M}}{7{,}000\text{M}} \approx 0.12\%
\]
\end{frame}

% ============================================================
% FRAME 11b: LoRA vs full fine-tuning (continued)
% ============================================================
\begin{frame}
\frametitle{LoRA vs.\ Full Fine-Tuning}

\begin{center}
\begin{tabular}{lcc}
\toprule
                & LoRA ($r=8$) & Full fine-tuning \\
\midrule
Trainable params & 8.4M (0.12\%) & 7,000M (100\%) \\
A100 training time & $\approx 90$ min & $\approx 40$ hr \\
Inference overhead & Zero (merge $BA$) & None \\
\bottomrule
\end{tabular}
\end{center}
\end{frame}

% ============================================================
% FRAME 13: RLHF 3-stage pipeline
% ============================================================
\begin{frame}
\frametitle{RLHF: Three-Stage Alignment Pipeline}

\begin{bigpicture}
  A base model predicts the next token; it does not know what humans \emph{prefer}.
  RLHF aligns behaviour to human judgement --- in finance, the gap between a
  ``plausible'' answer and an ``accurate, compliant'' one is exactly what alignment
  must close.
\end{bigpicture}

\medskip
\textbf{Stage 1 --- Supervised Fine-Tuning (SFT):}
Train base model on high-quality human-written demonstrations.
Produces initial policy $\pi_{\text{SFT}}$.

\medskip
\textbf{Stage 2 --- Reward Model Training:}
Human annotators rank pairs of responses $(y_w \succ y_l)$ for the same prompt $x$.
Bradley-Terry loss:
\[
  \mathcal{L}_{\text{RM}} = -\mathbb{E}\!\left[\log \sigma\!\left(r_\theta(x,y_w) - r_\theta(x,y_l)\right)\right]
\]
\end{frame}

% ============================================================
% FRAME 13b: RLHF Stage 3 (continued)
% ============================================================
\begin{frame}
\frametitle{RLHF: Stage 3 --- PPO Fine-Tuning}

\begin{underhood}
\textbf{Stage 3 --- PPO fine-tuning with KL penalty:}
\[
  \mathcal{L}_{\text{RL}}(\theta) = \mathbb{E}_{\pi_\theta}\!\left[r_\phi(x,y)
  - \beta\,\mathrm{KL}\!\left(\pi_\theta(\cdot|x) \,\|\, \pi_{\text{ref}}(\cdot|x)\right)\right]
\]
$\beta$ controls how far the policy can drift from the SFT reference.
\end{underhood}

\medskip
\begin{alertblock}{Limitations for finance}
  Reward models can be gamed; PPO is unstable; human annotation is expensive.
  A single annotator disagreement on ``plausible vs.\ accurate'' in a financial
  context can propagate to thousands of training examples.
\end{alertblock}
\end{frame}

% ============================================================
% FRAME 14: DPO
% ============================================================
\begin{frame}
\frametitle{Direct Preference Optimisation (DPO)}

\begin{underhood}
DPO eliminates the reward model entirely. It directly optimises the policy
on preference pairs by re-parameterising the reward in terms of log-ratios:

\[
  \mathcal{L}_{\text{DPO}}(\theta) = -\mathbb{E}_{(x,y_w,y_l)}\!\left[
    \log \sigma\!\left(
      \beta \log \frac{\pi_\theta(y_w|x)}{\pi_{\text{ref}}(y_w|x)}
      - \beta \log \frac{\pi_\theta(y_l|x)}{\pi_{\text{ref}}(y_l|x)}
    \right)
  \right]
\]
\end{underhood}
\end{frame}

% ============================================================
% FRAME 14b: DPO properties (continued)
% ============================================================
\begin{frame}
\frametitle{Direct Preference Optimisation (DPO): Properties}

\textbf{Key properties:}
\begin{itemize}
  \item No separate reward model to train or maintain
  \item Same preference data as RLHF; simpler single-stage training loop
  \item Empirically competitive with RLHF on alignment benchmarks
  \item $\beta$ still controls implicit KL penalty vs.\ $\pi_{\text{ref}}$
\end{itemize}

\medskip
\begin{block}{Interpretation}
  DPO increases the log-probability of the preferred response $y_w$ relative
  to $\pi_{\text{ref}}$ while decreasing the log-probability of $y_l$ ---
  a direct implementation of the preference signal.
\end{block}
\end{frame}

% FRAME 15: FinBERT
% ============================================================
\begin{frame}
\frametitle{FinBERT: Domain-Adapted Encoder for Finance}

\textbf{Architecture:} BERT-large continued pre-training on $\approx 4.9$B tokens
of Reuters news, Financial PhraseBank, and analyst reports (MLM objective).

\medskip
\textbf{Classification head:}
\[
  \hat{y} = \mathrm{softmax}(W_c \cdot h_{[\text{CLS}]} + b_c)
\]
$h_{[\text{CLS}]}$ is the final-layer representation of the \texttt{[CLS]} token.

\medskip
\textbf{Empirical results on Financial PhraseBank (all-agree split):}

\medskip
\begin{center}
\begin{tabular}{lc}
\toprule
\textbf{Model} & \textbf{Accuracy} \\
\midrule
Human upper bound & $\approx 97\%$ \\
FinBERT           & 88.5\% \\
BERT-large (general) & 80.7\% \\
Dictionary / rule-based & $\approx 72\%$ \\
\bottomrule
\end{tabular}
\end{center}

\medskip
The 7.8 pp gain over BERT-large comes almost entirely from domain-specific
continued pre-training, not architectural changes.
\end{frame}

% FRAME 16: BloombergGPT
% ============================================================
\begin{frame}
\frametitle{BloombergGPT: A 50B Finance Decoder}

\textbf{Training corpus:} 363B tokens total --- 51\% Bloomberg financial data
(news, filings, transcripts), 49\% general web and books.

\medskip
\textbf{ALiBi positional encoding:}
\[
  \text{score}(q_i, k_j) = \frac{q_i^\top k_j}{\sqrt{d_k}} - m\,(i - j)
\]
where $m$ is a head-specific slope and $(i-j)$ is query-key distance.
ALiBi extrapolates more gracefully to sequences longer than those seen in training
compared to learned absolute positional embeddings.

\medskip
\textbf{Chinchilla lens on BloombergGPT:}

\begin{center}
\begin{tabular}{lcc}
\toprule
 & BloombergGPT & Chinchilla optimal \\
\midrule
Parameters     & 50B   & 50B \\
Training tokens & 363B & $\approx 1$T (= $50\text{B} \times 20$) \\
Status         & \multicolumn{2}{c}{Under-trained by Chinchilla criterion} \\
\bottomrule
\end{tabular}
\end{center}

\medskip
Lesson: even a well-resourced organisation trained a model that violated the
compute-optimal scaling prescription.
\end{frame}

% ============================================================
% FRAME 17: Evaluation metrics
% ============================================================
\begin{frame}
\frametitle{Evaluation: Perplexity, Macro-F1, ROUGE}

\textbf{Perplexity} (language modelling):
\[
  \text{PPL} = 2^{H}, \qquad H = -\frac{1}{T}\sum_{t=1}^{T} \log_2 p(x_t \mid x_{<t})
\]
Three limitations: not a task metric; gameable by spreading mass; requires
token-level probabilities.

\medskip
\textbf{Macro-F1} (classification):
\[
  \text{Macro-F1} = \frac{1}{K}\sum_{k=1}^{K} \frac{2\,P_k R_k}{P_k + R_k}
\]
Treats all classes equally regardless of frequency. Appropriate for imbalanced
financial datasets (rare event classes matter as much as common ones).
\end{frame}

% ============================================================
% FRAME 17b: Evaluation -- ROUGE + FinBen (continued)
% ============================================================
\begin{frame}
\frametitle{Evaluation: ROUGE and the FinBen Benchmark}

\textbf{ROUGE-N} (summarisation):
\[
  \text{ROUGE-N} = \frac{\displaystyle\sum_{\text{ref}}\sum_{g \in \text{ngrams(ref},N)} \text{count}_\text{match}(g)}
                        {\displaystyle\sum_{\text{ref}}\sum_{g \in \text{ngrams(ref},N)} \text{count}(g)}
\]

\medskip
\begin{block}{FinBen benchmark}
  42 datasets, 24 tasks, 21 models. LLMs excel at extraction/classification but
  struggle on forecasting and decision tasks --- exactly what practitioners want most.
\end{block}
\end{frame}

% ============================================================
% FRAME 18: Red-teaming + Lopez-Lira
% ============================================================
\begin{frame}
\frametitle{Red-Teaming: Financial Failure Modes}

\textbf{Four recurring failure modes identified by red-teaming:}

\begin{enumerate}
  \item \textbf{Factual hallucination} --- generates plausible but incorrect price
        forecasts or earnings figures
  \item \textbf{Temporal hallucination} --- cites outdated regulations as current
        (e.g., pre-2018 MiFID II rules)
  \item \textbf{Regulatory compliance failure} --- provides what appears to be
        personalised investment advice without required licensing disclosures
  \item \textbf{Herd-bias amplification} --- echoes analyst consensus while
        suppressing dissenting signals
\end{enumerate}

\medskip
\begin{block}{Empirical illustration: Lopez-Lira \& Tang (2023)}
  A long-short strategy based on ChatGPT sentiment scores achieved a Sharpe ratio
  $> 6$ in 2021, declining to $\approx 1$ by late 2024 as the signal became widely
  known and arbitraged away.
  \textit{Any published alpha source degrades once it is public.}
\end{block}
\end{frame}

\section{Safety and Governance}

% ============================================================
% FRAME 19: Bias types
% ============================================================
\begin{frame}
\frametitle{Four Bias Types in Financial LLMs}

\begin{alertblock}{Survivorship bias}
  Training corpora overrepresent successful firms: public companies file EDGAR;
  bankrupt or never-public firms do not. The model learns a rosier world than
  the one practitioners inhabit.
\end{alertblock}

\medskip
\begin{block}{Temporal bias}
  Models trained on historical data are deployed in structurally different
  market regimes (post-pandemic, rising rates, de-globalisation). Performance
  observed on historical test sets may not persist.
\end{block}
\end{frame}

% ============================================================
% FRAME 19b: Bias types (continued)
% ============================================================
\begin{frame}
\frametitle{Four Bias Types in Financial LLMs (continued)}

\begin{block}{Geographic bias}
  English-language, US-centric text dominates open web corpora. Non-US financial
  conventions, currencies, and legal structures are underrepresented.
\end{block}

\medskip
\begin{block}{Analyst consensus bias}
  Financial commentary echoes dominant sell-side views; contrarian information
  is rarer in the training distribution and is systematically under-weighted.
\end{block}
\end{frame}

% ============================================================
% FRAME 20: Hallucination + mitigations
% ============================================================
\begin{frame}
\frametitle{Hallucination Types and Mitigations}

\textbf{Two hallucination types:}
\begin{itemize}
  \item \textbf{Factual hallucination} --- fabricated prices, dates, financial figures
        not present in any training document
  \item \textbf{Regulatory hallucination} --- describes rules that are outdated,
        misquoted, or jurisdiction-mismatched
\end{itemize}
\end{frame}

% ============================================================
% FRAME 20b: Hallucination mitigations (continued)
% ============================================================
\begin{frame}
\frametitle{Hallucination Mitigations}

\textbf{Four principal mitigations:}

\begin{enumerate}
  \item \textbf{Retrieval-Augmented Generation (RAG)} --- ground outputs in
        retrieved, verified passages; reduces reliance on parametric memory
  \item \textbf{Confidence calibration} --- train model to express uncertainty;
        reject or flag responses below a calibrated confidence threshold
  \item \textbf{Output verification} --- secondary model or rule-based checker
        flags claims inconsistent with a structured knowledge base
  \item \textbf{Human-in-the-loop} --- route low-confidence or high-stakes outputs
        to expert review before delivery to end users
\end{enumerate}

\medskip
\begin{alertblock}{No single mitigation is sufficient}
  Production deployments should layer at least two mitigations,
  with audit trails at every step.
\end{alertblock}
\end{frame}

% FRAME 21: Regulatory framework
% ============================================================
\begin{frame}
\frametitle{Regulatory Landscape: MiFID II, SR 11-7, EU AI Act}

\textbf{MiFID II Article 37:}
\begin{itemize}
  \item Investment research must be independent from commercial pressures
  \item LLM contributions require demonstrable \emph{suitability} (analysis
        matches recipient risk profile) and \emph{explainability}
  \item Attention weight visualisations are \emph{not sufficient} explanations:
        they reflect co-occurrence patterns, not causal reasoning chains
\end{itemize}

\medskip
\textbf{SR 11-7 (Federal Reserve Model Risk Management):}
\begin{itemize}
  \item Tiered validation proportional to model risk
  \item High-risk models (credit, capital, trading) require independent
        validation, assumption documentation, periodic performance monitoring
\end{itemize}

\medskip
\textbf{EU AI Act --- August 2026 deadline for high-risk systems:}
\begin{itemize}
  \item Conformity assessments, technical documentation, human oversight
  \item Registration in EU database for high-risk financial AI applications
  \item High-risk: models influencing credit scoring, insurance pricing,
        investment decisions at scale
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Governance: Four-Component Framework}

\textbf{Four-component governance framework:}
\begin{enumerate}
  \item \textbf{Risk classification} --- assign Low/Medium/High to each use case
        based on decision autonomy and downstream impact
  \item \textbf{Audit trails} --- capture input prompts, retrieved context,
        model outputs, and user actions immutably
  \item \textbf{Monitoring} --- track output distributions for drift; alert on
        anomalous response patterns or accuracy degradation
  \item \textbf{Incident response} --- documented process for rapid mitigation
        when model failures are detected in production
\end{enumerate}
\end{frame}

\begin{frame}
\frametitle{Lecture 3 Wrap-Up: Key Takeaways}

\textbf{Data:} Financial text is 0.5\% of the web. EDGAR, transcripts, and Bloomberg
are the primary sources. Quality pipelines and MinHash deduplication are essential.

\medskip
\textbf{Scaling:} Chinchilla says $D^*/N^* \approx 20$. GPT-3 was over-parameterised;
BloombergGPT was under-trained. Size alone is not the answer.

\medskip
\textbf{Fine-tuning:} LoRA achieves 0.12\% of 7B parameters for 90 min of training
vs.\ 40 hr full fine-tuning, with zero inference overhead.
\end{frame}

% ============================================================
% FRAME 23b: Wrap-up (continued)
% ============================================================
\begin{frame}
\frametitle{Lecture 3 Wrap-Up: Key Takeaways (continued)}

\textbf{Alignment:} RLHF (SFT $\to$ reward model $\to$ PPO) aligns behaviour.
DPO simplifies it to a single supervised objective on preference pairs.

\medskip
\textbf{Finance:} FinBERT: +7.8 pp over BERT on sentiment. BloombergGPT: under-trained
per Chinchilla. FinBen: LLMs still struggle on forecasting.

\medskip
\textbf{Governance:} MiFID II + SR 11-7 + EU AI Act 2026 require explainability,
audit trails, and tiered validation. Attention weights are not explanations.

\medskip

% ============================================================
\begin{frame}[noframenumbering]{Engineering Efficiency: bfloat16, Checkpointing, FlashAttention}

\textbf{Mixed-precision training (bfloat16):}
\begin{itemize}
  \item Weights stored in bfloat16 (16-bit); fp32 master copy for optimiser
  \item Halves memory and HBM bandwidth versus fp32
  \item Loss scaling: multiply loss by $s$ before backward; divide gradients by $s$
        to prevent underflow
\end{itemize}

\medskip
\textbf{Gradient checkpointing:}
\[
  \text{Activation memory} = O(L) \xrightarrow{\text{checkpoint every } \sqrt{L}} O(\sqrt{L})
\]
Discards intermediate activations; recomputes them during backward pass.
Cost: one extra forward pass.
\end{frame}

% ============================================================
% APPENDIX FRAME: FlashAttention (continued)
% ============================================================
\begin{frame}[noframenumbering]{Engineering Efficiency: FlashAttention}

\textbf{FlashAttention:}
Tiles $QK^\top$ into SRAM-resident blocks; avoids writing the full $N\times N$
attention matrix to HBM.  Reduces HBM reads/writes from $O(N^2)$ to
$O(N^2 / M) \cdot M$ (subquadratic in practice).
Exact attention, no approximation.

\medskip
\begin{block}{Combined effect}
  bfloat16 + checkpointing + FlashAttention enable training models 4--8$\times$
  larger on the same hardware compared with naive full-precision training.
\end{block}
\end{frame}

\begin{frame}[noframenumbering]{Prefix Tuning and Adapters}

\textbf{Prefix tuning:}

Learnable prefix vectors are prepended to $K$ and $V$ at every layer:
\[
  K' = \begin{bmatrix} P_K \\ K \end{bmatrix} \in \mathbb{R}^{(l+n)\times d_k},
  \qquad
  V' = \begin{bmatrix} P_V \\ V \end{bmatrix} \in \mathbb{R}^{(l+n)\times d_v}
\]
Only the prefix matrices $P_K, P_V$ (of length $l$) are trained.
Effective for generation tasks; steers the output distribution without touching weights.

\medskip
\textbf{Adapters:}

A bottleneck module inserted after each sub-layer:
\[
  h \leftarrow h + W_{\text{up}}\,\text{ReLU}(W_{\text{down}}\,h),
  \quad W_{\text{down}} \in \mathbb{R}^{d\times m},\; W_{\text{up}} \in \mathbb{R}^{m\times d},\; m \ll d
\]
Adds inference latency unless fused.

\medskip
\begin{block}{Why LoRA wins in practice}
  Zero inference overhead (merge $W_0 + BA$ before serving);
  multiple adapters swappable for different tasks;
  QLoRA extends to 4-bit quantised base models.
\end{block}
\end{frame}
